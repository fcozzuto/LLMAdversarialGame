def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    me = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    mx, my = int(me[0]), int(me[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def dcheb(a, b):
        dx = abs(a[0] - b[0])
        dy = abs(a[1] - b[1])
        return dx if dx > dy else dy

    # If no resources, head toward center.
    if not resources:
        cx, cy = (w - 1) // 2, (h - 1) // 2
        best = (0, 0)
        bestv = -10**9
        for dx, dy in dirs:
            nx, ny = mx + dx, my + dy
            if not inb(nx, ny):
                continue
            v = -dcheb((nx, ny), (cx, cy))
            if v > bestv:
                bestv = v
                best = (dx, dy)
        return [best[0], best[1]]

    # Choose move that most improves "claiming" of some resource vs opponent.
    best_move = (0, 0)
    best_val = -10**18
    opp_pos = (ox, oy)

    # Deterministic preference order: consistent scoring + tie-break by (dx,dy) order in dirs.
    for dx, dy in dirs:
        nx, ny = mx + dx, my + dy
        if not inb(nx, ny):
            continue
        my_pos = (nx, ny)

        # Evaluate: best advantage over resources (being first or close), plus avoid letting opponent get ahead.
        val = 0
        # Small tie-break toward nearer resources overall.
        nearest_ours = 10**9
        nearest_opp = 10**9

        for r in resources:
            dr_me = dcheb(my_pos, r)
            dr_op = dcheb(opp_pos, r)
            if dr_me < nearest_ours:
                nearest_ours = dr_me
            if dr_op < nearest_opp:
                nearest_opp = dr_op

            # Advantage shaping: big reward if we can reach no later than opponent.
            if dr_me <= dr_op:
                gap = dr_op - dr_me
                val += 50 - 10 * dr_me + 8 * gap
            else:
                gap = dr_me - dr_op
                val -= 40 + 12 * gap + 3 * dr_me

            # Immediate pickup incentive.
            if dr_me == 0:
                val += 2000

        # Penalize being stuck far while opponent is closer somewhere.
        if nearest_opp < nearest_ours:
            val -= 15 * (nearest_ours - nearest_opp)

        # Preference to reduce overall distance (prevents oscillation).
        val += -2 * nearest_ours

        if val > best_val:
            best_val = val
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
