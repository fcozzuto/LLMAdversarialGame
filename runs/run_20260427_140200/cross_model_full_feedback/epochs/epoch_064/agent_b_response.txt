def choose_move(observation):
    turn = int(observation.get("turn_index", 0))
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))

    me = observation.get("self_position", [0, 0])
    opp = observation.get("opponent_position", [0, 0])
    sx, sy = int(me[0]), int(me[1])
    ox, oy = int(opp[0]), int(opp[1])

    obstacles = set()
    for p in observation.get("obstacles", []) or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources", []) or []:
        if r and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def legal(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    def dist_cheb(ax, ay, bx, by):
        dx = abs(ax - bx)
        dy = abs(ay - by)
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1),  (0, 0),  (0, 1),
             (1, -1),  (1, 0),  (1, 1)]

    best = (0, 0)
    best_score = -10**9

    if resources:
        # deterministic: pick closest resource by chebyshev distance
        def score_for(nx, ny, rx, ry):
            d_me = dist_cheb(nx, ny, rx, ry)
            d_opp = dist_cheb(ox, oy, rx, ry)
            # prefer approaching own resource while keeping distance from opponent
            return -d_me + (d_opp * 0.0)

        # pick resource with smallest dist from me; break ties by lower dist to opponent
        target = None
        target_d = 10**9
        target_opp = 10**9
        for (rx, ry) in resources:
            d = dist_cheb(sx, sy, rx, ry)
            dopp = dist_cheb(ox, oy, rx, ry)
            if d < target_d or (d == target_d and dopp < target_opp):
                target = (rx, ry)
                target_d = d
                target_opp = dopp

        tx, ty = target
        best_move = (0,0)
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not legal(nx, ny):
                continue
            # prefer moving towards target
            if dist_cheb(nx, ny, tx, ty) < dist_cheb(sx, sy, tx, ty):
                best_move = (dx, dy)
                best_score = 0
                break
        else:
            best_move = (0,0)
    else:
        # no resources: approach center of grid away from opponent when possible
        cx, cy = w//2, h//2
        best_move = (0,0)
        best_score = -10**9
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not legal(nx, ny):
                continue
            d_to_center = dist_cheb(nx, ny, cx, cy)
            d_opp = dist_cheb(nx, ny, ox, oy)
            score = -d_to_center - d_opp
            if score > best_score:
                best_score = score
                best_move = (dx, dy)

    dx, dy = best_move
    if (sx + dx, sy + dy) == (sx, sy):
        return [0, 0]
    return [dx, dy]

def choose_move(observation):
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))

    me = observation.get("self_position", [0, 0])
    opp = observation.get("opponent_position", [0, 0])
    mx, my = int(me[0]), int(me[1])
    ox, oy = int(opp[0]), int(opp[1])

    obs = set()
    for p in observation.get("obstacles", []) or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obs.add((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1),  (0, 0),  (0, 1),
             (1, -1),  (1, 0),  (1, 1)]

    def legal(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obs

    def cheb(a, b):
        return max(abs(a[0]-b[0]), abs(a[1]-b[1]))

    resources = []
    for r in observation.get("resources", []) or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if legal(rx, ry):
                resources.append((rx, ry))

    target = None
    if resources:
        target = min(resources, key=lambda p: cheb((mx, my), p))

    best = None
    best_score = -10**9

    for dx, dy in moves:
        nx, ny = mx + dx, my + dy
        if not legal(nx, ny):
            continue
        score = 0
        if target is not None:
            score -= cheb((nx, ny), target)
        if (nx, ny) == (ox, oy):
            score += 1
        if (nx, ny) == (mx, my):
            score += 0
        if score > best_score:
            best_score = score
            best = (dx, dy)

    if best is None:
        for dx, dy in moves:
            nx, ny = mx + dx, my + dy
            if legal(nx, ny):
                best = (dx, dy)
                break

    if best is None:
        best = (0, 0)

    return [int(best[0]), int(best[1])]
