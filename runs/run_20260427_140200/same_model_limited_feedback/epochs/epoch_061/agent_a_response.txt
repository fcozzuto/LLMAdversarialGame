def choose_move(observation):
    w=observation["grid_width"]; h=observation["grid_height"]
    sx,sy=observation["self_position"]; ox,oy=observation["opponent_position"]
    obstacles=set((p[0],p[1]) for p in observation.get("obstacles",[]))
    resources=observation.get("resources",[])
    # If standing on a resource, try to stay (or move away if blocked); but engine likely handles scoring.
    moves=[(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    if resources:
        # Pick nearest resource (deterministic tie-break by coordinates)
        best=None
        for rx,ry in resources:
            if (rx,ry) in obstacles: 
                continue
            d=abs(rx-sx)+abs(ry-sy)
            key=(d,rx,ry)
            if best is None or key<best[0]:
                best=(key,(rx,ry))
        tx,ty=best[1]
    else:
        # Fallback: head to center
        tx,ty=(w-1)/2,(h-1)/2
    def clamp_inside(x,y):
        return 0<=x<w and 0<=y<h
    best_move=(0,(0,0),-1)
    for dx,dy in moves:
        nx,ny=sx+dx,sy+dy
        if not clamp_inside(nx,ny): 
            continue
        if (nx,ny) in obstacles:
            continue
        # Distance to target
        if resources:
            dist=abs(tx-nx)+abs(ty-ny)
            dkey=(dist, nx, ny)
        else:
            dist=abs(tx-nx)+abs(ty-ny)
            dkey=(dist, nx, ny)
        # Prefer moves that increase separation from opponent
        sep=abs(nx-ox)+abs(ny-oy)
        # Score smaller is better for target; larger sep is better
        cand=(dkey, -sep)
        if best_move is None or cand<best_move[0]:
            best_move=(cand,(dx,dy),sep)
    return [int(best_move[1][0]), int(best_move[1][1])]

def choose_move(observation):
    w = observation.get("grid_width", 0)
    h = observation.get("grid_height", 0)
    sx, sy = observation.get("self_position", (0, 0))
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []
    occ = set()
    for p in obstacles:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            occ.add((p[0], p[1]))
    moves = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    if resources:
        best = None
        for rx, ry in resources:
            if (rx, ry) in occ:
                continue
            d = abs(rx - sx) + abs(ry - sy)
            key = (d, rx, ry)
            if best is None or key < best[0]:
                best = (key, (rx, ry))
        if best is not None:
            tx, ty = best[1]
        else:
            tx, ty = w // 2, h // 2
    else:
        tx, ty = w // 2, h // 2
    best_key = None
    best_move = [0, 0]
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if nx < 0 or nx >= w or ny < 0 or ny >= h:
            continue
        if (nx, ny) in occ:
            continue
        dist = abs(tx - nx) + abs(ty - ny)
        key = (dist, nx, ny, dx, dy)
