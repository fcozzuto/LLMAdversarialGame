    opp_pos = observation.get('opponent_position', [w - 1, h - 1])
    resources = observation.get('resources', [])
    obstacles = observation.get('obstacles', [])
    ox = {tuple(p) for p in obstacles}
    sx, sy = int(self_pos[0]), int(self_pos[1])
    ex, ey = int(opp_pos[0]), int(opp_pos[1])

    def inb(x, y): return 0 <= x < w and 0 <= y < h
    def md(a, b, c, d): 
        dx = a - c
        dy = b - d
        if dx < 0: dx = -dx
        if dy < 0: dy = -dy
        return dx + dy
    def step_toward(cx, cy, tx, ty):
        dx = tx - cx
        dy = ty - cy
        if dx: dx = 1 if dx > 0 else -1
        if dy: dy = 1 if dy > 0 else -1
        return dx, dy

    # Deterministically pick a target where we are relatively closer than the opponent
    best_t = None
    best_ts = None

    if not resources:
        dx = 1 if ox > sx else (-1 if ox < sx else 0)
        dy = 1 if oy > sy else (-1 if oy < sy else 0)
        return [dx, dy]

    def man(x1, y1, x2, y2):
        d = x1 - x2
        if d < 0:
            d = -d
        e = y1 - y2
        if e < 0:
            e = -e
        return d + e

    best = None
