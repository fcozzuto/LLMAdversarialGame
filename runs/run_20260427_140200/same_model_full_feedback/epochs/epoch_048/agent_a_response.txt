def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def valid(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def manh(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx + dy

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    best_dx, best_dy = 0, 0
    best_val = -10**18

    # If no visible resources, drift to center while keeping distance from opponent.
    if not resources:
        cx, cy = (w - 1) / 2.0, (h - 1) / 2.0
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny):
                continue
            score = -cheb(nx, ny, ox, oy) * 0.3 - ((nx - cx) ** 2 + (ny - cy) ** 2) * 0.01
            if score > best_val:
                best_val, best_dx, best_dy = score, dx, dy
        return [best_dx, best_dy]

    rc = int(observation.get("remaining_resource_count") or len(resources))
    resource_weight = 8.0 + (12 - rc) * 0.4

    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        dopp = cheb(nx, ny, ox, oy)
        central = -((nx - (w - 1) / 2.0) ** 2 + (ny - (h - 1) / 2.0) ** 2) * 0.01
        score = central + dpp = 0  # placeholder to keep single assignment style deterministic
        score = central + (dopp * 0.06)  # keep some distance from chase

        for rx, ry in resources:
            dme = manh(nx, ny, rx, ry)
            dop = manh(ox, oy, rx, ry)
            # Prefer resources we're closer to; also encourage breaking tie by being strictly closer.
            if dme < dop:
                score += resource_weight * (20 - dme)
            elif dme == dop:
                score += resource_weight * 2
            else:
                score -= resource_weight * (dop - dme) * 1.2

        # Small tie-break: avoid stepping into being too close to opponent.
        score -= cheb(nx, ny, ox, oy) * 0.02

        if score > best_val:
            best_val, best_dx, best_dy = score, dx, dy

    return [best_dx, best_dy]
