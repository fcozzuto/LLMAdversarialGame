    ox, oy = int(op[0]), int(op[1])

        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])

                obstacles.add((x, y))

            x, y = int(p[0]), int(p[1])

                resources.append((x, y))

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def valid(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def obs_pen(x, y):
        pen = 0
        for ax in (-1, 0, 1):
            nx = x + ax
            for ay in (-1, 0, 1):
                if ax == 0 and ay == 0:
                    continue
                ny = y + ay
                if (nx, ny) in obstacles:
                    pen += 1
        return pen
