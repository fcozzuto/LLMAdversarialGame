def choose_move(observation):
    # Deterministic policy for agent_b in an 8x8 grid competition
    x, y = observation["self_position"]
    w, h = observation["grid_width"], observation["grid_height"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(o) for o in observation.get("obstacles", []))

    def in_bounds(nx, ny):
        return 0 <= nx < w and 0 <= ny < h

    # 1) Move toward closest resource not blocked
    if resources:
        best_r = None
        best_d = None
        for rx, ry in resources:
            if (rx, ry) in obstacles:
                continue
            d = abs(rx - x) + abs(ry - y)
            if best_d is None or d < best_d:
                best_d = d
                best_r = (rx, ry)
        if best_r is not None:
            rx, ry = best_r
            dx = 0 if rx == x else (1 if rx > x else -1)
            dy = 0 if ry == y else (1 if ry > y else -1)
            nx, ny = x + dx, y + dy
            if in_bounds(nx, ny) and (nx, ny) not in obstacles:
                return [dx, dy]

    # 2) Otherwise, move toward opponent if possible
    oppx, oppy = observation["opponent_position"]
    dx = 0 if oppx == x else (1 if oppx > x else -1)
    dy = 0 if oppy == y else (1 if oppy > y else -1)
    nx, ny = x + dx, y + dy
    if in_bounds(nx, ny) and (nx, ny) not in obstacles:
        return [dx, dy]

    # 3) If blocked, pick first valid non-stay move in deterministic order
    directions = [(-1, -1), (-1, 0), (-1, 1),
                  (0, -1), (0, 0), (0, 1),
                  (1, -1), (1, 0), (1, 1)]
    for ddx, ddy in directions:
        nx, ny = x + ddx, y + ddy
        if in_bounds(nx, ny) and (nx, ny) not in obstacles:
            if not (ddx == 0 and ddy == 0):
                return [ddx, ddy]
            return [0, 0]

    return [0, 0]
