def choose_move(observation):
    # Deterministic: move along Manhattan shortest path toward nearest resource.
    x, y = observation["self_position"]
    resources = observation.get("resources", [])
    if not resources:
        return [0, 0]
    # pick the closest resource by Manhattan distance
    target = min(resources, key=lambda p: abs(p[0] - x) + abs(p[1] - y))
    dx = 0 if target[0] == x else (1 if target[0] > x else -1)
