def choose_move(observation):
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []

    obst = set()
    for p in obstacles:
        try:
            x, y = p
            obst.add((int(x), int(y)))
        except:
            pass

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(ax, ay, bx, by):
        dx = abs(ax - bx)
        dy = abs(ay - by)
        return dx if dx > dy else dy

    deltas = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    if (0, 0) not in deltas:
        deltas.append((0, 0))

    moves = []
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if inb(nx, ny) and (nx, ny) not in obst:
            moves.append((dx, dy))
    if not moves:
        return [0, 0]

    # Pick a resource where we have advantage over opponent (self closer than opp).
    best_target = None
    best_adv = -10**18
    for r in resources:
        try:
            rx, ry = r
            rx, ry = int(rx), int(ry)
        except:
            continue
        sd = cheb(sx, sy, rx, ry)
        od = cheb(ox, oy, rx, ry)
        adv = od - sd  # positive means we are closer
        # also add slight preference for closer resources after advantage
        score = adv * 10 - sd
        if score > best_adv:
            best_adv = score
            best_target = (rx, ry)

    # If no resources, just drift toward opponent's side to contest.
    if best_target is None:
        tx, ty = (w - 1, h - 1) if (sx + sy) < (ox + oy) else (0, 0)
    else:
        tx, ty = best_target

    # Score each move with attraction to target, contesting, and opponent avoidance.
    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        myd = cheb(nx, ny, tx, ty)
        oppd = cheb(nx, ny, ox, oy)

        # Encourage reducing distance to target; discourage stepping into opponent's range.
        val = 0
        val += (-myd) * 5

        # Contest: would we become closer than opponent to the same target?
        if best_target is not None:
            opp_to_t = cheb(ox, oy, tx, ty)
            # if we reduce our distance below opponent distance, get big boost
            if myd < opp_to_t:
                val += (opp_to_t - myd) * 20
            else:
                val -= (myd - opp_to_t) * 8

        # Opponent proximity penalty; strong when too close.
        if oppd == 0:
            val -= 1000
        else:
            val -= (6 - min(6, oppd)) * 60

        # Obstacle-line avoidance: slight penalty for moves that reduce mobility.
        free_n = 0
        for ddx, ddy in deltas:
            mx, my = nx + ddx, ny + ddy
            if inb(mx, my) and (mx, my) not in obst:
                free_n += 1
        val += free_n * 3

        # Deterministic tie-breaker: prefer lexicographically smaller (dx,dy) after value.
        if val > best_val or (val == best_val and (dx, dy) < best_move):
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
