        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if inb(x, y):
                obstacles.add((x, y))

                resources.append((x, y))

        if inb(nx, ny) and (nx, ny) not in obstacles:
            legal.append((dx, dy))
