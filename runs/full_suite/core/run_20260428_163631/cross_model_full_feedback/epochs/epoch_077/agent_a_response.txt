        return 0 <= x < w and 0 <= y < h

    for p in (observation.get("obstacles", []) or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if inb(x, y):
                obstacles.add((x, y))

    resources = []
    for r in (observation.get("resources", []) or []):
        if r and len(r) >= 2:

                resources.append((x, y))

    legal = []

        nx, ny = sx + dx, sy + dy

            legal.append((dx, dy, nx, ny))
