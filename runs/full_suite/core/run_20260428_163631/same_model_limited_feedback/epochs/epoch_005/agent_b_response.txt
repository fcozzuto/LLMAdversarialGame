def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation["resources"] or []
    obstacles = set(tuple(p) for p in (observation["obstacles"] or []))

    def in_bounds(x, y):
        return 0 <= x < w and 0 <= y < h

    def man(a, b, c, d):
        return abs(a - c) + abs(b - d)

    if not resources:
        # No resources: move away from opponent if possible
        best = [0, 0]
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                nx, ny = sx + dx, sy + dy
                if not in_bounds(nx, ny) or (nx, ny) in obstacles:
                    continue
                dcur = man(nx, ny, ox, oy)
                dbest = man(best[0], best[1], ox, oy) if best != [0, 0] else man(sx, sy, ox, oy)
                if dcur > dbest:
                    best = [nx, ny]
        if best == [0, 0]:
            return [0, 0]
        return [best[0] - sx, best[1] - sy]

    # Choose target resource with best advantage (opponent far relative to us)
    best_res = None
    best_adv = None
    best_ds = None
    for rx, ry in resources:
        ds = man(sx, sy, rx, ry)
        do = man(ox, oy, rx, ry)
        adv = do - ds
        if (best_res is None or adv > best_adv or
            (adv == best_adv and (ds < best_ds or (ds == best_ds and (rx, ry) < best_res))))):
            best_res = (rx, ry)
            best_adv = adv
            best_ds = ds

    tx, ty = best_res
    # Evaluate candidate moves by resulting advantage and self distance
    moves = []
    cur_ds = man(sx, sy, tx, ty)
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if not in_bounds(nx, ny) or (nx, ny) in obstacles:
                continue
            ds = man(nx, ny, tx, ty)
            do = man(ox, oy, tx, ty)
            adv = do - ds
            step_pen = (1 if (dx == 0 and dy == 0) else 0)
            # Prefer decreasing distance; if tied, prefer larger advantage; avoid stalling
            score = (adv, -ds, -step_pen, -man(ox, oy, nx, ny))
            moves.append((score, dx, dy))
    if not moves:
        return [0, 0]
    moves.sort(reverse=True)
    return [moves[0][1], moves[0][2]]
