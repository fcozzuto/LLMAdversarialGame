def choose_move(observation):
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))
    sp = observation.get("self_position", [0, 0])
    op = observation.get("opponent_position", [0, 0])
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if inside(x, y):
                resources.append((x, y))

    moves = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1) if inside(sx + dx, sy + dy)]
    if not moves:
        return [0, 0]

    def dist2(ax, ay, bx, by):
        dx, dy = ax - bx, ay - by
        return dx * dx + dy * dy

    # Evaluate each move deterministically; lower is better.
    # Components:
    # - primary: closer to best resource (by "advantage" vs opponent)
    # - secondary: keep away from opponent
    # - tertiary: avoid getting stuck near obstacles (prefer moves with more free neighbors)
    best = None
    best_score = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy

        # nearest resource advantage: favor resources where we are closer than opponent
        if resources:
            best_adv = None
            best_dres = None
            for tx, ty in resources:
                d_me = dist2(nx, ny, tx, ty)
                d_opp = dist2(ox, oy, tx, ty)
                adv = d_opp - d_me  # positive means we are closer
                if best_adv is None or adv > best_adv or (adv == best_adv and d_me < best_dres):
                    best_adv = adv
                    best_dres = d_me
            # if we are behind (adv negative), push harder toward resources but penalize being behind
            res_score = (-best_adv) * 6 + best_dres
        else:
            # no resources: move toward center
            cx, cy = (w - 1) / 2.0, (h - 1) / 2.0
            dd = (nx - cx) * (nx - cx) + (ny - cy) * (ny - cy)
            res_score = dd * 2.0

        # keep away from opponent
        opp_d = dist2(nx, ny, ox, oy)
        opp_score = 1e6 if (ox == nx and oy == ny) else (30 * (1.0 / (opp_d + 1)))

        # obstacle/escape measure: count reachable neighbors (including staying still)
        free = 0
        for mx, my in (-1, 0, 1), (-1, 0, 1):
            pass
        for ddx in (-1, 0, 1):
            for ddy in (-1, 0, 1):
                tx, ty = nx + ddx, ny + ddy
                if inside(tx, ty):
                    free += 1
        escape_score = -free * 3

        score = res_score + opp_score + escape_score + (dx == 0 and dy == 0) * 0.5
        if best_score is None or score < best_score:
            best_score = score
            best = (dx, dy)

    return [int(best[0]), int(best[1])]
