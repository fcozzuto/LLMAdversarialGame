def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def obs_near(x, y):
        r = 0
        for ax in (-1, 0, 1):
            for ay in (-1, 0, 1):
                if ax == 0 and ay == 0:
                    continue
                if (x + ax, y + ay) in obstacles:
                    r += 1
        return r

    # Candidate move scoring: deterministic, no search.
    moves = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            if not ok(sx + dx, sy + dy):
                continue
            moves.append((dx, dy))

    if not moves:
        return [0, 0]

    # If no resources, drift away from opponent while staying safe.
    if not resources:
        best = None
        bestv = None
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            v = 10 * cheb(nx, ny, ox, oy) - 5 * obs_near(nx, ny)
            if best is None or v > bestv:
                best = (dx, dy)
                bestv = v
        return [best[0], best[1]]

    best_move = None
    best_score = None

    # Strategy change: choose a target where we can reach first AND that we can "pinch" opponent by denying its approach.
    # For each target, score our candidate move by (our gain) - (opponent gain), plus safety.
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        my_best = -10**9
        for tx, ty in resources:
            md = cheb(nx, ny, tx, ty)
            od = cheb(ox, oy, tx, ty)

            # If opponent is closer, bias against that target; otherwise reward it heavily.
            lead = od - md  # positive means we are closer
            # Pinch: prefer targets near the line between opponent and resource by maximizing opponent distance after our move.
            opp_to_target = od
            # Also softly prefer clearing nearer resources.
            my_progress = -md

            # Safety: avoid stepping near obstacles more than necessary.
            safety = -2 * obs_near(nx, ny)

            # Overall deterministic score for this move given target.
            # Strongly prefer targets where lead is positive; if none, still prefer "least bad".
            v = 20 * lead + 2 * my_progress + 0.3 * opp_to_target + safety

            if v > my_best:
                my_best = v

        # Final tie-break: prefer larger distance from opponent (prevents risky contest) if similar.
        tie = 0.01 * cheb(nx, ny, ox, oy)
        total = my_best + tie

        if best_score is None or total > best_score:
            best_score = total
            best_move = (dx, dy)

    if best_move is None:
        return [0, 0]
    return [int(best_move[0]), int(best_move[1])]
