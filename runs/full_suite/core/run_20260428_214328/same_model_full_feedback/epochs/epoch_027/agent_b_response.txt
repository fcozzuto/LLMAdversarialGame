def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    center = ((w - 1) / 2.0, (h - 1) / 2.0)

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    def best_target(cx, cy):
        if not resources:
            return None, 10**9, 10**9
        best = None
        best_score = -10**18
        best_dus = 10**9
        best_dou = 10**9
        for rx, ry in resources:
            du = cheb(cx, cy, rx, ry)
            do = cheb(ox, oy, rx, ry)
            # Prefer resources we can reach no later than opponent; then closer ones.
            contest = 0
            if du <= do: contest = 1
            score = (2 if contest else 0) - du + (0.5 if (du == do and contest) else 0) - 0.01 * (abs(rx - center[0]) + abs(ry - center[1]))
            if score > best_score or (score == best_score and (du < best_dus or (du == best_dus and (rx, ry) < best))):
                best_score = score
                best = (rx, ry)
                best_dus = du
                best_dou = do
        return best, best_dus, best_dou

    # If resource exists, chase best contested target; otherwise drift to center.
    target, d_us, d_op = best_target(sx, sy)

    best_move = [0, 0]
    best_val = -10**18

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny):
            continue

        # Avoid stepping adjacent to opponent unless it helps securing a contested resource.
        opp_adj = 1 if cheb(nx, ny, ox, oy) == 1 else 0
        opp_close_pen = opp_adj * 0.8

        if target is None:
            val = -((nx - center[0]) ** 2 + (ny - center[1]) ** 2) * 0.01
        else:
            tr_x, tr_y = target
            new_du = cheb(nx, ny, tr_x, tr_y)
            # Bonus for becoming (or staying) the closer reacher to target.
            new_do = cheb(ox, oy, tr_x, tr_y)
            win_bonus = 2.0 if new_du <= new_do else -0.5
            val = win_bonus - new_du - 0.05 * opp_close_pen

            # Small tie-break: prefer moves that reduce distance to the opponent only when equal contested.
            if new_du == new_do:
                val += 0.01 * (cheb(nx, ny, ox, oy) * -1)

        # Deterministic tie-break: lexicographic by (dx,dy) after maximizing val.
        if val > best_val or (val == best_val and (dx, dy) < (best_move[0], best_move[1])):
            best_val = val
