    obstacles = set(tuple(p) for p in observation.get("obstacles", []))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def bfs_dist(start):
        from collections import deque
        q = deque([start])
        dist = {start: 0}
        while q:
            x, y = q.popleft()
            d = dist[(x, y)]
            for dx in (-1, 0, 1):
                for dy in (-1, 0, 1):
                    if dx == 0 and dy == 0:
                        continue
                    nx, ny = x + dx, y + dy
                    if not inb(nx, ny) or (nx, ny) in obstacles:
                        continue
                    if (nx, ny) not in dist:
                        dist[(nx, ny)] = d + 1
                        q.append((nx, ny))
        return dist

    sd = bfs_dist((sx, sy))
    od = bfs_dist((ox, oy))
