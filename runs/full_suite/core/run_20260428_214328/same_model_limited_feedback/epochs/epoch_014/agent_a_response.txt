def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    obstacles = set((p[0], p[1]) for p in observation["obstacles"])
    resources = observation["resources"] if observation["resources"] is not None else []
    moves = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    def inb(x, y): return 0 <= x < w and 0 <= y < h
    def dist(a, b):  # chebyshev for grid moves with diagonals
        return max(abs(a[0]-b[0]), abs(a[1]-b[1]))
    # If standing on invalid or blocked, try to step out deterministically
    if (sx, sy) in obstacles:
        for dx, dy in moves:
            nx, ny = sx+dx, sy+dy
            if inb(nx, ny) and (nx, ny) not in obstacles:
                return [dx, dy]
        return [0, 0]
    # Find best target resource to pursue/deny based on relative closeness
    best = None
    for rx, ry in resources:
        ds = dist((sx, sy), (rx, ry))
        dp = dist((ox, oy), (rx, ry))
        # Lower is better: prefer closer to self, farther from opponent
        score = ds - 0.85 * dp
        # Encourage capturing nearer resources strongly
        score += 0.03 * (ds * ds)
        if best is None or score < best[0]:
            best = (score, rx, ry, ds, dp)
    # If no resources, head toward center while avoiding obstacles/opponent pressure
    if best is None:
        tx, ty = (w-1)//2, (h-1)//2
    else:
        _, tx, ty, _, _ = best
    # If opponent is extremely close, move away (deny potential collision/capture)
    opp_close = dist((sx, sy), (ox, oy)) <= 1
    # Choose move with deterministic tie-breaking
    best_move = (10**9, 10**9, None)
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obstacles:
            continue
        # stay penalty if harmful (deterministic preference)
        ds_t = dist((nx, ny), (tx, ty))
        ds_now = dist((sx, sy), (tx, ty))
        dp_next = dist((nx, ny), (ox, oy))
        # Primary objective: reduce distance to target; secondary: increase distance from opponent if close
        primary = ds_t
        secondary = -dp_next if opp_close else dist((ox, oy), (tx, ty))
        # Encourage progress (prefer moves that reduce distance rather than stall)
        progress = (ds_now - ds_t)
        # Deterministic tie-break: smaller dx, then smaller dy
        tie = (abs(dx), abs(dy), dx, dy, 0 if (dx==0 and dy==0) else 1)
        val = (primary, secondary, -progress)
        if val < (best_move[0], best_move[1], best_move[2] if best_move[2] is not None else 0):
            best_move = (primary, secondary, -progress), tie, (dx, dy)
    if best_move[2] is None:
        return [0, 0]
    return [best_move[2][0], best_move[2][1]]
