def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [w - 1, h - 1])
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []
    obst = set((p[0], p[1]) for p in obstacles)

    def inb(x, y): 
        return 0 <= x < w and 0 <= y < h

    def dist(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    if not resources:
        deltas = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1) if inb(sx + dx, sy + dy) and (sx + dx, sy + dy) not in obst]
        return [min(deltas, key=lambda d: (-(abs((sx + d[0]) - ox) + abs((sy + d[1]) - oy))), abs(d[0]) + abs(d[1])))[0],
                min(deltas, key=lambda d: (-(abs((sx + d[0]) - ox) + abs((sy + d[1]) - oy))), abs(d[0]) + abs(d[1])))[1]] if deltas else [0, 0]

    best_t = None
    best_val = None
    for r in resources:
        rx, ry = r
        ds = dist((sx, sy), (rx, ry))
        do = dist((ox, oy), (rx, ry))
        val = (do - ds, -ds, -rx, -ry)
        if best_val is None or val > best_val:
            best_val = val
            best_t = (rx, ry)

    tx, ty = best_t
    deltas = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1) if inb(sx + dx, sy + dy) and (sx + dx, sy + dy) not in obst]
    if not deltas:
        return [0, 0]

    def sqdist(x, y, tx, ty):
        dx = x - tx
        dy = y - ty
        return dx * dx + dy * dy

    best = None
    best_score = None
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        approach = sqdist(nx, ny, tx, ty)
        opp = dist((nx, ny), (ox, oy))
        # prefer reducing target distance; break ties by keeping away from opponent; then deterministic by dx,dy
        score = ( -approach, opp, -abs(dx) - abs(dy), dx, dy )
        if best_score is None or score > best_score:
            best_score = score
            best = (dx, dy)

    return [int(best[0]), int(best[1])]
