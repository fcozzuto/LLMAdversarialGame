def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))
    if w != 8 or h != 8:
        pass

    moves = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    def inside(x, y):
        return 0 <= x < w and 0 <= y < h
    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    best = (float("-inf"), 0, 0)
    if resources:
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not inside(nx, ny) or (nx, ny) in obstacles:
                continue
            # advantage = how much closer we are than opponent to our best next resource
            best_adv = float("-inf")
            for rx, ry in resources:
                ds = cheb(nx, ny, rx, ry)
                do = cheb(ox, oy, rx, ry)
                adv = do - ds
                if adv > best_adv:
                    best_adv = adv
            # tie-breakers: prefer reaching any resource now; then smaller self distance; then deterministic lexicographic
            reached_now = 1 if (nx, ny) in set(tuple(p) for p in resources) else 0
            # deterministic small bias to avoid loops: prefer lower distance to the closest resource, then toward opponent
            closest_ds = min(cheb(nx, ny, rx, ry) for rx, ry in resources)
            toward_opp = cheb(nx, ny, ox, oy)
            cand = (best_adv, reached_now, -closest_ds)  # higher is better; reached_now and -closest_ds are comparable
            if cand > (best[0], best[1], best[2]):
                best = (best_adv, reached_now, -closest_ds)
                best_move = (dx, dy)
        if "best_move" in locals():
            return [best_move[0], best_move[1]]

    # No resources or all moves blocked: move toward center while staying legal
    cx, cy = (w - 1) // 2, (h - 1) // 2
    best_d = float("inf")
    best_move = (0, 0)
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inside(nx, ny) or (nx, ny) in obstacles:
            continue
        d = cheb(nx, ny, cx, cy)
        if d < best_d or (d == best_d and (dx, dy) < best_move):
            best_d = d
            best_move = (dx, dy)
    return [best_move[0], best_move[1]]
