    obstacles = set(map(tuple, observation.get('obstacles', [])))

    def cheb(x1, y1, x2, y2):

        dy = y1 - y2

    # Correct chebyshev without extra lines:
    def cd(x1, y1, x2, y2):
        dx = x1 - x2
        dy = y1 - y2
        adx = dx if dx >= 0 else -dx
        ady = dy if dy >= 0 else -dy
        return adx if adx >= ady else ady

    # Choose a target resource where we are relatively closer than opponent.
    best_res = None
    best_val = None
    for rx, ry in resources:
        sd = cd(sx, sy, rx, ry)
        od = cd(ox, oy, rx, ry)
        val = (od - sd) * 10 - sd  # prioritize being closer, then closer overall
        if best_val is None or val > best_val:
            best_val = val
            best_res = (rx, ry)

    if best_res is None:
        # No resources: move to maximize distance from opponent while avoiding obstacles.
        target = (ox, oy)
        maximize = True
    else:
        target = best_res
        maximize = False

    deltas = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    tx, ty = target
    best_move = (0, 0)
    best_score = None

    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if nx < 0 or nx >= w or ny < 0 or ny >= h:
            continue
        if (nx, ny) in obstacles:
            continue

        if maximize:
            score = cd(nx, ny, tx[0], tx[1])  # maximize away from opponent
        else:
            dist_to_t = cd(nx, ny, tx, ty)
            dist_self = cd(nx, ny, tx, ty)
            dist_opp = cd(ox, oy, tx, ty)
            # Encourage approaching target while preserving lead over opponent

            score = lead * 12 - dist_to_t * 2

        score += (cd(nx, ny, ox, oy) * 0.3)

        if best_score is None or score > best_score:
            best_score = score
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
