def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation["resources"] or []
    obstacles = set(tuple(p) for p in (observation["obstacles"] or []))
    if not resources:
        dx = 0 if ox == sx else (1 if ox > sx else -1)
        dy = 0 if oy == sy else (1 if oy > sy else -1)
        if (sx + dx, sy + dy) in obstacles:
            dx = 0 if dx == 0 else dx
            dy = 0
        return [dx, dy]

    def man(a, b, c, d):
        return abs(a - c) + abs(b - d)

    targets = []
    for rx, ry in resources:
        ds = man(sx, sy, rx, ry)
        do = man(ox, oy, rx, ry)
        # Prefer resources we can reach sooner; break ties by being closer overall
        pr = (ds - do, ds, (rx - sx) * (rx - sx) + (ry - sy) * (ry - sy))
        targets.append((pr, rx, ry))
    targets.sort()
    _, tx, ty = targets[0]

    if (sx, sy) == (tx, ty):
        return [0, 0]

    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    best = (10**9, None)
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if nx < 0 or nx >= w or ny < 0 or ny >= h:
            continue
        if (nx, ny) in obstacles:
            continue
        d = man(nx, ny, tx, ty)
        rel = man(nx, ny, tx, ty) - man(ox, oy, tx, ty)
        sc = (d, rel, abs((nx - tx) - (ny - ty)))
        if sc < best[0]:
            best = (sc, (dx, dy))
    if best[1] is not None:
        return [int(best[1][0]), int(best[1][1])]

    # If surrounded by obstacles, take a deterministic safe-ish step (may be rejected by engine)
    dx = 0 if tx == sx else (1 if tx > sx else -1)
    dy = 0 if ty == sy else (1 if ty > sy else -1)
    if (sx + dx, sy + dy) in obstacles:
        if (sx + dx, sy) not in obstacles:
            dy = 0
        elif (sx, sy + dy) not in obstacles:
            dx = 0
        else:
            dx = 0
            dy = 0
    return [int(dx), int(dy)]
