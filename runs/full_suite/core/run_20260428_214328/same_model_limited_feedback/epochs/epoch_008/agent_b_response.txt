def choose_move(observation):
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))
    grid_w = observation.get("grid_width", 8)
    grid_h = observation.get("grid_height", 8)

    def in_bounds(x, y):
        return 0 <= x < grid_w and 0 <= y < grid_h

    def dist2(x1, y1, x2, y2):
        dx, dy = x1 - x2, y1 - y2
        return dx * dx + dy * dy

    deltas = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    if not resources:
        # No resources known: drift to maximize distance from opponent
        best = (-(dist2(sx, sy, ox, oy)), 0, 0)
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not in_bounds(nx, ny) or (nx, ny) in obstacles:
                continue
            d = dist2(nx, ny, ox, oy)
            key = (-d, nx, ny)
            if key < best:
                best = key
                best_step = (dx, dy)
        return [best_step[0], best_step[1]] if "best_step" in locals() else [0, 0]

    best_res = None
    best_score = None
    # Choose resource where we are relatively closer than opponent, with tie-break toward nearest
    for rx, ry in resources:
        if (rx, ry) in obstacles:
            continue
        sd = dist2(sx, sy, rx, ry)
        od = dist2(ox, oy, rx, ry)
        score = (od - sd, -sd)  # want larger (we closer), then smaller distance
        if best_score is None or score > best_score:
            best_score = score
            best_res = (rx, ry)
    rx, ry = best_res

    # Pick move that minimizes distance to target while avoiding obstacles and bounds
    best = None
    best_step = (0, 0)
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not in_bounds(nx, ny) or (nx, ny) in obstacles:
            continue
        d_to = dist2(nx, ny, rx, ry)
        d_opp = dist2(nx, ny, ox, oy)
        # Prefer approaching target; if equal, prefer being farther from opponent
        key = (d_to, -d_opp, nx, ny)
        if best is None or key < best:
            best = key
            best_step = (dx, dy)

    return [best_step[0], best_step[1]]
