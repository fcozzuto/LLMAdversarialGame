def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obst = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obst.add((x, y))

    res = []
    seen = set()
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obst and (x, y) not in seen:
                res.append((x, y))
                seen.add((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def man(ax, ay, bx, by):
        d = ax - bx
        if d < 0: d = -d
        e = ay - by
        if e < 0: e = -e
        return d + e

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def best_resource(fromx, fromy):
        if not res:
            return None, 10**9
        best = None
        bestd = 10**9
        for rx, ry in res:
            d = cheb(fromx, fromy, rx, ry)
            if d < bestd or (d == bestd and (ry, rx) < (best[1], best[0])):
                bestd = d
                best = (rx, ry)
        return best, bestd

    target, td = best_resource(sx, sy)
    if target is None:
        target = (ox, oy)
        td = cheb(sx, sy, ox, oy)

    if res:
        opt_t, od = best_resource(ox, oy)
        if opt_t is not None and opt_t == target and od + 1 < td:
            # Switch to the best alternative to reduce being out-contested.
            alt = None
            alt_d = 10**9
            for rx, ry in res:
                if (rx, ry) == target:
                    continue
                d = cheb(sx, sy, rx, ry)
                # prefer also far from opponent (so it's less likely to be stolen)
                s = d - 0.5 * cheb(ox, oy, rx, ry)
                if s < alt_d or (s == alt_d and (ry, rx) < (alt[1], alt[0])):
                    alt_d = s
                    alt = (rx, ry)
            if alt is not None:
                target = alt

    tx, ty = int(target[0]), int(target[1])

    # Greedy: pick move that minimizes chebyshev distance to target, avoid obstacles.
    bestm = (0, 0)
    best_score = 10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not (0 <= nx < w and 0 <= ny < h):
            continue
        if (nx, ny) in obst:
            continue
        dist_to_t = cheb(nx, ny, tx, ty)
        # small penalty for giving opponent better progress if very close
        dist_opp = cheb(ox, oy, tx, ty)
        score = dist_to_t * 10 + man(nx, ny, tx, ty)
        if dist_opp <= 2:
            score += 5
        # deterministic tie-break: prefer staying, then lexicographically smallest delta
        if score < best_score or (score == best_score and (dx, dy) < bestm):
            best_score = score
            bestm = (dx, dy)

    # If all moves invalid, stay put.
    if best_score == 10**18:
        return [0, 0]
    return [int(bestm[0]), int(bestm[1])]
