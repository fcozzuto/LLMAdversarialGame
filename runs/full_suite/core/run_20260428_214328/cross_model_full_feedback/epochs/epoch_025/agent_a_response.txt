        dy = ay - by

            dy = -dy
        return dx if dx >= dy else dy

    if resources:
        rs = [(int(p[0]), int(p[1])) for p in resources]
        best = None
        best_key = None
        for dx, dy, nx, ny in legal:
            our = (nx, ny)
            min_adv = 10**9
            best_res_dist = 10**9

                d_opp = cheb((ox, oy), r)

                if adv < min_adv:
