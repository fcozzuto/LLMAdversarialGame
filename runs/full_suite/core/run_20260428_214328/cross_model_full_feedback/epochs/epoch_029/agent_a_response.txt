        except:
            pass

            resources.append((int(p[0]), int(p[1])))

        return 0 <= x < w and 0 <= y < h

        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    legal = []

            legal.append((dx, dy, nx, ny))
