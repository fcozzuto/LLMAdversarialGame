def choose_move(observation):

    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obst = set()
    for p in (observation.get("obstacles") or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obst.add((x, y))

    resources = []
    for p in (observation.get("resources") or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obst:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    parity = int(observation.get("turn_index") or 0) & 1

    if not resources:

        cx, cy = (w - 1) / 2.0, (h - 1) / 2.0
        best = (-10**9, 0, 0)
        for dx, dy in moves:

            if not (0 <= nx < w and 0 <= ny < h) or (nx, ny) in obst:
                continue
            val = -cheb(nx, ny, int(cx), int(cy))
            if val > best[0]:

        return [best[1], best[2]]

    # Choose mode: even=collect advantage, odd=intercept by increasing opponent's approach to its best target

        d = cheb(sx, sy, rx, ry)
        od = cheb(ox, oy, rx, ry)
        advantage = od - d  # positive => we're closer
        best_targets.append((advantage, r))
