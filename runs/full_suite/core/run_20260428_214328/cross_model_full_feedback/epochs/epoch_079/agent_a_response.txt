def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position", [0, 0]) or [0, 0]
    op = observation.get("opponent_position", [w - 1, h - 1]) or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obst = set()
    for p in observation.get("obstacles") or []:
        if p is not None and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obst.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if p is not None and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obst:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    # Candidate next positions from allowed move deltas
    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    candidates = []
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obst:
            candidates.append((nx, ny, dx, dy))
        else:
            # even if engine rejects, still include staying outcome by evaluating as no-op
            if dx == 0 and dy == 0:
                candidates.append((sx, sy, 0, 0))
    if not candidates:
        return [0, 0]

    # No visible resources: contest by moving to reduce distance to opponent while preferring center-ish
    if not resources:
        best = None
        bestv = None
        cx, cy = (w - 1) / 2.0, (h - 1) / 2.0
        for nx, ny, dx, dy in candidates:
            d_to_opp = cheb(nx, ny, ox, oy)
            d_center = abs(nx - cx) + abs(ny - cy)
            # deterministic tie-break: lexicographic of (dx,dy)
            score = (d_to_opp * -10) + (-d_center)
            t = (score, dx, dy)
            if bestv is None or t > bestv:
                bestv = t
                best = (dx, dy)
        return [best[0], best[1]]

    # Resource contest heuristic:
    # Prefer resources we can reach sooner than opponent; otherwise pick closest resource.
    # Also include mild repulsion from opponent to prevent easy steals.
    best = None
    bestt = None
    for nx, ny, dx, dy in candidates:
        d_opp_next = cheb(nx, ny, ox, oy)
        # Evaluate best attainable resource under this move
        best_res_value = None
        for rx, ry in resources:
            my_d = cheb(nx, ny, rx, ry)
            op_d = cheb(ox, oy, rx, ry)
            # If we are closer/equal, good; if we are farther, still possible but less
            reach_diff = my_d - op_d  # <=0 means we are not worse
            # Encourage grabbing nearest favorable, but prioritize blocking opponent access by reach_diff
            val = (-my_d) + (-4.0 * reach_diff) - 0.1 * cheb(nx, ny, ox, oy)
            # deterministic tie-break: smallest resource then lexicographic val
            t2 = (val, -rx, -ry)
            if best_res_value is None or t2 > best_res_value:
                best_res_value = t2

        score = best_res_value[0]  # val
        # Prefer moving to equalize distance with opponent if contesting
        bonus = 0.0
        if score is not None:
            bonus = 0.05 * (d_opp_next * -1)

        t = (score + bonus, -d_opp_next, dx, dy)
        if bestt is None or t > bestt:
            bestt = t
            best = (dx, dy)

    return [int(best[0]), int(best[1])]

    def dist(a, b):
        ax, ay = a
        bx, by = b
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    if res:
        # choose nearest resource (Chebyshev), tie by closer to opponent (to reduce their chance indirectly)
        best = None
        for r in res:
            d = dist((sx, sy), r)
            if best is None:
                best = (d, dist((ox, oy), r), r)
            else:
                bd, bo, _ = best
                nd = d
                nb = dist((ox, oy), r)
                if nd < bd or (nd == bd and nb < bo):
                    best = (nd, nb, r)
        tx, ty = best[2]
    else:
        # no visible resources: drift toward corner away from opponent deterministically
        tx, ty = (0, 0) if (sx + sy) <= (w - 1 - sx + h - 1 - sy) else (w - 1, h - 1)

    best_move = None
    best_key = None

            continue
