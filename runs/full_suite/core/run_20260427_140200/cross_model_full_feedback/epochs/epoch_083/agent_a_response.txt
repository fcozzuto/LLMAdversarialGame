def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    me = observation.get("self_position") or [0, 0]
    opp = observation.get("opponent_position") or [0, 0]
    mx, my = int(me[0]), int(me[1])
    ox, oy = int(opp[0]), int(opp[1])

    obstacles = set()
    for p in observation.get("obstacles", []) or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources", []) or []:
        if r and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def in_bounds(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    moves = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]

    # If no resources, keep distance from opponent cornering
    if not resources:
        # Prefer moving to increase chebyshev distance from opponent
        best = None
        bestv = -10**9
        for dx, dy in moves:
            nx, ny = mx + dx, my + dy
            if not in_bounds(nx, ny):
                continue
            v = cheb(nx, ny, ox, oy)
            if v > bestv:
                bestv = v
                best = (dx, dy)
        if best is None:
            return [0, 0]
        return [best[0], best[1]]

    # Choose a "contested" target: maximize our advantage (dop - dme), tie-break by closeness to finish.
    best_adv = None
    best_t = None
    for rx, ry in resources:
        dme = cheb(mx, my, rx, ry)
        dop = cheb(ox, oy, rx, ry)
        adv = dop - dme
        # Prefer resources where we are already ahead; if behind, pick the least-bad contest.
        key = (adv, -dme)
        if best_adv is None or key > best_adv:
            best_adv = key
            best_t = (rx, ry)

    tx, ty = best_t

    # For each possible move, score the resulting advantage toward the chosen target.
    # Add a mild term to avoid steps that worsen "potential" relative to top few resources.
    top_resources = []
    for rx, ry in resources:
        dme = cheb(mx, my, rx, ry)
        dop = cheb(ox, oy, rx, ry)
        top_resources.append((dop - dme, -dme, rx, ry))
    top_resources.sort(reverse=True)
    top_resources = top_resources[:3]

    best = None
    best_score = -10**18
    for dx, dy in moves:
        nx, ny = mx + dx, my + dy
        if not in_bounds(nx, ny):
            continue
        dme_n = cheb(nx, ny, tx, ty)
        dop_n = cheb(ox, oy, tx, ty)  # opponent unchanged for one step
        score = (dop_n - dme_n) * 1000 - dme_n

        # Small extra: prefer improving average advantage over top resources
        extra = 0
        for _, _, rx, ry in top_resources:
            extra += (cheb(ox, oy, rx, ry) - cheb(nx, ny, rx, ry))
        score += extra
