def choose_move(observation):
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))
    me = observation.get("self_position", [0, 0]) or [0, 0]
    opp = observation.get("opponent_position", [0, 0]) or [0, 0]
    sx, sy = int(me[0]), int(me[1])
    ox, oy = int(opp[0]), int(opp[1])

    obstacles = set()
    for p in observation.get("obstacles", []) or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources", []) or []:
        if r and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    def legal(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    if not resources:
        best = None
        bestv = -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not legal(nx, ny):
                continue
            v = cheb(nx, ny, ox, oy)
            if v > bestv:
                bestv = v
                best = (dx, dy)
        return list(best if best is not None else (0, 0))

    # Choose a strategic target resource where we are (currently) relatively closer than opponent.
    my_best = None
    best_adv = -10**18
    for rx, ry in resources:
        my_d = cheb(sx, sy, rx, ry)
        op_d = cheb(ox, oy, rx, ry)
        adv = op_d - my_d  # positive => we are closer
        if adv > best_adv:
            best_adv = adv
            my_best = (rx, ry)

    # Score each move using advantage at the next position; also reward collecting immediately.
    tx, ty = my_best
    best_move = (0, 0)
    best_score = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not legal(nx, ny):
            continue
        my_next = cheb(nx, ny, tx, ty)
        opp_next = cheb(ox, oy, tx, ty)  # opponent position assumed fixed this turn
        adv_next = opp_next - my_next
        collect = 1 if (nx == tx and ny == ty) else 0

        # If not on target, consider also drifting toward the closest "currently winnable" resource.
        # Small deterministic bonus to avoid tunnel-vision.
        aux = 0
        closest_gap = 10**9
        for rx, ry in resources:
            my_d = cheb(nx, ny, rx, ry)
            op_d = cheb(ox, oy, rx, ry)
            gap = my_d - op_d  # smaller => better for us
            if gap < closest_gap:
                closest_gap = gap
                aux = -my_d

        score = 80 * collect + 35 * adv_next - my_next + 0.15 * cheb(nx, ny, ox, oy) + 2 * aux
        if score > best_score or (score == best_score and (dx, dy) < best_move):
            best_score = score
            best_move = (dx, dy)
