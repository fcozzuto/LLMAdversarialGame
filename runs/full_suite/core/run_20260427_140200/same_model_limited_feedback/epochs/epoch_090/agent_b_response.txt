def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))

    def clamp(v, a, b):
        return a if v < a else b if v > b else v

    def dist(ax, ay, bx, by):
        dx = ax - bx
        dy = ay - by
        if dx < 0: dx = -dx
        if dy < 0: dy = -dy
        return dx + dy

    if not resources:
        # No visible resources: drift away from opponent if possible, else toward center
        target = (w // 2, h // 2)
        best = None
        best_val = -10**18
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                nx, ny = sx + dx, sy + dy
                if (nx, ny) in obstacles or not (0 <= nx < w and 0 <= ny < h):
                    continue
                val = dist(nx, ny, ox, oy) - dist(nx, ny, target[0], target[1])
                if val > best_val:
                    best_val = val
                    best = (dx, dy)
        return [best[0], best[1]] if best is not None else [0, 0]

    # Pick a resource where we have a relative speed advantage over opponent
    best_target = None
    best_score = -10**18
    for rx, ry in resources:
        sd = dist(sx, sy, rx, ry)
        od = dist(ox, oy, rx, ry)
        # Prefer resources closer to us and farther from opponent; slight bias to corners/edges
        corner_bias = 0
        if rx in (0, w - 1) or ry in (0, h - 1):
            corner_bias = 0.5
        val = (od - sd) * 10 - sd + corner_bias
        if val > best_score:
            best_score = val
            best_target = (rx, ry)

    tx, ty = best_target
    # Greedy step toward chosen target, but avoid obstacles and strongly discourage moving closer to opponent if contested
    contested_penalty = 0
    if resources:
        # if opponent is very close to the same target, we should try a different angle
        if dist(ox, oy, tx, ty) <= dist(sx, sy, tx, ty):
            contested_penalty = 8

    best_move = (0, 0)
    best_val = -10**18
    cur_sd = dist(sx, sy, tx, ty)
    cur_od = dist(sx, sy, ox, oy)

    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if not (0 <= nx < w and 0 <= ny < h):
                continue
            if (nx, ny) in obstacles:
                continue
            nd = dist(nx, ny, tx, ty)
            no = dist(nx, ny, ox, oy)
            # reward improvement to our distance; if contested, also reward increasing opponent distance
            val = (cur_sd - nd) * 12 + (no - cur_od) * contested_penalty
            # Also avoid stepping adjacent to obstacles too much (keep mobility)
            adj_obs = 0
            for ax in (-1, 0, 1):
                for ay in (-1, 0, 1):
                    if ax == 0 and ay == 0:
                        continue

                    if (px, py) in obstacles:
                        adj_obs += 1
            val -= adj_obs * 0.6
            if val > best_val:
                best_val = val
                best_move = (dx, dy)

    def man(a, b, c, d):

        if x < 0:
            x = -x
        y = b - d
        if y < 0:

        return x + y
