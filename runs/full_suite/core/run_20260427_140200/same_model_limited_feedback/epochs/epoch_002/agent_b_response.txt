def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))

    deltas = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    def inb(x,y):
        return 0 <= x < w and 0 <= y < h
    def dist(a,b,c,d):
        dx = a - c
        dy = b - d
        if dx < 0: dx = -dx
        if dy < 0: dy = -dy
        return dx if dx > dy else dy  # Chebyshev (diagonal allowed)
    def blocked(x,y):
        return (x,y) in obstacles

    if not resources:
        # No resources: move to reduce distance to opponent (block-ish)
        best = (10**9, 10**9, 0, 0)
        for dx,dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not inb(nx,ny) or blocked(nx,ny):
                continue
            d = dist(nx,ny,ox,oy)
            cand = (d, 0, nx, ny)
            if cand < best:
                best = cand
                bd = (dx,dy)
        return list(bd) if "bd" in locals() else [0,0]

    # Pick target resource we can reach at least as well as opponent; else pick best lead.
    best_r = None
    best_key = None
    for rx, ry in resources:
        ds = dist(sx, sy, rx, ry)
        do = dist(ox, oy, rx, ry)
        if ds <= do:
            key = (0, ds, do, rx, ry)  # prioritize contested/advantaged, then closest
        else:
            key = (1, do - ds, ds, rx, ry)  # if behind, pick where we are relatively closer
        if best_key is None or key < best_key:
            best_key = key
            best_r = (rx, ry)

    tx, ty = best_r

    # Choose move minimizing distance to target, with tie-breakers: avoid opponent, avoid obstacles.
    best_move = None
    best_score = None
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not inb(nx,ny) or blocked(nx,ny):
            continue
        d_to_t = dist(nx, ny, tx, ty)
        d_to_opp = dist(nx, ny, ox, oy)
        # Score: primary reduce target distance; secondary keep distance from opponent; tertiary prefer staying when equal
        score = (d_to_t, -d_to_opp, abs(dx)+abs(dy), 0 if (dx==0 and dy==0) else 1)
        if best_score is None or score < best_score:
            best_score = score
            best_move = (dx, dy)

    if best_move is None:
        return [0, 0]
    return [int(best_move[0]), int(best_move[1])]
