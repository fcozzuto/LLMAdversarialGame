            x, y = int(p[0]), int(p[1])

                obstacles.add((x, y))

        return 0 <= x < w and 0 <= y < h

        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

        return inb(nx, ny) and (nx, ny) not in obstacles
