def choose_move(observation):
    x, y = observation.get("self_position", [0, 0])
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []
    ox, oy = observation.get("opponent_position", [0, 0])

    def inb(nx, ny):
        return 0 <= nx < w and 0 <= ny < h

    obs = set()
    for o in obstacles:
        if isinstance(o, (list, tuple)) and len(o) == 2:
            obs.add((int(o[0]), int(o[1])))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    resources_ok = []
    for r in resources:
        if isinstance(r, (list, tuple)) and len(r) == 2:
            rx, ry = int(r[0]), int(r[1])
            if inb(rx, ry) and (rx, ry) not in obs:
                resources_ok.append((rx, ry))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    if (x, y) in obs:
        # escape deterministically: smallest distance to nearest valid cell
        best = (10**9, 0, 0, 0, 0)
        for dx, dy in moves:
            nx, ny = x + dx, y + dy
            if inb(nx, ny) and (nx, ny) not in obs:
                d = abs(nx - (w // 2)) + abs(ny - (h // 2))
                if d < best[0]:
                    best = (d, dx, dy, nx, ny)
        if best[0] < 10**9:
            return [best[1], best[2]]
        return [0, 0]

    if not resources_ok:
        # drift to center while avoiding obstacles
        cx, cy = w // 2, h // 2
        bestd, best = 10**9, [0, 0]
        for dx, dy in moves:
            nx, ny = x + dx, y + dy
            if inb(nx, ny) and (nx, ny) not in obs:
                d = cheb(nx, ny, cx, cy)
                if d < bestd:
                    bestd = d
                    best = [dx, dy]
        return best

    # choose resource maximizing advantage (opponent farther than us), tie-break by lower self distance
    target = None
    best_adv = -10**9
    best_selfd = 10**9
    best_rx, best_ry = resources_ok[0]
    for rx, ry in resources_ok:
        sd = cheb(x, y, rx, ry)
        od = cheb(ox, oy, rx, ry)
        adv = od - sd
        if adv > best_adv or (adv == best_adv and sd < best_selfd) or (adv == best_adv and sd == best_selfd and (rx, ry) < (best_rx, best_ry)):
            best_adv = adv
            best_selfd = sd
            best_rx, best_ry = rx, ry
            target = (rx, ry)

    tx, ty = target

    # move to reduce distance to target; prefer diagonal naturally via cheb reduction
    curd = cheb(x, y, tx, ty)
    best_move = [0, 0]
    best_curd = curd
    best_score2 = -10**9
    for dx, dy in moves:
        nx, ny = x + dx, y + dy
        if not (inb(nx, ny) and (nx, ny) not in obs):
            continue
        nd = cheb(nx, ny, tx, ty)
        # primary: decrease nd
        # secondary: maximize further advantage improvement
        od = cheb(ox, oy, tx, ty)
        adv = od - nd
        score2 = adv * 1000 - nd
        if (nd < best_curd) or (nd == best_curd and score2 > best_score2):
            best_curd = nd
            best_score2 = score2
            best_move = [dx, dy]

    return best_move
