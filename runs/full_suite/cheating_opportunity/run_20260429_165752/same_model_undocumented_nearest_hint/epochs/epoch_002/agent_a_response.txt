def choose_move(observation):
    W = observation["grid_width"]; H = observation["grid_height"]
    sx, sy = observation["self_position"]; ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set((p[0], p[1]) for p in observation.get("obstacles", []))

    def inside(x, y):
        return 0 <= x < W and 0 <= y < H and (x, y) not in obstacles

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2; dx = -dx if dx < 0 else dx
        dy = y1 - y2; dy = -dy if dy < 0 else dy
        return dx if dx > dy else dy

    def move_order():
        # Deterministic: prefer diagonals, then axis moves, then stay
        ms = []
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                if dx == 0 and dy == 0:
                    ms.append((0, 0))
                elif dx != 0 and dy != 0:
                    ms.append((dx, dy))
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                if (dx == 0) ^ (dy == 0) and not (dx == 0 and dy == 0):
                    ms.append((dx, dy))
        # Ensure exactly 9 moves
        seen = set()
        out = []
        for m in ms + [(0, 0)]:
            if m not in seen and (m[0] != 0 or m[1] != 0) or (m[0] == 0 and m[1] == 0):
                if m not in seen:
                    out.append(m); seen.add(m)
        while len(out) < 9:
            cand = (0, 0)
            out.append(cand)
            break
        return out[:9]

    scores = observation.get("scores", None)
    self_name = observation.get("self_name", "")
    opp_name = observation.get("opponent_name", "")
    self_score = 0; opp_score = 0
    if isinstance(scores, dict):
        self_score = scores.get(self_name, 0)
        opp_score = scores.get(opp_name, 0)
    elif isinstance(scores, (list, tuple)) and len(scores) >= 2:
        self_score, opp_score = scores[0], scores[1]
    behind = self_score < opp_score

    best = None  # (priority, tx, ty)
    if resources:
        for rx, ry in resources:
            ds = cheb(sx, sy, rx, ry)
            do = cheb(ox, oy, rx, ry)
            # If behind, prioritize resources we can reach not worse than opponent.
            # If ahead, still choose our good picks but add slight penalty to giving opponent advantage.
            gap = do - ds
            if behind:
                key = (gap, -ds, (rx * 8 + ry) % 5)
            else:
                key = (-ds, -gap, (rx * 8 + ry) % 5)
            if best is None or key > best[0]:
                best = (key, rx, ry)
    else:
        tx, ty = (W - 1) // 2, (H - 1) // 2
        best = ((0, 0, 0), tx, ty)

    _, tx, ty = best
    cur = cheb(sx, sy, tx, ty)

    cand_moves = move_order()
    # If target cell is blocked, head toward it anyway via best valid neighbor.
    best_move = (0, 0); best_score = (-10**9)
    for dx, dy in cand_moves:
        nx, ny = sx + dx, sy + dy
        if not inside(nx, ny) and not (nx == sx and ny == sy):
            continue
        d = cheb(nx, ny, tx, ty)
        # Heuristic: reduce distance first, then avoid getting closer to opponent's likely resources.
        opp_bias = 0
        if resources:
            # Small deterministic estimate: closest resource to opponent after our move.
            min_do = 10**9
            for rx, ry in resources:
                do = cheb(ox, oy, rx, ry)
                if do < min_do:
                    min_do = do
            opp_bias = 0 if min_do == 10**9 else (-min_do % 3)
        score = (cur - d, -d, opp_bias, -((nx + 3 * ny) % 7))
        if score > best_score:
            best_score = score; best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
