    obs = set()
    for p in obstacles:
        if p and len(p) >= 2:

            if 0 <= x < W and 0 <= y < H:
                obs.add((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    legal = []

    if not legal:
        return [0, 0]

    def mindist(x, y, pts):
        if not pts:
            return 10**9
        m = 10**9
        for tx, ty in pts:
            d = abs(x - tx) + abs(y - ty)
            if d < m:
                m = d
        return m

    valid_res = []
    for r in resources:
        if r and len(r) >= 2:
            x, y = r[0], r[1]

    opp_d0 = abs(sx - ox) + abs(sy - oy)

    best_key = None
    for dx, dy in legal:

        ds = mindist(nx, ny, valid_res)
