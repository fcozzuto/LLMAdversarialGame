def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in (observation.get("obstacles", []) or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in (observation.get("resources", []) or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))
    if not resources:
        return [0, 0]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    # Candidate move evaluation: pick best target and best first step.
    dirs = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    best_move = [0, 0]
    best_score = -10**18

    # Determine best targets to chase/contest (static ranking).
    ranked = []
    for rx, ry in resources:
        sd = cheb(sx, sy, rx, ry)
        od = cheb(ox, oy, rx, ry)
        lead = od - sd  # positive means we arrive first (or tie if 0)
        clos = -(sd)
        # Prefer resources that are not behind obstacles in a naive way:
        # small penalty if target shares x or y with an obstacle near start.
        penalty = 0
        for (bx, by) in obstacles:
            if cheb(sx, sy, bx, by) <= 2 and (rx == bx or ry == by):
                penalty += 1
        ranked.append((lead, clos, -penalty, rx, ry))
    ranked.sort(reverse=True)

    top_targets = ranked[:4]  # enough to stay concise/deterministic

    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if nx < 0 or nx >= w or ny < 0 or ny >= h:
            nx, ny = sx, sy
        if (nx, ny) in obstacles:
            nx, ny = sx, sy

        move_score = -10**18
        # Evaluate against top targets: chase those where we can beat opponent.
        for i, (lead, clos, pen, rx, ry) in enumerate(top_targets):
            sd2 = cheb(nx, ny, rx, ry)
            od2 = cheb(ox, oy, rx, ry)  # opponent doesn't move this turn
            gain = od2 - sd2
            # If both can reach, contest if we are not too far behind.
            # Add tie-breaker: progress toward target; and slight preference to reduce distance.
            s = 1000 * (1 if gain > 0 else (0 if gain == 0 else -1))
            s += 20 * gain
            s += -sd2
            s += -i  # slight preference for earlier-ranked targets
            move_score = s if s > move_score else move_score

        if move_score > best_score:
            best_score = move_score
            best_move = [dx, dy]

    # Small endgame improvement: if opponent is closer to all resources, block by stepping toward median resource.
    if best_score < -500:

        for r in resources[1:]:
            # choose resource closest to opponent mid-way to reduce opponent advantage
            if cheb(ox, oy, r[0], r[1]) < cheb(ox, oy, mx, my):

        dy = 0 if my == sy else (1 if my > sy else -1)
        if (sx + dx, sy + dy) in obstacles:
            dx, dy = 0, 0
