def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position", [0, 0]) or [0, 0]
    op = observation.get("opponent_position", [w - 1, h - 1]) or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []

    obs = set()
    for p in obstacles:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obs.add((x, y))

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def obstacle_risk(x, y):
        if (x, y) in obs:
            return 10**7
        r = 0
        for (oxx, oyy) in obs:
            d = cheb(x, y, oxx, oyy)
            if d == 0:
                return 10**7
            if d == 1:
                r += 3
            elif d == 2:
                r += 1
        return r

    # If no resources, retreat to safer corner (opposite direction from opponent)
    if not resources:
        tx = 0 if sx > (w - 1 - sx) else (w - 1)
        ty = 0 if sy > (h - 1 - sy) else (h - 1)
        tx = tx if cheb(tx, sy, ox, oy) >= cheb(w - 1 - tx, sy, ox, oy) else (w - 1 - tx)
        ty = ty if cheb(sx, ty, ox, oy) >= cheb(sx, h - 1 - ty, ox, oy) else (h - 1 - ty)
        return [0 if tx == sx else (1 if tx > sx else -1), 0 if ty == sy else (1 if ty > sy else -1)]

    # Choose resource that we can reach earlier than opponent, while avoiding obstacles.
    # Deterministic tie-break by lexicographic target coordinates.
    best = None
    for r in resources:
        rx, ry = int(r[0]), int(r[1])
        myd = cheb(sx, sy, rx, ry)
        opd = cheb(ox, oy, rx, ry)
        risk = obstacle_risk(rx, ry)

        # If opponent is closer, penalize strongly; otherwise reward slightly.
        rel = opd - myd
        opp_term = -6 if rel < 0 else (-2 if rel == 0 else 0)

        # Prefer closer targets, but also prefer those with smaller risk.
        score = myd * 10 + opp_term * 10 + risk * 2 + (0 if (rx, ry) != (sx, sy) else -5)

        cand = (score, abs(rx - sx) + abs(ry - sy), rx, ry)
        if best is None or cand < best:
            best = cand

    _, _, tx, ty = best

    dx = 0 if tx == sx else (1 if tx > sx else -1)
    dy = 0 if ty == sy else (1 if ty > sy else -1)

    # Small deterministic obstacle-avoidance: try alternative deltas ordered by best target progress minus risk.
    options = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    options.sort(key=lambda d: (0 if d == (dx, dy) else 1, d[0], d[1]))
    best_move = (dx, dy)
    best_val = None
    for mx, my in options:
        nx, ny = sx + mx, sy + my
        if nx < 0 or nx >= w or ny < 0 or ny >= h:
            continue
        if (nx, ny) in obs and (mx, my) != (0, 0):
            continue
        prog = cheb(nx, ny, tx, ty)
        val = (prog, obstacle_risk(nx, ny), mx == dx and my == dy, mx, my)
        if best_val is None or val < best_val:
            best_val = val
