def choose_move(observation):
    w = observation.get("grid_width", 8) or 8
    h = observation.get("grid_height", 8) or 8
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [0, 0])
    resources = observation.get("resources") or []
    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            obstacles.add((p[0], p[1]))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    dirs = (-1, 0, 1)
    moves = [(dx, dy) for dx in dirs for dy in dirs]
    moves.sort(key=lambda t: (abs(t[0]) + abs(t[1]), t[0], t[1]))

    def dist_cheb(x1, y1, x2, y2):
        dx = x1 - x2
        dy = y1 - y2
        if dx < 0: dx = -dx
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def step_score(nx, ny):
        best_r = None
        best_key = None
        for rx, ry in resources:
            d_me = dist_cheb(nx, ny, rx, ry)
            d_op = dist_cheb(ox, oy, rx, ry)
            # Prefer resources where we are closer; otherwise deny by picking resources opponent is farther from
            # Key: primary closeness advantage, then our distance, then deterministic resource ordering
            adv = d_op - d_me
            key = (-(adv), d_me, rx, ry)  # higher adv -> smaller -adv
            if best_key is None or key < best_key:
                best_key = key
                best_r = (rx, ry)
        if best_r is None:
            return 10**9
        rx, ry = best_r
        # Additional terms:
        # - discourage moving closer to obstacles too tightly (one-step look)
        adj_block = 0
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                if dx == 0 and dy == 0:
                    continue
                tx, ty = nx + dx, ny + dy
                if (tx, ty) in obstacles or not inb(tx, ty):
                    adj_block += 1
        # - avoid stepping onto/adjacent to opponent (slight)
        opp_adj = dist_cheb(nx, ny, ox, oy)
        return (best_key[0] * 100 + best_key[1] * 3 + adj_block * 2 + (0 if opp_adj > 1 else 5))

    # If no resources, move to reduce risk by heading toward center
    if not resources:
        cx, cy = (w - 1) // 2, (h - 1) // 2
        best = None
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny) or (nx, ny) in obstacles:
                continue
            d = dist_cheb(nx, ny, cx, cy)
            key = (d, dx, dy)
            if best is None or key < best:
                best = key
        if best is None:
            return [0, 0]
        return [best[1], best[2]]

    best = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obstacles:
            continue
        # If we can land on a resource, prioritize heavily (cheapest deterministic)
        land = 0
        for rx, ry in resources:
            if nx == rx and ny == ry:
                land = 1
                break
        sc = step_score(nx, ny) - land * 10**6
        key = (sc, abs(dx) + abs(dy), dx, dy)
        if best is None or key < best:
            best = key

    if best is None:
        return [0, 0]
    return [best[2], best[3]]
