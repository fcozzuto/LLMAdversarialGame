def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    if not resources:
        return [0, 0]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    def line_of_sight_gain(nx, ny):
        # Prefer moving toward closest resource; slight bias away from obstacles-adjacent cells
        best = 10**9
        best2 = 10**9
        for rx, ry in resources:
            d = cheb(nx, ny, rx, ry)
            if d < best:
                best2 = best
                best = d
            elif d < best2:
                best2 = d
        # obstacle adjacency penalty (deterministic)
        adj = 0
        for tx in (-1, 0, 1):
            for ty in (-1, 0, 1):
                if tx == 0 and ty == 0:
                    continue
                x, y = nx + tx, ny + ty
                if (x, y) in obstacles:
                    adj += 1
        return (best, best2, adj)

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    valid_moves = []
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles:
            valid_moves.append((dx, dy))

    best_move = (0, 0)
    best_val = -10**18

    # Precompute opponent advantage baseline for tie-breaking
    opp_best_now = 10**9
    for rx, ry in resources:
        d = cheb(ox, oy, rx, ry)
        if d < opp_best_now:
            opp_best_now = d

    for dx, dy in valid_moves:
        nx, ny = sx + dx, sy + dy

        my_best, my_second, adj_pen = line_of_sight_gain(nx, ny)
        # Opponent's closest after our move (opponent position unchanged this turn)
        # Use resource-specific comparison implicitly by comparing my_best vs opp_best_now.
        lead = opp_best_now - my_best  # positive means we are closer than opponent to some resource

        # Extra: reward if we can "steal" closer than opponent to best resource deterministically by sampling
        steal = 0
        for rx, ry in resources:
            dme = cheb(nx, ny, rx, ry)
            dop = cheb(ox, oy, rx, ry)
            if dme < dop:
                steal = 1
                break

        # Score: maximize lead, then prefer faster second best, avoid obstacles-adjacent
        val = (lead * 1000) + (opp_best_now - my_second) + (200 if steal else 0) - adj_pen
        if val > best_val:
            best_val = val
            best_move = (dx, dy)
        elif val == best_val:
            # deterministic tie-break: prefer moves with smaller distance to resources, then lexicographic
            if cheb(nx, ny, ox, oy) < cheb(sx + best_move[0], sy + best_move[1], ox, oy):
                best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
