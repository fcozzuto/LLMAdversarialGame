def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in (observation.get("obstacles") or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in (observation.get("resources") or []):
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def valid(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def best_target():
        best = None
        for rx, ry in resources:
            ds = cheb(sx, sy, rx, ry)
            do = cheb(ox, oy, rx, ry)
            # Prefer resources where we are closer; then prefer shorter absolute distance.
            gain = do - ds
            score = (1 if gain > 0 else 0, gain, -ds, -rx, -ry)
            if best is None or score > best[0]:
                best = (score, (rx, ry))
        return best[1] if best else None

    tgt = best_target()

    # If we can't find a "winning" resource, aim to intercept: step toward our nearest resource
    # that is closer to us than to opponent, otherwise toward opponent's nearest resource.
    if resources and tgt is None:
        min_ds = None
        min_dop = None
        for rx, ry in resources:
            ds = cheb(sx, sy, rx, ry)
            do = cheb(ox, oy, rx, ry)
            if do - ds > 0 and (min_ds is None or ds < min_ds):
                min_ds = ds
                tgt = (rx, ry)
        if tgt is None:
            for rx, ry in resources:
                do = cheb(ox, oy, rx, ry)
                if min_dop is None or do < min_dop:
                    min_dop = do
                    tgt = (rx, ry)

    # If still no target (no resources visible), move to reduce distance to opponent.
    if tgt is None:
        best_mv = None
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny): 
                continue
            score = (-cheb(nx, ny, ox, oy), 0, -nx, -ny)
            if best_mv is None or score > best_mv[0]:
                best_mv = (score, (dx, dy))
        return best_mv[1] if best_mv else [0, 0]

    rx, ry = tgt
    # Move to minimize our distance to tgt; tie-break by maximizing opponent distance to tgt.
    best_mv = None
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        ds2 = cheb(nx, ny, rx, ry)
        do2 = cheb(ox, oy, rx, ry)
        # Since opponent position doesn't change this turn, higher ds2 is bad and do2 is fixed;
        # tie-break with our distance advantage (if we step onto it, ds2 becomes 0).
        adv = do2 - ds2
        score = (-(ds2), adv, cheb(nx, ny, ox, oy), -nx, -ny)
        if best_mv is None or score > best_mv[0]:
            best_mv = (score, (dx, dy))

    return list(best_mv[1] if best_mv else (0, 0))
