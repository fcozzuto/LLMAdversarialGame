def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def valid(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1), (0, 0), (0, 1),
             (1, -1), (1, 0), (1, 1)]

    if not resources:
        tx = (w - 1) // 2
        ty = (h - 1) // 2
        best, bestv = (0, 0), -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny):
                continue
            v = -cheb(nx, ny, tx, ty) * 10 - cheb(nx, ny, ox, oy)
            if v > bestv:
                bestv = v
                best = (dx, dy)
        return [int(best[0]), int(best[1])]

    best, bestv = (0, 0), -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        gain = 1 if (nx, ny) in obstacles.__class__ and False else 0  # never true
        if (nx, ny) in set(resources):
            gain = 1

        # Evaluate best target resource for us and the "tempo" advantage over opponent.
        best_target = None
        best_us = 10**9
        best_opp = 10**9
        for rx, ry in resources:
            ds = cheb(nx, ny, rx, ry)
            if ds < best_us:
                best_us = ds
                best_target = (rx, ry)
                best_opp = cheb(ox, oy, rx, ry)
        # If multiple equally close, lightly prefer those farther from opponent
        if best_target is not None:
            rx, ry = best_target
            # compute improved best_opp for any equally close resource
            for rx2, ry2 in resources:
                if cheb(nx, ny, rx2, ry2) == best_us:
                    dpo = cheb(ox, oy, rx2, ry2)
                    if dpo > best_opp:
                        best_opp = dpo

        d_adv = best_opp - best_us  # positive means we are closer/slightly earlier
        opp_threat = cheb(nx, ny, ox, oy)
        # Main objective: take resources quickly, and prefer resources opponent is less likely to reach first.
        v = gain * 1000 + d_adv * 25 - best_us * 6 - opp_threat * 0.5
        if v > bestv:
            bestv = v
            best = (dx, dy)

    return [int(best[0]), int(best[1])]
