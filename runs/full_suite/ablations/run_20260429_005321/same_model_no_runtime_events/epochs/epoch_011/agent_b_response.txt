def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))

    obstacles = set(tuple(p) for p in (observation.get("obstacles") or []))
    resources = [tuple(r) for r in (observation.get("resources") or []) if tuple(r) not in obstacles]

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    deltas = (-1, 0, 1)
    legal = []
    for dx in deltas:
        for dy in deltas:
            nx, ny = sx + dx, sy + dy
            if inside(nx, ny) and (nx, ny) not in obstacles:
                legal.append((dx, dy, nx, ny))

    if not legal:
        return [0, 0]

    if resources:
        best_key = None
        best_move = (0, 0)
        for dx, dy, nx, ny in legal:
            # Prefer resources where we can beat the opponent to arrival.
            # Use our next distance as tie-breaker; then keep distance from opponent.
            key = None
            for rx, ry in resources:
                myd = cheb(nx, ny, rx, ry)
                opd = cheb(ox, oy, rx, ry)
                beat = opd - myd  # positive if we are closer after this move
                # Large weight to bias toward "winning races" for resources
                k = (-(beat * 1000 + myd * 2) + cheb(nx, ny, ox, oy) * 1)
                if key is None or k < key:
                    key = k
            # If multiple moves have similar best resource pick, further prefer opponent distance.
            k2 = (key, cheb(nx, ny, ox, oy), myd if 'myd' in locals() else 0, dx, dy)
            if best_key is None or k2 < best_key:
                best_key = k2
                best_move = (dx, dy)
        return [int(best_move[0]), int(best_move[1])]

    # No visible resources: retreat while drifting toward center deterministically.
    cx, cy = (w - 1) // 2, (h - 1) // 2
    best_key = None
    best_move = (0, 0)
    for dx, dy, nx, ny in legal:
        k = (cheb(nx, ny, cx, cy), -cheb(nx, ny, ox, oy), dx, dy)
        if best_key is None or k < best_key:
            best_key = k
            best_move = (dx, dy)
    return [int(best_move[0]), int(best_move[1])]
