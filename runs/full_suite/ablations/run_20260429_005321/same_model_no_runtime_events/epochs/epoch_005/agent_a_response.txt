    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources") or []
    obs_list = observation.get("obstacles") or []
    obstacles = set((p[0], p[1]) for p in obs_list)

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h

    def dist(ax, ay, bx, by):
        return max(abs(ax - bx), abs(ay - by))

    # Allowed 8-neighbor + stay (deterministic iteration order)
    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1), (0, 0), (0, 1),
             (1, -1), (1, 0), (1, 1)]

    # Direction "front" from us toward opponent (for progress bias)
    fx = 0
    if ox > sx: fx = 1
    elif ox < sx: fx = -1
    fy = 0
    if oy > sy: fy = 1
    elif oy < sy: fy = -1

    cx, cy = (w - 1) // 2, (h - 1) // 2

    legal = []
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if inside(nx, ny) and (nx, ny) not in obstacles:
            legal.append((dx, dy, nx, ny))

    if not legal:
        return [0, 0]

    best_move = (0, 0)
    best_val = -10**18

    def eval_at(px, py):
        # Score best contest target with progress bias toward opponent direction.
        best = -10**18
        for rx, ry in resources:
            if (rx, ry) in obstacles:
                continue
            ds = dist(px, py, rx, ry)
            do = dist(ox, oy, rx, ry)
            lead = do - ds  # positive = we are closer
            # Progress toward opponent from our current pos (discourages drifting backward)
            prog = fx * (rx - px) + fy * (ry - py)
            # Mild anti-overcommit: prefer targets that are not "too far" if lead ties
            closeness = -ds
            # Deterministic tie-breakers via tuple ordering
            val = (lead * 1000) + prog + closeness
            if val > best:
                best = val
        return best
