def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = [tuple(p) for p in observation.get("resources", [])]
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))

    moves = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    moves = [m for m in moves if m != (0, 0)] + [(0, 0)]

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h

    def valid(x, y):
        return inside(x, y) and (x, y) not in obstacles

    def man(x1, y1, x2, y2):
        return abs(x1 - x2) + abs(y1 - y2)

    def adj_obst_pen(x, y):
        pen = 0
        for dx, dy in (-1, 0, 1), (-1, 0, 1):
            pass
        for mx in (-1, 0, 1):
            for my in (-1, 0, 1):
                if mx == 0 and my == 0:
                    continue
                if (x + mx, y + my) in obstacles:
                    pen += 1
        return pen

    if not resources:
        tx, ty = (w - 1) // 2, (h - 1) // 2
        best = None
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny):
                continue
            key = (man(nx, ny, tx, ty), adj_obst_pen(nx, ny))
            if best is None or key < best[0]:
                best = (key, dx, dy)
        return [best[1], best[2]] if best is not None else [0, 0]

    best_key = None
    best_move = (0, 0)
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue

        # Main objective: maximize opponent disadvantage while still progressing for ourselves.
        our_best_d = 10**9
        opp_best_d = 10**9
        # Fractional-ish progress using reciprocal distance (deterministic ints via scaling).
        gain = 0
        opp_near_bonus = 0
        for rx, ry in resources:
            ourd = man(nx, ny, rx, ry)
            opd = man(ox, oy, rx, ry)
            if ourd < our_best_d:
                our_best_d = ourd
            if opd < opp_best_d:
                opp_best_d = opd
            # If we move closer to a resource than the opponent currently is, we get advantage.
            # Scale by a fixed weight to keep all integer arithmetic.
            gain += 40 * (opd - ourd) - 3 * ourd + (1 if ourd == 0 else 0)
            # Also reward moves that make resources "near-us" (where opponent is likely to be contested later).
            if ourd <= 2:
                opp_near_bonus += 3 * max(0, 6 - opd)

        # Secondary objective: deny resources that are very close for opponent compared to ours.
        # Compute a compact contest measure: sum over resources of max(0, ourd - opd) (prefer smaller).
        contest = 0
        for rx, ry in resources:
            ourd = man(nx, ny, rx, ry)
            opd = man(ox, oy, rx, ry)
            if ourd > opd:
                contest += ourd - opd

        # Tertiary: reduce adjacency to obstacles to avoid getting stuck.
        obst = adj_obst_pen(nx, ny)

        # Convert to a single key where smaller is better.
        # We primarily want: large gain, large opp_near_bonus, small our_best_d, small contest, small obst.
        # So use negatives for the "larger is better" parts.
        key = (-gain, -opp_near_bonus, our_best_d, contest, obst)
        if best_key is None or key < best_key:
            best_key = key
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
