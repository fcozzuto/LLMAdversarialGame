def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    moves = []
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if inside(nx, ny):
            moves.append((dx, dy))

    if not moves:
        return [0, 0]

    opp_now = {}
    for r in resources:
        opp_now[r] = cheb(ox, oy, r[0], r[1])

    my_now = {}
    for r in resources:
        my_now[r] = cheb(sx, sy, r[0], r[1])

    if resources:
        best_r = resources[0]
        best_gain = -10**9
        # Prefer resources we can reach no slower than opponent; break ties by opponent distance (more urgent) then resource position.
        for r in resources:
            gain = opp_now[r] - my_now[r]
            if gain > best_gain:
                best_gain = gain
                best_r = r
            elif gain == best_gain:
                if opp_now[r] < opp_now[best_r]:
                    best_r = r
                elif opp_now[r] == opp_now[best_r] and (r[0], r[1]) < (best_r[0], best_r[1]):
                    best_r = r
    else:
        best_r = (sx, sy)

    tx, ty = best_r
    best_move = moves[0]
    best_val = -10**18

    # Add small tie-breakers: keep away from opponent slightly and prefer reducing distance to chosen target.
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        d_target = cheb(nx, ny, tx, ty)

        # Competition signal for target: how close we are vs opponent to same target (approx, opponent might move next).
        opp_d_target = cheb(ox, oy, tx, ty)
        rel = opp_d_target - d_target  # higher is better

        # If target is same as current, don't get stuck; encourage exploring toward any resource we have a gain on.
        explore_bonus = 0
        if (nx, ny) == (sx, sy) and resources and best_r == (sx, sy):
            # scan a couple of resources deterministically
            for r in resources[:8]:
                if cheb(nx, ny, r[0], r[1]) < my_now[r] or (cheb(ox, oy, r[0], r[1]) - cheb(nx, ny, r[0], r[1])) >= 0:
                    explore_bonus += 1
        # Opponent distance (slight)
        d_opp = cheb(nx, ny, ox, oy)

        val = (rel * 1000) - (d_target * 3) + (d_opp * 0.2) + explore_bonus
        if val > best_val:
            best_val = val
            best_move = (dx, dy)
