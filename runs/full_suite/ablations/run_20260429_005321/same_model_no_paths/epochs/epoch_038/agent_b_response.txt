def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h

    def ok(x, y):
        return inside(x, y) and (x, y) not in obstacles

    def cheb(x1, y1, x2, y2):
        ax = x1 - x2
        if ax < 0: ax = -ax
        ay = y1 - y2
        if ay < 0: ay = -ay
        return ax if ax > ay else ay

    def best_capture_target():
        if not resources:
            return None
        best_r, best_margin = None, -10**9
        for x, y in resources:
            d_me = cheb(sx, sy, x, y)
            d_op = cheb(ox, oy, x, y)
            # Prefer resources we can arrive no later than opponent; break ties by closer to us.
            margin = d_op - d_me
            if d_me == d_op:
                margin = margin + (w + h - d_me)
            if margin > best_margin:
                best_margin, best_r = margin, (x, y)
        return best_r

    target = best_capture_target()

    best_move, best_score = (0, 0), -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue

        # Main objective: move toward intercept target if we can plausibly win it.
        if target is not None:
            tx, ty = target
            d_me_next = cheb(nx, ny, tx, ty)
            d_op_now = cheb(ox, oy, tx, ty)
            intercept = (d_op_now - d_me_next)  # higher is better
            dist_to_target = d_me_next
        else:
            intercept = 0
            dist_to_target = 0

        # Risk: avoid getting too close to opponent unless it helps intercept.
        d_op_next = cheb(nx, ny, ox, oy)
        danger = 0
        if d_op_next == 0:
            danger = 1000
        elif d_op_next == 1:
            danger = 120
        elif d_op_next == 2:
            danger = 40

        # Obstacle-adjacent soft penalty.
        adj_block = 0
        for ddx, ddy in moves:
            px, py = nx + ddx, ny + ddy
            if inside(px, py) and (px, py) in obstacles:
                adj_block += 1

        # Tie-break deterministically by preferring moves that reduce target distance, then increase opponent distance, then stable ordering.
        score = 0
        score += intercept * 50
        score += -dist_to_target * 3
        score += d_op_next * 2
        score += -danger
        score += -adj_block * 0.5

        if score > best_score:
            best_score, best_move = score, [dx, dy]
        elif score == best_score:
            # deterministic tie-break: prefer larger d_op_next, then smaller dist_to_target, then lexicographic dx,dy
            if d_op_next > cheb(best_move[0] + sx, best_move[1] + sy, ox, oy):
                best_move = [dx, dy]
