def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def inside(x, y): return 0 <= x < w and 0 <= y < h
    def ok(x, y): return inside(x, y) and (x, y) not in obstacles
    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # New policy vs prior: "race + harassment"
    # 1) If we can beat opponent to some resource, go to the best such resource.
    # 2) Otherwise, move to increase our distance from opponent while still improving approach to their nearest resource,
    #    making it harder for them to secure contested points.
    opp_nearest = None
    if resources:
        bestd = 10**9
        for rx, ry in resources:
            d = cheb(ox, oy, rx, ry)
            if d < bestd:
                bestd = d
                opp_nearest = (rx, ry)

    best_move = (0, 0)
    best_val = -10**18

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue

        if not resources:
            val = cheb(nx, ny, ox, oy)  # flee if no resources known
            if val > best_val:
                best_val = val
                best_move = (dx, dy)
            continue

        # Evaluate races
        best_race_score = -10**18
        for rx, ry in resources:
            d_me = cheb(nx, ny, rx, ry)
            d_op = cheb(ox, oy, rx, ry)
            # race advantage term: strongly prefer being closer than opponent
            adv = d_op - d_me
            # also prefer consuming closer overall to avoid wandering
            score = adv * 20 - d_me
            # tiny bias to head toward resources not currently "at opponent distance"
            if adv > 0:
                score += 5
            if score > best_race_score:
                best_race_score = score

        # If no winning race, use harassment: increase distance to opponent while reducing distance to opponent's nearest
        my_to_opp = cheb(nx, ny, ox, oy)
        if opp_nearest is None:
            harassment = 0
        else:
            rx, ry = opp_nearest
            harassment = -cheb(nx, ny, rx, ry)  # get in their way by being closer too
        val = best_race_score
        # Blend: even when racing, keep from suicidal proximity to opponent
        val += my_to_opp * 2 + harassment * 0.5

        if val > best_val:
            best_val = val
            best_move = (dx, dy)

    dx, dy = best_move
    return [int(dx), int(dy)]
