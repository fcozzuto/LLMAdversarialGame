    def bfs(start, targets):

        dist[start[0]][start[1]] = 0

        while q:

            d = dist[x][y]
            for dx, dy in dirs:

                    q.append((nx, ny))
        best = None
        for tx, ty in targets:
            if dist[tx][ty] is not None:
                dd = dist[tx][ty]
                if best is None or dd < best[0]:
                    best = (dd, tx, ty)
        return best, dist

    def inb(x, y):

    def best_res_dist(x, y):
        if not resources:

        m = 10**9
        for rx, ry in resources:
            d = abs(rx - x) + abs(ry - y)
