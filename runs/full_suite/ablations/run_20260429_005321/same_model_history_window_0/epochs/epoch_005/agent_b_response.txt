def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    obstacles = observation["obstacles"]
    resources = observation["resources"]

    if not resources:
        return [0, 0]

    obs_set = set((x, y) for x, y in obstacles)

    def cheb(a, b, c, d):
        dx = a - c
        dy = b - d
        if dx < 0: dx = -dx
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    res_t = observation["turn_index"]
    prefer_far = (res_t % 2 == 1)

    best_res = None
    best_adv = None
    for rx, ry in resources:
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        adv = do - ds  # positive => we are closer
        if best_adv is None:
            best_adv = adv
            best_res = (rx, ry)
        else:
            if prefer_far:
                # diversify when behind: sometimes choose less contested but also not too far
                key_new = (adv, ds)
                key_old = (best_adv, cheb(sx, sy, best_res[0], best_res[1]))
                if key_new > key_old:
                    best_adv = adv
                    best_res = (rx, ry)
            else:
                key_new = (adv, -do)
                key_old = (best_adv, -cheb(sx, sy, best_res[0], best_res[1]))
                if key_new > key_old:
                    best_adv = adv
                    best_res = (rx, ry)

    tx, ty = best_res
    # If opponent is about to take the chosen resource and we're behind, retarget.
    ds0 = cheb(sx, sy, tx, ty)
    do0 = cheb(ox, oy, tx, ty)
    if do0 <= 1 and ds0 > do0:
        best_res = None
        best_adv = None
        for rx, ry in resources:
            ds = cheb(sx, sy, rx, ry)
            do = cheb(ox, oy, rx, ry)
            adv = do - ds
            # prefer resources that are either ours (adv>=0) or at least not immediate for them
            imm_pen = 0 if do > 1 else -1000
            score = adv + imm_pen
            if best_adv is None or score > best_adv:
                best_adv = score
                best_res = (rx, ry)
        tx, ty = best_res

    moves = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obs_set:
                moves.append((dx, dy, nx, ny))

    best_move = (0, 0, sx, sy)
    best_score = None
    for dx, dy, nx, ny in moves:
        d_to = cheb(nx, ny, tx, ty)
        d_opp = cheb(nx, ny, ox, oy)
        # Prefer getting closer to target, keep distance from opponent to avoid being trapped/contested.
        score = (-d_to) + 0.15 * d_opp
        # If opponent is very close to target, try to steer away to avoid immediate contention.
        if cheb(ox, oy, tx, ty) <= 1:
            score += 0.25 * (d_opp)
        if best_score is None or score > best_score:
            best_score = score
            best_move
