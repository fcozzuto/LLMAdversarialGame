def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    self_pos = observation.get("self_position", [0, 0])
    opp_pos = observation.get("opponent_position", [0, 0])
    resources = observation.get("resources", [])
    obstacles = observation.get("obstacles", [])
    sx, sy = self_pos[0], self_pos[1]
    ox, oy = opp_pos[0], opp_pos[1]

    if [sx, sy] in resources:
        return [0, 0]

    obs = set()
    for p in obstacles:
        if p and len(p) == 2:
            obs.add((p[0], p[1]))
    res_set = set((p[0], p[1]) for p in resources if p and len(p) == 2)

    moves = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    dirs = [m for m in moves if not (m[0] == 0 and m[1] == 0)] + [(0, 0)]

    def neighbors(x, y):
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obs:
                yield nx, ny

    def bfs_dist(start):
        from collections import deque
        INF = 10**9
        dist = [[INF] * h for _ in range(w)]
        q = deque()
        dist[start[0]][start[1]] = 0
        q.append((start[0], start[1]))
        while q:
            x, y = q.popleft()
            d0 = dist[x][y] + 1
            for nx, ny in neighbors(x, y):
                if d0 < dist[nx][ny]:
                    dist[nx][ny] = d0
                    q.append((nx, ny))
        return dist

    ds = bfs_dist((sx, sy))
    do = bfs_dist((ox, oy))

    best = None
    best_score = 10**18
    for rx, ry in res_set:
        d1 = ds[rx][ry]
        d2 = do[rx][ry]
        if d1 >= 10**8:
            continue
        # Prefer reaching earlier; break ties toward closer to self
        score = d1 - 0.9 * d2
        if score < best_score or (score == best_score and d1 < (best_score if best is None else ds[best[0]][best[1]])):
            best_score = score
            best = (rx, ry)

    if best is None:
        # No reachable resources: drift toward center, avoiding obstacles
        tx, ty = (w // 2, h // 2)
    else:
        tx, ty = best

    # Choose next step that minimizes distance to target (greedy using BFS distances)
    best_step = (0, 0)
    best_d = ds[tx][ty] + 1
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not (0 <= nx < w and 0 <= ny < h) or (nx, ny) in obs:
            continue
        if ds[nx][ny] == 10**9:
            continue
        # We need distance to target; use ds from target by running BFS once if needed
        # Since grid is tiny, run BFS from target deterministically.
    dist_to_target = bfs_dist((tx, ty))
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not (0 <= nx < w and 0 <= ny < h) or (nx, ny) in obs:
            continue
        d = dist_to_target[nx][ny]
        if d < best_d:
            best_d = d
            best_step = (dx, dy)

    return [int(best_step[0]), int(best_step[1])]
