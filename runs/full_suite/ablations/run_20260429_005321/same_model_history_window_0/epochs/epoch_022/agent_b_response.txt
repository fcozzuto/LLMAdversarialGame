def choose_move(observation):
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation["resources"]
    obstacles = set(tuple(p) for p in observation["obstacles"])
    candidates = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            candidates.append((dx, dy, nx, ny))

    def dist2(a, b):
        ax, ay = a
        bx, by = b
        dx = ax - bx
        dy = ay - by
        return dx * dx + dy * dy

    # Choose a target resource where we have an advantage
    if resources:
        best = None
        best_key = None
        for r in resources:
            rs = dist2((sx, sy), r)
            ro = dist2((ox, oy), r)
            # Prefer: self closer than opponent, then overall closer
            key = (rs > ro, ro - rs, -(rs), r[0] + 3 * r[1])
            if best_key is None or key > best_key:
                best_key = key
                best = r
        target = best
    else:
        target = (sx, sy)

    # If opponent is closer to all resources, switch to a different objective:
    # move to increase separation while still drifting toward the farthest resource.
    if resources:
        self_closests = sorted(resources, key=lambda r: dist2((sx, sy), r))
        opp_closests = sorted(resources, key=lambda r: dist2((ox, oy), r))
        r_self = self_closests[0]
        r_opp = opp_closests[0]
        mode = "adv"
        if dist2((sx, sy), r_self) >= dist2((ox, oy), r_opp):
            mode = "def"
    else:
        mode = "def"

    best_move = (0, 0, -10**18)
    for dx, dy, nx, ny in candidates:
        # out of bounds treated as invalid; engine would keep us in place
        if nx < 0 or nx >= observation["grid_width"] or ny < 0 or ny >= observation["grid_height"]:
            continue
        if (nx, ny) in obstacles:
            continue

        ns = (nx, ny)
        d_t = dist2(ns, target)
        d_o = dist2(ns, (ox, oy))

        # Use opponent target estimate: nearest resource for opponent
        if resources:
            opp_target = min(resources, key=lambda r: dist2((ox, oy), r))
            d_opp_to_oppT = dist2((ox, oy), opp_target)
            d_opp_after = dist2((ox + dx, oy + dy), opp_target)  # rough intercept pressure proxy
        else:

            d_opp_to_oppT = 10**9
