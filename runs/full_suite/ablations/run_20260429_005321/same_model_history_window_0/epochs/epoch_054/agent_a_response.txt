def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    obstacles = set((p[0], p[1]) for p in observation["obstacles"])
    resources = observation["resources"]

    def in_bounds(x, y):
        return 0 <= x < w and 0 <= y < h

    def manh(a, b, c, d):
        dx = a - c
        if dx < 0: dx = -dx
        dy = b - d
        if dy < 0: dy = -dy
        return dx + dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    viable = []
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if in_bounds(nx, ny) and (nx, ny) not in obstacles:
            viable.append((dx, dy, nx, ny))
    if not viable:
        return [0, 0]

    # Target selection: prefer resources where we can "arrive" no later than opponent (Manhattan proxy).
    best_resources = []
    for rx, ry in resources:
        ds = manh(rx, ry, sx, sy)
        do = manh(rx, ry, ox, oy)
        score_gap = ds - do  # <=0 means we are not worse
        best_resources.append((score_gap, ds, rx, ry))
    best_resources.sort()
    chosen = None
    for gap, ds, rx, ry in best_resources:
        if gap <= 0:
            chosen = (rx, ry)
            break
    if chosen is None:
        # If we are behind everywhere, contest: pick a resource that the opponent is closest to (we need to disrupt).
        best_resources.sort(key=lambda t: (t[1] - 0, t[0]))  # keep deterministic
        # Choose resource maximizing opponent closeness advantage against us, but also not too far.
        best = None
        for gap, ds, rx, ry in best_resources:
            do = manh(rx, ry, ox, oy)
            cand = (do, -ds)  # opponent close, we far => strong contest target
            if best is None or cand > best[0]:
                best = (cand, (rx, ry))
        chosen = best[1]

    # Evaluate moves deterministically: go toward chosen, avoid getting too close to opponent.
    rx, ry = chosen
    best_move = None
    best_val = None
    for dx, dy, nx, ny in viable:
        d_self = manh(nx, ny, rx, ry)
        d_opp = manh(nx, ny, ox, oy)
        # Also lightly prefer staying away from edges if tied, to keep movement flexibility.
        edge_pen = min(nx, ny, (w - 1) - nx, (h - 1) - ny)
        # Opponent avoidance weight: slightly more aggressive if we are behind on the chosen target.
        ds_here = manh(rx, ry, nx, ny)
        do_target = manh(rx, ry, ox, oy)
        behind = 1 if ds_here > do_target else 0
        avoid = (10 - d_opp) * (2 + behind)  # higher when closer to opponent
        # Deterministic tie-breaker via dx,dy ordering
        val = (d_self + avoid * 0.75, -edge_pen, -nx, -ny, dx, dy)
        if best_val is None or val < best_val:
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
