def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [w - 1, h - 1])
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def clamp(x, y):
        if x < 0: x = 0
        elif x >= w: x = w - 1
        if y < 0: y = 0
        elif y >= h: y = h - 1
        return x, y

    def man(ax, ay, bx, by):
        return abs(ax - bx) + abs(ay - by)

    # Candidate moves (dx,dy)
    moves = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            if dx == 0 and dy == 0:
                moves.append((0, 0))
            else:
                nx, ny = sx + dx, sy + dy
                moves.append((dx, dy))

    # If no resources, move away from opponent slightly to avoid immediate contention
    if not resources:
        best = None
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            nx, ny = clamp(nx, ny)
            if (nx, ny) in obstacles:
                nx, ny = sx, sy
            score = man(nx, ny, ox, oy) - 0.1 * man(nx, ny, (w - 1 - ox), (h - 1 - oy))
            if best is None or score > best[0]:
                best = (score, dx, dy)
        return [int(best[1]), int(best[2])]

    # Evaluate each move by targeting a resource where we can arrive earlier than opponent, deterministically.
    def obstacle_pen(nx, ny):
        # Penalize stepping adjacent to obstacles to reduce collisions
        pen = 0
        for (ax, ay) in obstacles:
            if abs(ax - nx) <= 1 and abs(ay - ny) <= 1:
                pen += 0.25
        return pen

    best_move = None
    best_val = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        nx, ny = clamp(nx, ny)
        if (nx, ny) in obstacles:
            nx, ny = sx, sy

        # Choose the best resource for this move; then rank the move by the resulting advantage.
        move_best = None
        for rx, ry in sorted(resources):
            our_t = man(nx, ny, rx, ry)
            opp_t = man(ox, oy, rx, ry)

            # Core objective: win contested resources by arriving earlier.
            # Secondary: finish sooner; tertiary: parity tie-break toward lower coordinates.
            adv = opp_t - our_t
            finish = -our_t
            parity = -((rx + ry) & 1)
            val = (adv * 10.0) + (finish * 1.0) + parity - obstacle_pen(nx, ny)

            # Additional deterministic nudge: avoid committing to resources in the immediate vicinity of opponent
            if man(nx, ny, ox, oy) <= 1 and man(ox, oy, rx, ry) <= man(sx, sy, rx, ry):
                val -= 2.5

            if move_best is None or val > move_best:
                move_best = val

        if best_val is None or move_best > best_val:
            best_val = move_best
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
