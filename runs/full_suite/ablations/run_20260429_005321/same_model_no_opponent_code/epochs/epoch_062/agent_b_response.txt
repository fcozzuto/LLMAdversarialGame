def choose_move(observation):
    w = int(observation.get("grid_width", 8))
    h = int(observation.get("grid_height", 8))
    sp = observation.get("self_position", [0, 0])
    op = observation.get("opponent_position", [0, 0])
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles", []) or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            obstacles.add((int(p[0]), int(p[1])))

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h

    def legal(dx, dy):
        nx, ny = sx + dx, sy + dy
        return inside(nx, ny) and (nx, ny) not in obstacles

    cands = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            if legal(dx, dy):
                cands.append((dx, dy))
    if not cands:
        return [0, 0]

    resources = observation.get("resources", []) or []
    rt = []
    for r in resources:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if inside(x, y):
                rt.append((x, y))
        elif isinstance(r, int):
            x, y = r % w, r // w
            if inside(x, y):
                rt.append((x, y))

    def md(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx + dy

    # If adjacent to any resource, prioritize collecting it.
    for dx, dy in cands:
        nx, ny = sx + dx, sy + dy
        if (nx, ny) in set(rt):
            return [dx, dy]

    # Target resource that we can reach earlier than opponent (or least worse).
    best_target = None
    best_score = None
    for tx, ty in rt:
        myd = md(sx, sy, tx, ty)
        opd = md(ox, oy, tx, ty)
        # Positive means opponent is closer; we prefer lower/negative.
        score = (myd - opd) * 10 + myd
        if best_score is None or score < best_score:
            best_score = score
            best_target = (tx, ty)

    if best_target is None:
        # No resources visible: move away from opponent and toward center.
        cx, cy = (w - 1) // 2, (h - 1) // 2
        best = None
        best_val = None
        for dx, dy in cands:
            nx, ny = sx + dx, sy + dy
            val = -md(nx, ny, ox, oy) + (md(nx, ny, cx, cy) // 2)
            if best_val is None or val > best_val:
                best_val = val
                best = (dx, dy)
        return [best[0], best[1]]

    tx, ty = best_target
    # Among legal moves, minimize our distance to target while keeping advantage.
    best_move = cands[0]
    best_val = None
    for dx, dy in cands:
        nx, ny = sx + dx, sy + dy
        myd = md(nx, ny, tx, ty)
        opd = md(ox, oy, tx, ty)
        # Prefer moves that keep us not worse; tie-break by increasing opponent distance to target.
        val = myd - opd
        tie = myd + (md(nx, ny, ox, oy) // 3)
        combined = val * 100 + tie
        if best_val is None or combined < best_val:
            best_val = combined
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
