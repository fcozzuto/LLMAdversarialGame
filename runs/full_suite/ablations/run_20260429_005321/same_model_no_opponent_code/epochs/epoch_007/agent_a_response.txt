def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", []) or []
    obstacles = set((p[0], p[1]) for p in observation.get("obstacles", []) or [])

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def blocked(x, y):
        return not inb(x, y) or (x, y) in obstacles

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    cand = []
    safe_moves = {}
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if blocked(nx, ny):
            continue
        cnt = 0
        for ex in (-1, 0, 1):
            for ey in (-1, 0, 1):
                if ex == 0 and ey == 0:
                    continue
                tx, ty = nx + ex, ny + ey
                if not blocked(tx, ty):
                    cnt += 1
        safe_moves[(dx, dy)] = cnt
        cand.append((dx, dy))

    if not resources:
        best = (0, 0)
        bestv = -10**9
        for dx, dy in cand:
            nx, ny = sx + dx, sy + dy
            v = safe_moves[(dx, dy)]
            # drift away from opponent when no resources
            dfo = cheb(nx, ny, ox, oy)
            v += dfo * 0.1
            if v > bestv or (v == bestv and (dx, dy) < best):
                bestv = v
                best = (dx, dy)
        return [best[0], best[1]]

    # Choose best resource deterministically: minimize distance, then tie-break by coordinates
    best_res = None
    best_d = 10**9
    for rx, ry in resources:
        d = cheb(sx, sy, rx, ry)
        if d < best_d or (d == best_d and (rx, ry) < (best_res[0], best_res[1]) if best_res else True):
            best_d = d
            best_res = (rx, ry)

    rx, ry = best_res
    near_target = (best_d <= 1)
    best = (0, 0)
    bestv = -10**18

    for dx, dy in cand:
        nx, ny = sx + dx, sy + dy
        d_to = cheb(nx, ny, rx, ry)
        d_from_opp = cheb(nx, ny, ox, oy)
        v = 0
        # Main objective: get to a resource quickly
        v += (0 if near_target else 50) - d_to * 10
        # If moving onto/adjacent to a resource becomes available, boost strongly
        if any(cheb(nx, ny, r[0], r[1]) == 0 for r in resources):
            v += 2000
        if any(cheb(nx, ny, r[0], r[1]) == 1 for r in resources):
            v += 120
        # Avoid being cornered
        v += safe_moves[(dx, dy)] * 2
        # Don't let opponent get too close to the same area
        v += d_from_opp * 0.4
        # Small tie-breaker: reduce distance to (rx,ry) in a stable way
        v += (0 - (d_to * 0.01))
        if v > bestv or (v == bestv and (dx, dy) < best):
            bestv = v
            best = (dx, dy)

    return [best[0], best[1]]
