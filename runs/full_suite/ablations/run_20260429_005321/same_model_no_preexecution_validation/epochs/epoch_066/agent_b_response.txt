def choose_move(observation):
    w = int(observation["grid_width"]); h = int(observation["grid_height"])
    sx, sy = observation["self_position"]; ox, oy = observation["opponent_position"]
    resources = observation.get("resources") or []
    obstacles = observation.get("obstacles") or []
    obs = {(int(p[0]), int(p[1])) for p in obstacles}
    res = [(int(p[0]), int(p[1])) for p in resources]

    dirs = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    def inb(x, y): return 0 <= x < w and 0 <= y < h
    def cheb(x1, y1, x2, y2):
        dx = x1 - x2; dx = dx if dx >= 0 else -dx
        dy = y1 - y2; dy = dy if dy >= 0 else -dy
        return dx if dx > dy else dy
    def blocked(x, y): return (not inb(x, y)) or ((x, y) in obs)
    def near_obs(x, y):
        c = 0
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if (nx, ny) in obs: c += 1
        return c

    candidates = []
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not blocked(nx, ny):
            candidates.append((dx, dy, nx, ny))
    if not candidates:
        return [0, 0]

    best = (0, 0); best_val = -10**18
    opp_dist_now = cheb(sx, sy, ox, oy)

    if res:
        for dx, dy, nx, ny in candidates:
            my_to_opp = cheb(nx, ny, ox, oy)
            obs_pen = near_obs(nx, ny)
            # Prefer moves that make us reach a resource no later than the opponent, and get us close.
            best_res_val = -10**18
            for rx, ry in res:
                myd = cheb(nx, ny, rx, ry)
                opd = cheb(ox, oy, rx, ry)
                reach_adv = opd - myd  # positive means we can get there first/sooner (in chebyshev steps)
                # Slightly favor smaller myd to finish quickly after securing reach.
                val = 3.0 * reach_adv - 1.0 * myd
                if val > best_res_val: best_res_val = val
            # If opponent is very close, prefer not to drift toward them unless resource advantage is strong.
            opp_pen = 0.0
            if my_to_opp <= 2: opp_pen = 2.5 * (2 - my_to_opp)
            # Also nudge toward overall resource closeness even if reach_adv is similar.
            avg_myd = 0
            # deterministic small sample: check up to 4 resources in given order
            for i in range(min(4, len(res))):
                rx, ry = res[i]
                avg_myd += cheb(nx, ny, rx, ry)
            avg_myd /= float(min(4, len(res)))
            total = best_res_val - 0.8 * obs_pen - opp_pen - 0.15 * avg_myd
            if total > best_val:
                best_val = total
                best = (dx, dy)
