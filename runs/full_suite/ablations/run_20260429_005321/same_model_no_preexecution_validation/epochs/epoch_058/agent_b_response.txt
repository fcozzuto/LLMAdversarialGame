def choose_move(observation):
    w = observation["grid_width"]; h = observation["grid_height"]
    sx, sy = observation["self_position"]; ox, oy = observation["opponent_position"]
    resources = observation.get("resources") or []
    obstacles = observation.get("obstacles") or []
    obs = set((p[0], p[1]) for p in obstacles)

    def inb(x, y): return 0 <= x < w and 0 <= y < h
    def blocked(x, y): return (not inb(x, y)) or ((x, y) in obs)

    if blocked(sx, sy):
        return [0, 0]
    dirs8 = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    INF = 10**9

    def bfs(start):
        if blocked(start[0], start[1]): return None
        dist = [[INF] * w for _ in range(h)]
        dist[start[1]][start[0]] = 0
        qx = [start[0]]; qy = [start[1]]; qi = 0
        while qi < len(qx):
            x, y = qx[qi], qy[qi]; qi += 1
            nd = dist[y][x] + 1
            for dx, dy in dirs8:
                nx, ny = x + dx, y + dy
                if not blocked(nx, ny) and nd < dist[ny][nx]:
                    dist[ny][nx] = nd
                    qx.append(nx); qy.append(ny)
        return dist

    myd = bfs((sx, sy))
    if myd is None or not resources:
        return [0, 0]
    opd = bfs((ox, oy))
    if opd is None:
        opd = [[INF] * w for _ in range(h)]

    best = None
    best_score = -10**18
    for rx, ry in resources:
        d1 = myd[ry][rx]
        if d1 >= INF:
            continue
        d2 = opd[ry][rx]
        lead = d2 - d1  # positive means we arrive earlier
        # Heuristic: prioritize resources we can win for sure; otherwise, minimize our time.
        score = 0
        if lead > 0:
            score = 100000 + lead * 100 - d1
        else:
            score = 5000 - d1 * 3 + lead * 10
        # Small deterministic tie-breaker
        score -= (rx * 3 + ry * 5) * 1e-6
        if score > best_score:
            best_score = score
            best = (rx, ry, d1)

    if best is None:
        # Fallback: move toward nearest reachable resource by greedy distance
        target = None; bestd = INF
        for rx, ry in resources:
            d1 = myd[ry][rx]
            if d1 < bestd:
                bestd = d1; target = (rx, ry)
        if target is None or bestd >= INF:
            return [0, 0]
        rx, ry = target
    else:
        rx, ry, _ = best

    curd = myd[sy][sx]
    best_move = [0, 0]
    best_nd = curd
    # Prefer moves that reduce our distance to the chosen target and also don't step toward opponent path (softly).
    for dx, dy in dirs8:
        nx, ny = sx + dx, sy + dy
        if blocked(nx, ny):
            continue
        d_to_target = myd[ny][nx]
        if d_to_target < myd[ry][rx] and d_to_target >= INF:
            continue
        nd = myd[ny][nx]
        if nd < best_nd:
            best_nd = nd; best_move = [dx, dy]
        elif nd == best_nd:

            # (use only dist values; deterministic)
