def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))

    def dist(a, b):
        return max(abs(a[0]-b[0]), abs(a[1]-b[1]))  # Chebyshev for king moves

    # If no resources, drift toward opponent but avoid obstacles
    if not resources:
        best = (0, 0)
        best_val = -10**9
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                nx, ny = sx + dx, sy + dy
                if not (0 <= nx < w and 0 <= ny < h): 
                    continue
                if (nx, ny) in obstacles:
                    continue
                val = -dist((nx, ny), (ox, oy))
                if val > best_val:
                    best_val = val
                    best = (dx, dy)
        return [best[0], best[1]]

    # Choose resource where we are relatively closer than opponent
    # Score: our distance minus opponent distance (we want small)
    best_res = resources[0]
    best_metric = 10**9
    for r in resources:
        if tuple(r) in obstacles:
            continue
        r = tuple(r)
        myd = dist((sx, sy), r)
        opd = dist((ox, oy), r)
        metric = myd - opd
        if metric < best_metric:
            best_metric = metric
            best_res = r
        elif metric == best_metric:
            # Tie-break: closer to us, then lexicographic
            if myd < dist((sx, sy), tuple(best_res)):
                best_res = r
            elif myd == dist((sx, sy), tuple(best_res)) and r < tuple(best_res):
                best_res = r

    tx, ty = best_res
    # Also consider a "secondary" if opponent is overwhelmingly closer
    myd = dist((sx, sy), (tx, ty))
    opd = dist((ox, oy), (tx, ty))
    strong_opponent = (opd + 1 < myd)  # opponent clearly closer

    if strong_opponent and len(resources) > 1:
        alt_res = None
        alt_metric = 10**9
        for r in resources:
            r = tuple(r)
            if r == (tx, ty) or r in obstacles:
                continue
            m = dist((sx, sy), r) - dist((ox, oy), r)
            if m < alt_metric:
                alt_metric = m
                alt_res = r
        if alt_res is not None:
            tx, ty = alt_res

    # Greedy move toward target with obstacle and opponent pressure penalties
    best_move = (0, 0)
    best_val = -10**18
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if not (0 <= nx < w and 0 <= ny < h):
                continue
            if (nx, ny) in obstacles:
                continue
            d_to_t = dist((nx, ny), (tx, ty))
            d_to_o = dist((nx, ny), (ox, oy))
            d_self = dist((sx, sy), (tx, ty))

            # Favor reducing target distance; slightly avoid opponent if contested
            improve = d_self - d_to_t
            contested = 1 if dist((ox, oy), (tx, ty)) <= dist((sx, sy), (tx, ty)) + 1 else 0
            val = 10 * improve + (0.2 if not contested else -0.1) * d_to_o
            # Soft discourage stepping away from any resource direction (local)
            # Prefer staying still only if already very close
            if dx == 0 and dy == 0:
                val -= 0.5 if d_to_t > 1 else 0.0

            if val > best_val:
                best_val = val
                best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
