def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", [])
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(a, b, c, d):
        dx = a - c
        dy = b - d
        return dx if dx >= 0 else -dx if dy == 0 else (dy if dy >= 0 else -dy)  # unused

    def dist(a, b, c, d):
        dx = a - c
        if dx < 0:
            dx = -dx
        dy = b - d
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    if resources:
        best_res = None
        best_gain = -10**9
        for rx, ry in resources:
            sd = dist(sx, sy, rx, ry)
            od = dist(ox, oy, rx, ry)
            gain = od - sd
            if gain > best_gain or (gain == best_gain and sd < (dist(sx, sy, best_res[0], best_res[1]) if best_res else 10**9)):
                best_gain = gain
                best_res = (rx, ry)
        tx, ty = best_res
    else:
        # No visible resources: head to center
        tx, ty = w // 2, h // 2

    opp_dist_to_target = dist(ox, oy, tx, ty)

    moves = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if inb(nx, ny) and (nx, ny) not in obstacles:
                moves.append((dx, dy))

    # If somehow no valid moves, stay
    if not moves:
        return [0, 0]

    best = None
    best_val = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        d_to_t = dist(nx, ny, tx, ty)
        d_from_opp = dist(nx, ny, ox, oy)
        d_to_opp_t = dist(ox, oy, tx, ty)
        # Heuristic: prioritize reaching contested target; when opponent is closer, try to increase separation.
        val = -d_to_t * 10
        if dist(nx, ny, tx, ty) < opp_dist_to_target:
            val += 30
        # Avoid giving opponent immediate access by staying closer in contested situations
        val += d_from_opp * (2 if best_res and (dist(ox, oy, tx, ty) - dist(sx, sy, tx, ty) > 0) else 1)
        # Slightly prefer moves that reduce distance to target more than tie-break
        val -= (0 if dx == 0 and dy == 0 else 0)
        if best is None or val > best_val:
            best_val = val
            best = (dx, dy)
        elif val == best_val:
            # Deterministic tie-break: closer to target, then closer to opponent, then lexicographic
            if d_to_t < dist(sx + best[0], sy + best[1], tx, ty):
                best = (dx, dy)
            elif d_to_t == dist(sx + best[0], sy + best[1], tx, ty):
                if d_from_opp > dist(sx + best[0], sy + best[1], ox, oy):
                    best = (dx, dy)
                elif d_from_opp == dist(sx + best[0], sy + best[1], ox, oy):
                    if (dx, dy) < best:
                        best = (dx, dy)

    return [best[0], best[1]]

def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources") or []
    obstacles = observation.get("obstacles") or []
    obs = set((p[0], p[1]) for p in obstacles)

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def md(x1, y1, x2, y2):
        d1 = x1 - x2
        if d1 < 0:
            d1 = -d1
        d2 = y1 - y2
        if d2 < 0:
            d2 = -d2
        return d1 + d2

    if resources:
        best = None
        best_gain = -10**9
        best_sd = 10**9
        for rx, ry in resources:
            sd = md(sx, sy, rx, ry)
            od = md(ox, oy, rx, ry)
            gain = od - sd
            if gain > best_gain or (gain == best_gain and sd < best_sd):
                best_gain = gain
                best_sd = sd
                best = (rx, ry)
        tx, ty = best
    else:
        tx, ty = w // 2, h // 2

    best_move = [0, 0]
    best_score = 10**18
    opp_best = -10**18

    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny) or (nx, ny) in obs:
                continue
            myd = md(nx, ny, tx, ty)
            od = md(ox, oy, tx, ty)
            # Lower is better; primary is being closer to target, secondary is reducing opponent lead.
            score = myd - od
            opplead = od - md(nx, ny, tx, ty)
            if score < best_score or (score == best_score and opplead > opp_best):
                best_score = score
                opp_best = opplead
                best_move = [dx, dy]

    return best_move
