def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)

    def to_xy(p):
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            return (int(p[0]), int(p[1]))
        if isinstance(p, dict):
            if "position" in p:
                q = p["position"]
                if isinstance(q, (list, tuple)) and len(q) >= 2:
                    return (int(q[0]), int(q[1]))
            if "x" in p and "y" in p:
                return (int(p["x"]), int(p["y"]))
        return None

    self_p = to_xy(observation.get("self_position", (0, 0))) or (0, 0)
    opp_p = to_xy(observation.get("opponent_position", (0, 0))) or (0, 0)
    sx, sy = self_p
    ox, oy = opp_p

    obst = set()
    for p in observation.get("obstacles", []) or []:
        q = to_xy(p)
        if q is not None:
            obst.add(q)

    res = []
    for p in observation.get("resources", []) or []:
        q = to_xy(p)
        if q is not None:
            res.append(q)

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx >= dy else dy

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    # Choose target resource: prioritize ones I'm closer to; then maximize lead.
    target = None
    best = None
    for rx, ry in res:
        myd = cheb(sx, sy, rx, ry)
        opd = cheb(ox, oy, rx, ry)
        # If tie, prefer resource with larger distance from opponent path advantage (simple tie-break).
        lead = opd - myd  # positive means I am closer
        score = (lead, -myd, -(abs(rx - w // 2) + abs(ry - h // 2)))
        if best is None or score > best:
            best = score
            target = (rx, ry)

    if target is None:
        target = (w // 2, h // 2)

    tx, ty = target

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    oppd = cheb(ox, oy, tx, ty)

    # Move greedily toward target while avoiding obstacles; also avoid stepping too close if it would likely lose.
    best_move = (0, 0)
    best_val = -10**9
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obst:
            continue
        nd = cheb(nx, ny, tx, ty)
        # If opponent is closer to this target than we are, don't chase directly; angle to reduce our distance to improve lead.
        chase_pen = 0
        if mydist := cheb(sx, sy, tx, ty):
            chase_pen = 0
        # Safety/spacing term to reduce collisions: keep some distance from opponent.
        nopp = cheb(nx, ny, ox, oy)
        val = -nd
        val += 0.2 * (nopp)
        # Encourage improving lead when currently behind.
        cur_lead = oppd - mydist
        new_lead = cheb(ox, oy, tx, ty) - nd
        val += 0.8 * (new_lead - cur_lead)
        if (dx, dy) == (0, 0):
            val -= 0.05
        if val > best_val:
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
