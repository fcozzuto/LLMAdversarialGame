def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if inb(x, y):
                resources.append((x, y))

    cx, cy = w // 2, h // 2

    def best_resource():
        if not resources:
            return (cx, cy), 0
        best_t = resources[0]
        best_val = -10**9
        for tx, ty in resources:
            ds = cheb(sx, sy, tx, ty)
            do = cheb(ox, oy, tx, ty)
            # Prefer resources I'm closer to; break ties toward ones slightly away from opponent.
            val = (do - ds) * 10 - ds
            # Small preference for nearer center to reduce path risk.
            val -= 0 if (tx == cx and ty == cy) else cheb(tx, ty, cx, cy) * 0.1
            if val > best_val:
                best_val = val
                best_t = (tx, ty)
        return best_t, best_val

    target, _ = best_resource()
    tx, ty = target

    # If opponent is adjacent to my target, try to flank: aim for a neighbor of target that I can reach sooner.
    if cheb(ox, oy, tx, ty) <= 1:
        neigh = [(tx + dx, ty + dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1) if not (dx == 0 and dy == 0)]
        cand = [p for p in neigh if inb(p[0], p[1])]
        if cand:
            best = cand[0]
            bestv = -10**9
            for px, py in cand:
                ds = cheb(sx, sy, px, py)
                do = cheb(ox, oy, px, py)
                v = (do - ds) * 10 - ds
                if v > bestv:
                    bestv = v
                    best = (px, py)
            tx, ty = best

    best_move = (0, 0)
    best_eval = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny):
            continue
        d_me = cheb(nx, ny, tx, ty)
        d_op = cheb(nx, ny, ox, oy)
        # Move to reduce distance to target, while keeping away from opponent when possible.
        eval = -d_me * 20 + d_op
        if (nx, ny) == (tx, ty):
            eval += 1000
        # Deterministic tie-break: lexicographically favor smaller dx, then smaller dy.
        if eval > best_eval or (eval == best_eval and (dx, dy) < best_move):
            best_eval = eval
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]

    def dist(a, b):
        dx = a[0] - b[0]
        if dx < 0:
            dx = -dx
        dy = a[1] - b[1]
        if dy < 0:

    if resources:
        best = None
        best_score = None
        for tx, ty in resources:
            myd = dist((sx, sy), (tx, ty))
            opd = dist((ox, oy), (tx, ty))
            score = myd - (opd * 0.9)
            if best is None or score < best_score or (score == best_score and (tx, ty) < best):
                best = (tx, ty)
                best_score = score

    best_val = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):

        # Prefer moving closer, and prefer positions where opponent is relatively farther.
        val = myd - (opd * 0.9)
