def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []

    obst = set()
    for p in obstacles:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            obst.add((p[0], p[1]))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx >= dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # If no resources, just drift to center while keeping a bit of space.
    if not resources:
        tx, ty = w // 2, h // 2
        best = (0, -10**9, 0, 0)
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny) or (nx, ny) in obst:
                continue
            myd = cheb(nx, ny, tx, ty)
            oppd = cheb(nx, ny, ox, oy)
            key = (-myd, oppd, dx, dy)
            if key > best[1:]:
                best = (0, key[1], dx, dy)
        # best stores dx,dy in last two positions
        return [best[2], best[3]]

    # Choose best target resource deterministically: maximize advantage (opp_dist - my_dist).
    best_res = None
    best_adv = None
    for rx, ry in resources:
        if not inb(rx, ry):
            continue
        myd = cheb(sx, sy, rx, ry)
        oppd = cheb(ox, oy, rx, ry)
        adv = oppd - myd
        # tie-break: smaller myd, then lexicographic
        key = (adv, -myd, -rx, -ry)
        if best_adv is None or key > best_adv:
            best_adv = key
            best_res = (rx, ry)

    rx, ry = best_res
    myd0 = cheb(sx, sy, rx, ry)
    oppd0 = cheb(ox, oy, rx, ry)

    # Evaluate one-step moves: prefer hitting/approaching target and denying opponent.
    best_move = (0, 0)
    best_score = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obst:
            continue

        myd = cheb(nx, ny, rx, ry)
        oppd = cheb(ox, oy, rx, ry)  # opponent stays this turn in delta evaluation

        hit = 1 if (nx, ny) == (rx, ry) else 0

        # Secondary: avoid moving near obstacles poorly (simple penalty)
        near_obst = 0
        for ax in (-1, 0, 1):
            for ay in (-1, 0, 1):
                if ax == 0 and ay == 0:
                    continue
                xx, yy = nx + ax, ny + ay
                if (xx, yy) in obst:
                    near_obst += 1

        # Primary score components
        adv = oppd - myd
        dist_improve = myd0 - myd
        score = (
            hit * 1000000,
            adv,
            dist_improve,
            -myd,
            -near_obst,
            -cheb(nx, ny, ox, oy),  # also avoid getting too close to opponent
            dx,
            dy,
        )

        if best_score is None or score > best_score:
            best_score = score
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]

        return dx if dx >= dy else dy

    def get_xy(p):

            if "x" in p and "y" in p:

            return (p[0], p[1])
        return None

        if q is not None:
            obst.add(q)

        q = get_xy(p)
        if q is not None:
            res.append(q)

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    if res:
