def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []

    def to_xy(p):
        if isinstance(p, dict):
            if "position" in p and isinstance(p["position"], (list, tuple)) and len(p["position"]) >= 2:
                return (p["position"][0], p["position"][1])
            if "x" in p and "y" in p:
                return (p["x"], p["y"])
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            return (p[0], p[1])
        return None

    def inb(x, y): return 0 <= x < w and 0 <= y < h
    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    obst = set()
    for p in obstacles:
        q = to_xy(p)
        if q is not None:
            obst.add(q)

    res = []
    for p in resources:
        q = to_xy(p)
        if q is not None:
            res.append(q)

    deltas = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]

    if not res:
        dx = 0
        dy = 0
        # fallback: drift to center
        tx, ty = (w - 1) / 2, (h - 1) / 2
        if sx < tx: dx = 1
        elif sx > tx: dx = -1
        if sy < ty: dy = 1
        elif sy > ty: dy = -1
        nx, ny = sx + dx, sy + dy
        if inb(nx, ny) and (nx, ny) not in obst:
            return [dx, dy]
        # obstacle-free nearest delta
        best = [0, 0]
        bestv = -10**9
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny) or (nx, ny) in obst: continue
            v = -cheb(nx, ny, tx, ty)
            if v > bestv:
                bestv = v
                best = [dx, dy]
        return best

    # Choose a resource that is hard for opponent but easy for us to reach (deny if possible)
    best = None
    best_score = -10**18
    for rx, ry in res:
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        # Prefer winning/denial: maximize (do - ds), then closeness
        score = (do - ds) * 100 - ds
        if score > best_score:
            best_score = score
            best = (rx, ry)

    rx, ry = best
    # Prefer steps that reduce our distance while not walking into obstacles; also avoid giving opponent a shortcut
    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obst: 
            continue
        ds_new = cheb(nx, ny, rx, ry)
        do_new = cheb(nx, ny, ox, oy)  # crude: being near opponent can be bad
        # Also consider whether this step blocks their likely approach to the target
        do_target_new = cheb(ox, oy, rx, ry)
        val = -ds_new * 10 - do_new * 0.1 + (do_target_new - ds_new)
        # deterministic tie-breaker: lexicographic preference
        if val > best_val or (val == best_val and (dx, dy) < best_move):
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
