def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sx, sy = observation.get("self_position") or [0, 0]
    ox, oy = observation.get("opponent_position") or [0, 0]
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))
    if not resources:
        return [0, 0]

    def manh(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx + dy

    def obstacle_pressure(x, y):
        # Penalize paths that go near obstacles by counting obstacles in a small neighborhood
        # around the "midpoint direction" from self to target.
        mx = (sx + x) // 2
        my = (sy + y) // 2
        pen = 0
        for dx in (-1, 0, 1):
            for dy in (-1, 0, 1):
                xx = mx + dx
                yy = my + dy
                if (xx, yy) in obstacles:
                    pen += 2
        return pen

    # Choose a resource that maximizes (opponent advantage) while discouraging obstacle-heavy corridors.
    best = None
    best_key = None
    for rx, ry in resources:
        ds = manh(sx, sy, rx, ry)
        do = manh(ox, oy, rx, ry)
        # Prefer resources where we are not worse; also break ties towards lower ds to finish.
        adv = do - ds
        key = (adv, -(ds + obstacle_pressure(rx, ry)), -rx, -ry)
        if best_key is None or key > best_key:
            best_key = key
            best = (rx, ry)

    tx, ty = best
    # Move step maximizing progress to target while avoiding obstacles and preventing giving opponent a direct capture.
    dirs = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            dirs.append((dx, dy))

    # Determine a simple "opponent pressure" direction: if opponent is closer to target than us,
    # prioritize moves that increase distance to opponent (while still going to target if possible).
    base_ds = manh(sx, sy, tx, ty)
    base_do = manh(ox, oy, tx, ty)
    opp_is_closer = base_do < base_ds

    best_move = (0, 0)
    best_score = None
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not (0 <= nx < w and 0 <= ny < h):
            continue
        if (nx, ny) in obstacles:
            continue
        d_to_target = manh(nx, ny, tx, ty)
        d_to_opp = manh(nx, ny, ox, oy)
        # "Don't step into opponent race": if opponent is closer to target, maximize separation first.
        race_term = -d_to_target
        if opp_is_closer:
            score = (d_to_opp, race_term)
        else:
            score = (race_term, d_to_opp)
        # Slight penalty for obstacle neighborhood midpoint from candidate to target
        mid_pen = 0
        mx = (nx + tx) // 2
        my = (ny + ty) // 2
        for ddx in (-1, 0, 1):
            for ddy in (-1, 0, 1):
                if (mx + ddx, my + ddy) in obstacles:
                    mid_pen += 1
        score = (score[0], score[1] - mid_pen)
        if best_score is None or score > best_score:
            best_score = score
            best_move = (dx, dy)

    dx, dy = best_move
    return [int(dx), int(dy)]
