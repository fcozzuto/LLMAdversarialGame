def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def valid(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    def dist(ax, ay, bx, by):
        return abs(ax - bx) + abs(ay - by)

    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    moves = []
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if valid(nx, ny):
            moves.append((dx, dy))

    # Choose target resource by "race advantage" against opponent.
    if resources:
        best_t = None
        best_score = -10**18
        for rx, ry in resources:
            d1 = dist(sx, sy, rx, ry)
            d2 = dist(ox, oy, rx, ry)
            # Prefer resources where we are closer; otherwise still prefer minimizing our deficit.
            adv = d2 - d1
            # Tie-break: also prefer closer-to-middle resources to reduce path risk.
            midx, midy = (w - 1) / 2.0, (h - 1) / 2.0
            mid_pen = -(abs(rx - midx) + abs(ry - midy)) * 0.01
            # Deterministic combined score
            score = adv * 1000 + mid_pen - d1 * 0.1
            # Another tie-break: smaller coordinates
            if best_t is None or score > best_score or (score == best_score and (rx, ry) < best_t):
                best_score = score
                best_t = (rx, ry)

        rx, ry = best_t
        best_move = (0, 0)
        best_obj = -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            our_d = dist(nx, ny, rx, ry)
            opp_d = dist(ox, oy, rx, ry)
            # Objective: maximize advantage if we move there
            obj = (opp_d - our_d) * 1000 - our_d
            # Slight preference for increasing distance from opponent when objectives are similar
            opp_to_our = dist(nx, ny, ox, oy)
            obj += opp_to_our * 0.01
            if obj > best_obj:
                best_obj = obj
                best_move = (dx, dy)
        return [int(best_move[0]), int(best_move[1])]

    # No resources left: move to reduce distance to opponent (deterministic fallback)
    best_move = (0, 0)
    best_obj = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        obj = -dist(nx, ny, ox, oy)
        if obj > best_obj:
            best_obj = obj
            best_move = (dx, dy)
