def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position", [0, 0]) or [0, 0]
    op = observation.get("opponent_position", [7, 7]) or [7, 7]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    moves = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]

    resources = []
    for p in observation.get("resources") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if inb(x, y):
                resources.append((x, y))

    if not inb(sx, sy):
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if inb(nx, ny):
                return [dx, dy]
        return [0, 0]

    if not resources:
        # No visible resources: move to maximize separation from opponent while staying safe
        best = None
        bestv = -10**9

            if not inb(nx, ny): 
                continue
            v = cheb(nx, ny, ox, oy)
            if v > bestv:
                bestv, best = v, (dx, dy)
        return [best[0], best[1]] if best is not None else [0, 0]

    # Choose target using deterministic competition-aware scoring

    best_score = -10**18

        od = cheb(ox, oy, rx, ry)
        # Prefer: self can reach no later than opponent; then minimize self distance.
        # If opponent is closer, still consider targets where the advantage gap is largest (so we pivot).
        gap = od - sd
        # Remaining resource count can change urgency; keep deterministic but slightly responsive.
        rem = int(observation.get("remaining_resource_count", len(resources)) or len(resources))
        urgency = 1 + (0 if rem > 6 else (2 if rem > 2 else 3))
        score = (1 if sd <= od else 0) * 1000000 * urgency + gap * 10 - sd
        if score > best_score:
            best_score = score
            best_target = (rx, ry)
