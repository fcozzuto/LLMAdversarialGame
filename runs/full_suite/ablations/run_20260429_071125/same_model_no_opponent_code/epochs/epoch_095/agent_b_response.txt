def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles", []) or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources", []) or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # If no resources: drift to center-ish deterministically
    if not resources:
        tx = (w - 1) // 2
        ty = (h - 1) // 2
        best = (-10**9, 0, 0)
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny):
                continue
            d = cheb(nx, ny, tx, ty)
            score = -d
            if score > best[0]:
                best = (score, dx, dy)
        return [best[1], best[2]]

    # Strategic change: prioritize resources we are not behind on (win races),
    # and prefer paths that also keep opponent further from contested targets.
    # Additionally, create "pressure" by moving toward the most contested resource.
    best_move = ( -10**18, 0, 0 )
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny):
            continue

        my_future_best = None
        for rx, ry in resources:
            d_us = cheb(nx, ny, rx, ry)
            d_op = cheb(ox, oy, rx, ry)

            # If opponent can beat us, downweight strongly; if we can match, moderate.
            # This changes targeting from pure proximity to contested-resource race control.
            race = d_op - d_us  # positive means we arrive earlier
            base = 0
            if race >= 2:
                base += 50
            elif race >= 0:
                base += 25
            elif race >= -1:
                base += 10
            else:
                base -= 60

            # Prefer closer arrival and (slightly) farther opponent distance to that target.
            score = base + (40 - d_us) + (d_op * 2)

            if my_future_best is None or score > my_future_best:
                my_future_best = score

        # Secondary term: avoid stepping into obvious "losing" corridor by preferring
        # moves that increase our distance from opponent when we can't secure a contest.
        # Deterministic tie-break via lexicographic preference order in dirs.
        opp_dist = cheb(nx, ny, ox, oy)
        adj = 0
        # If we are likely behind on the best target, boost retreat.
        if my_future_best is not None and my_future_best < 25:
            adj = opp_dist // 2
        total = my_future_best + adj

        if total > best_move[0]:
            best_move = (total, dx, dy)

    return [int(best_move[1]), int(best_move[2])]
