        return [0, 0]

    if not resources:
        best = (0, 0, -10**18)
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if inb(nx, ny) and (nx, ny) not in obstacles:
                score = -(abs(nx - ox) + abs(ny - oy))

        dx = a[0] - b[0]
        dy = a[1] - b[1]
        return dx * dx + dy * dy

        if not inb(nx, ny) or (nx, ny) in obstacles:
            continue

        my_pos = (nx, ny)
        opp_pos = (ox, oy)

        # Find my closest and opponent closest resources
        my_best = 10**18
        opp_best = 10**18
        pick = None
        for r in targets:
            dr_my = dist2(my_pos, r)
            if dr_my < my_best:
                my_best = dr_my
                pick = r

            dr_opp = dist2(opp_pos, r)
