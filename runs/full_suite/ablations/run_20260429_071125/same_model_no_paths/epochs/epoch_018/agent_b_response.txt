def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if inside(x, y):
                resources.append((x, y))

    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    if not inside(sx, sy):
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if inside(nx, ny):
                return [dx, dy]
        return [0, 0]

    if not resources:
        tx, ty = 0, 0
        if (sx + sy) % 2 == 0:
            tx, ty = w - 1, h - 1
        best = (10**9, None)
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not inside(nx, ny):
                continue
            d = cheb(nx, ny, tx, ty)
            if d < best[0]:
                best = (d, (dx, dy))
        return list(best[1] or [0, 0])

    def score_to_target(nx, ny, tx, ty):
        d_res = cheb(nx, ny, tx, ty)
        d_opp = cheb(nx, ny, ox, oy)
        # Prefer reducing distance to resource, while not allowing easy opponent access.
        # Also slightly prefer cells closer to resource than to opponent.
        return d_res * 10 - d_opp

    # Choose a deterministic target: lexicographically smallest among nearest resources.
    best_t = None
    best_d = 10**9
    for tx, ty in resources:
        d = cheb(sx, sy, tx, ty)
        if d < best_d or (d == best_d and (ty, tx) < (best_t[1], best_t[0])):
            best_d = d
            best_t = (tx, ty)

    tx, ty = best_t
    # Move toward target but with opponent-aware evaluation among legal moves.
    best = (10**18, None, None)
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not inside(nx, ny):
            continue
        s = score_to_target(nx, ny, tx, ty)
        # Tie-break: prefer moves that reduce resource distance most; then prefer lower x,y delta order.
        d_res = cheb(nx, ny, tx, ty)
        d_opp = cheb(nx, ny, ox, oy)
        key = (s, d_res, -d_opp, dx, dy)
        if key < best:
            best = (s, (dx, dy), key)

    return list(best[1] or [0, 0])
