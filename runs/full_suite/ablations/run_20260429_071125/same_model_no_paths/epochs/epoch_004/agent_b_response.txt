def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]

    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []
    if not resources:
        return [0, 0]

    obs = set((p[0], p[1]) for p in obstacles)

    def in_bounds(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1), (0, 0), (0, 1),
             (1, -1), (1, 0), (1, 1)]

    cx = (w - 1) / 2.0
    cy = (h - 1) / 2.0

    # Precompute opponent distances to resources (constant this turn).
    opp_ds = []
    for rx, ry in resources:
        opp_ds.append(cheb(ox, oy, rx, ry))

    best_score = -10**18
    best_move = (0, 0)

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not in_bounds(nx, ny) or (nx, ny) in obs:
            continue

        # Evaluate: prioritize resources where we're getting closer than opponent.
        # Use only top-3 "contested" resources to keep it cheap and stable.
        items = []
        for i, (rx, ry) in enumerate(resources):
            our_d = cheb(nx, ny, rx, ry)
            opp_d = opp_ds[i]
            gap = opp_d - our_d  # positive is good
            # Contested resources are those with small combined distance.
            comb = our_d + opp_d
            items.append((comb, gap, our_d, opp_d))

        # Deterministic selection of three smallest comb values (no sorting needed).
        c1 = c2 = c3 = 10**9
        s1 = s2 = s3 = -10**18
        for comb, gap, our_d, opp_d in items:
            sc = gap * 3
            if our_d == 0:
                sc += 1000
            if opp_d == 0:
                sc -= 600
            # Encourage reducing our distance when still behind.
            if gap < 0:
                sc += gap  # negative penalty proportional

            if comb < c1:
                c3, s3 = c2, s2
                c2, s2 = c1, s1
                c1, s1 = comb, sc
            elif comb < c2:
                c3, s3 = c2, s2
                c2, s2 = comb, sc
            elif comb < c3:
                c3, s3 = comb, sc

        score = s1 + s2 + s3

        # Subtle tiebreakers: stay within bounds and move slightly toward center.
        center_dist = abs(nx - cx) + abs(ny - cy)
        score -= int(center_dist * 2)

        # Prefer moves that don't give opponent an immediate capture advantage
        # (if opponent is adjacent to a resource, we try not to move away from all resources).
        near_opp = 0
        for i, (rx, ry) in enumerate(resources):
            if opp_ds[i] <= 1:
                near_opp += 1
        if near_opp > 0:
            # Ensure we have at least one resource with small our distance.
            min_our = 10**9
            for i, (rx, ry) in enumerate(resources):
                our_d = cheb(nx, ny, rx, ry)
                if our_d < min_our:
                    min_our = our_d
            score += max(0, 6 - min_our) * 2

        if score > best_score:
            best_score = score
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]

def choose_move(observation):
    w = observation.get("grid_width", 0)
    h = observation.get("grid_height", 0)
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))
    resources = observation.get("resources", []) or []
    obstacles = observation.get("obstacles", []) or []
    if not (w and h):
        return [0, 0]

    blocked = set()
    for p in obstacles:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            blocked.add((p[0], p[1]))

    def dist(a, b, c, d):
        dx = a - c
        if dx < 0:
            dx = -dx
        dy = b - d
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1), (0, 0), (0, 1),
             (1, -1), (1, 0), (1, 1)]

    best_move = (0, 0)
    best_val = -10**18

    # Consider up to 6 closest resources to keep it fast and stable.
    if resources:
        rs = []
        for r in resources:
            rx, ry = r[0], r[1]
            rs.append((dist(sx, sy, rx, ry), rx, ry, dist(ox, oy, rx, ry)))
        rs.sort(key=lambda x: x[0])
        rs = rs[:6]
    else:
        rs = []

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if nx < 0 or nx >= w or ny < 0 or ny >= h:
            continue
        if (nx, ny) in blocked:
            continue

        if not rs:
            val = -abs(nx - sx) - abs(ny - sy)
        else:
            val = 0
            for _, rx, ry, do in rs:
                dn = dist(nx, ny, rx, ry)
                gain = do - dn
                if gain > 0:
                    val += 10 * gain
                else:
                    val += gain
            # Mild tie-break: prefer moving towards nearest resource.
            if rs:
                _, rx0, ry0, _ = rs[0]
                val += -dist(nx, ny, rx0, ry0)

        if val > best_val:
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
