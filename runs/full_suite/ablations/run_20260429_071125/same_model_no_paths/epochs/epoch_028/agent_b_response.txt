    blocked = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                blocked.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in blocked:
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1), (0, 0), (0, 1),
             (1, -1), (1, 0), (1, 1)]
    def valid(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    def bfs_dists(start):
        INF = 10**9
        dist = [[INF] * h for _ in range(w)]
        x0, y0 = start
        if not valid(x0, y0):
            return dist
        qx, qy = [x0], [y0]
        dist[x0][y0] = 0
        qi = 0
        while qi < len(qx):
            x, y = qx[qi], qy[qi]
            qi += 1
            nd = dist[x][y] + 1
            nx0 = x - 1
            nx1 = x
            nx2 = x + 1
            for nx in (nx0, nx1, nx2):
                for ny in (y - 1, y, y + 1):
                    if nx == x and ny == y:
                        continue
                    if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in blocked:
                        if nd < dist[nx][ny]:
                            dist[nx][ny] = nd
                            qx.append(nx)
                            qy.append(ny)
        return dist

    if not resources:
        tx, ty = w // 2, h // 2
        best = (0, 0)
        bestv = -10**9

            nx, ny = sx + dx, sy + dy

                continue
            d = max(abs(nx - tx), abs(ny - ty))
            if -d > bestv:
                bestv = -d
                best = (dx, dy)
        return [best[0], best[1]]
