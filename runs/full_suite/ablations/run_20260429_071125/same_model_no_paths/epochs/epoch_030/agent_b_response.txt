def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    if not ok(sx, sy):
        for dx, dy in moves:
            if ok(sx + dx, sy + dy):
                return [dx, dy]
        return [0, 0]

    if not resources:
        bestd = None
        best = (0, 0)
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not ok(nx, ny): 
                continue
            d = cheb(nx, ny, ox, oy)
            if bestd is None or d > bestd:
                bestd, best = d, (dx, dy)
        return [best[0], best[1]]

    # Target resources we can beat (or at least approach with advantage).
    best_move = (0, 0)
    best_score = None

    # Tie-break deterministically: prefer moves in this order.
    order = {m: i for i, m in enumerate(moves)}

    # Precompute opponent distances to each resource.
    opp_dist = {res: cheb(ox, oy, res[0], res[1]) for res in resources}

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue

        my_best = 10**9
        opp_best = 10**9
        for rx, ry in resources:
            my_d = cheb(nx, ny, rx, ry)
            my_best = my_d if my_d < my_best else my_best
            od = opp_dist[(rx, ry)]
            opp_best = od if od < opp_best else opp_best

        # Find whether any resource becomes "contested" in our favor next step.
        advantage = -10**9
        closest_opp_target = 10**9
        for rx, ry in resources:
            my_d = cheb(nx, ny, rx, ry)
            od = opp_dist[(rx, ry)]
            adv = od - my_d  # positive if we are closer than opponent
            if adv > advantage:
                advantage = adv
            if od < closest_opp_target:
                closest_opp_target = od

        # Score: strong incentive to gain/maintain advantage; also chase nearest viable.
        score = 0
        score += 1000 * (1 if advantage >= 0 else 0)
        score += 50 * advantage
        score += -my_best
        score += -0.1 * closest_opp_target  # slight bias away from resources opponent is already closest to

        # Deterministic tie-break.
        if best_score is None or score > best_score or (score == best_score and order[(dx, dy)] < order[best_move]):
            best_score = score
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
