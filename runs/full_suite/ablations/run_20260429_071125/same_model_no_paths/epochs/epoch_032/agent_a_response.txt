def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position", [0, 0]) or [0, 0]
    op = observation.get("opponent_position", [0, 0]) or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                blocked.add((x, y))

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if ok(x, y):
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    if not ok(sx, sy):
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if ok(nx, ny):
                return [dx, dy]
        return [0, 0]

    # If resources exist, pick the one we can claim earlier, else stay near the best race lead.
    if resources:
        best = None
        for tx, ty in resources:
            ds = cheb(sx, sy, tx, ty)
            do = cheb(ox, oy, tx, ty)
            lead = do - ds  # positive means we are closer
            # tie-break: prefer shorter ds, then deterministic by coordinates
            key = (-lead, ds, tx, ty)
            if best is None or key < best[0]:
                best = (key, (tx, ty))
        tx, ty = best[1]

        # Greedy step towards (tx,ty), but avoid squares that let opponent get closer to our current target too quickly.
        cand_best = None
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not ok(nx, ny):
                continue
            my_d = cheb(nx, ny, tx, ty)
            op_d = cheb(ox, oy, tx, ty)
            # If we step in a way that increases our distance, heavily penalize.
            # Also penalize moving closer to opponent (helps avoid interception).
            closer_op = cheb(nx, ny, ox, oy)
            key = (my_d, -lead_strength(my_d, op_d), -closer_op, abs(dx) + abs(dy), nx, ny)
            if cand_best is None or key < cand_best[0]:
                cand_best = (key, (dx, dy))
        if cand_best:
            return [int(cand_best[1][0]), int(cand_best[1][1])]

    # Fallback: move to increase distance from opponent, else head to center.
    cx, cy = (w - 1) // 2, (h - 1) // 2
    target = (cx, cy)

    best = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue
        dist_op = cheb(nx, ny, ox, oy)
        dist_t = cheb(nx, ny, target[0], target[1])
        key = (-dist_op, dist_t, nx, ny)
        if best is None or key < best[0]:
            best = (key, (dx, dy))
    if best:
        return [int(best[1][0]), int(best[1][1])]
    return [0, 0]

def lead_strength(my_d, op_d):
    # deterministic helper returning a coarse value to stabilize tie-breaking
    return (op_d - my_d)
