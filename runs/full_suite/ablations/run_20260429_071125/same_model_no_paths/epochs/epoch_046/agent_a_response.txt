def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                blocked.add((x, y))

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if ok(x, y):
                resources.append((x, y))

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    def best_target_for(nx, ny):
        if not resources:
            return None
        # Prefer resources where we are not worse than opponent, and prefer those we can reach quickly.
        # Deterministic tie-breaking by coordinate.
        best = None
        best_key = None
        for (rx, ry) in resources:
            d1 = cheb(nx, ny, rx, ry)
            d2 = cheb(ox, oy, rx, ry)
            # If we are closer or equal, score high; if behind, score low.
            rel = d2 - d1  # positive if we are closer
            # Also avoid targets very near opponent when we are worse.
            near_pen = 0
            if d2 <= 2 and rel < 0:
                near_pen = 50
            # Key: maximize rel, then minimize our distance, then prefer lower coords.
            key = (-(rel - near_pen), d1, rx, ry)
            if best_key is None or key < best_key:
                best_key = key
                best = (rx, ry)
        return best

    # If current position invalid, step to a valid neighbor.
    if not ok(sx, sy):
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if ok(nx, ny):
                return [dx, dy]
        return [0, 0]

    # Evaluate each move by distance to the best target and anti-opponent positioning.
    best_move = [0, 0]
    best_val = -10**18
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue
        target = best_target_for(nx, ny)
        if target is None:
            # No resources: fall back to staying away from being trapped by obstacles/opponent.
            val = cheb(nx, ny, ox, oy)
            if val > best_val:
                best_val = val
                best_move = [dx, dy]
            continue

        rx, ry = target
        d_self = cheb(nx, ny, rx, ry)
        d_opp = cheb(ox, oy, rx, ry)

        # Encourage reaching target sooner than opponent.
        reach_adv = d_opp - d_self
        # Slightly discourage moving toward opponent directly.
        opp_direct = cheb(nx, ny, ox, oy)

        # Tiny deterministic preference for diagonals every other turn to change policy if needed.
        diag = 1 if dx != 0 and dy != 0 else 0
        turn = int(observation.get("turn_index") or 0)

        val = (1000 * (reach_adv) - 10 * d_self + 0.2 * opp_direct + 0.01 * (diag if turn % 2 == 0 else -diag))
        if val > best_val:
            best_val = val
            best_move = [dx, dy]

    return [int(best_move[0]), int(best_move[1])]
