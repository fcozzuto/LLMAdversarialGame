def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                blocked.add((x, y))

    def valid(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    res = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if valid(x, y):
                res.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    if not valid(sx, sy):
        for dx, dy in moves:
            if valid(sx + dx, sy + dy):
                return [dx, dy]
        return [0, 0]

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def step_toward(tx, ty):
        best = None
        bestv = -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny):
                continue
            d = cheb(nx, ny, tx, ty)
            score = -d
            if (nx, ny) == (tx, ty):
                score += 1000
            # discourage stepping into lines near opponent when contesting
            score -= 0.15 * cheb(nx, ny, ox, oy)
            # tiny preference for moving in lexicographic order deterministically
            if best is None or score > bestv:
                bestv = score
                best = (dx, dy)
        return best if best is not None else (0, 0)

    # Choose a resource we can plausibly reach earlier than opponent.
    # If none, pick best overall and move to reduce gap to opponent (contest).
    candidates = []
    for rx, ry in res:
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        gain = (do - ds)  # positive means we're closer
        # prioritize closer-now but only if not hopeless
        candidates.append((gain, -ds, rx, ry))

    candidates.sort(reverse=True)
    target = None
    # First pass: prefer resources where we lead by at least a margin
    for gain, nds, rx, ry in candidates[:len(candidates)]:
        if gain >= 2:
            target = (rx, ry)
            break
    if target is None and candidates:
        # Second pass: if we can match or slightly trail, still go for best reachable
        target = (candidates[0][2], candidates[0][3])

    if target is not None:
        tx, ty = target
        return list(step_toward(tx, ty))

    # Fallback: no visible resources -> move toward board center while avoiding obstacles
    cx, cy = (w - 1) // 2, (h - 1) // 2
    best = None
    bestv = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        dcent = cheb(nx, ny, cx, cy)
        dop = cheb(nx, ny, ox, oy)
        score = -dcent - 0.05 * dop
        if best is None or score > bestv:
            bestv = score
            best = (dx, dy)
    return list(best) if best is not None else [0, 0]
