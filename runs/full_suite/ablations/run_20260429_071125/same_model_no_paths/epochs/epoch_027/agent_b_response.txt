def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                blocked.add((x, y))

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in blocked:
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1),
             (0, -1), (0, 0), (0, 1),
             (1, -1), (1, 0), (1, 1)]

    def in_bounds(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    best_dx, best_dy = 0, 0
    best_val = -10**18

    # Choose a resource where we are relatively closer than opponent (break ties by absolute closeness).
    target = None
    target_rank = None
    for rx, ry in resources:
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        rank = (do - ds, -min(ds, do), -rx, -ry)  # prefer positive reach advantage; then nearer; then deterministic
        if target_rank is None or rank > target_rank:
            target_rank = rank
            target = (rx, ry)

    if target is None:
        # Fallback: move away from opponent while avoiding obstacles toward any free direction deterministically
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not in_bounds(nx, ny):
                continue
            val = cheb(nx, ny, ox, oy)  # maximize distance
            if val > best_val or (val == best_val and (dx, dy) < (best_dx, best_dy)):
                best_val = val
                best_dx, best_dy = dx, dy
        return [best_dx, best_dy]

    tx, ty = target
    # Scoring: (1) reduce our distance to target, (2) keep or improve distance advantage vs opponent,
    # (3) lightly prefer not moving toward opponent.
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not in_bounds(nx, ny):
            continue
        ds = cheb(nx, ny, tx, ty)
        do = cheb(nx, ny, ox, oy)
        cur_ds = cheb(sx, sy, tx, ty)
        cur_adv = cheb(ox, oy, tx, ty) - cur_ds
        new_adv = cheb(ox, oy, tx, ty) - ds
        val = 0
        val += (cur_ds - ds) * 100  # immediate improvement
        val += new_adv * 30        # maintain advantage
        val += do * 2              # distance from opponent as tie-breaker-ish
        # also discourage stepping onto squares adjacent to many obstacles (minor)
        adj_obs = 0
        for ax in (-1, 0, 1):
            for ay in (-1, 0, 1):
                if ax == 0 and ay == 0:
                    continue
                x2, y2 = nx + ax, ny + ay
                if (x2, y2) in blocked:
                    adj_obs += 1
        val -= adj_obs * 3

        if val > best_val or (val == best_val and (dx, dy) < (best_dx, best_dy)):
            best_val = val
            best_dx, best_dy = dx, dy

    return [int(best_dx), int(best_dy)]
