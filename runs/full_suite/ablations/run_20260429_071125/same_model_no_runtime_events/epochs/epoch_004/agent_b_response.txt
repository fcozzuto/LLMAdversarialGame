def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", []) or []
    obstacles = set(tuple(p) for p in (observation.get("obstacles", []) or []))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    moves = []
    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if inb(nx, ny):
                moves.append((dx, dy))

    if not moves:
        return [0, 0]

    tx, ty = 0, 0
    if resources:
        my_best_dist = 10**9
        opp_best_dist = 10**9
        for rx, ry in resources:
            d1 = cheb(sx, sy, rx, ry)
            d2 = cheb(ox, oy, rx, ry)
            if d1 < my_best_dist: my_best_dist = d1
            if d2 < opp_best_dist: opp_best_dist = d2

        best_move = (0, 0)
        best_val = -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            move_val = 0.0
            for rx, ry in resources:
                ds = cheb(nx, ny, rx, ry)
                do = cheb(ox, oy, rx, ry)
                # Value: prefer closing gap to a resource; big bonus if we become first.
                rel = (do - ds)
                gain = rel * 20 - ds
                # If opponent is currently closer to that resource, reward denial by reducing their advantage.
                denial = 0
                if do <= d for d in [do]:
                    denial = 0
                # Add a small incentive to go after the globally best (closest-to-me) and to deny opponent when ahead.
                move_val += gain
            # Normalize-ish: also add focus on our closest and opponent-closest to reduce dithering.
            # Recompute focused targets.
            my_closest = min(resources, key=lambda p: cheb(nx, ny, p[0], p[1]))
            opp_closest = min(resources, key=lambda p: cheb(ox, oy, p[0], p[1]))
            ds_c = cheb(nx, ny, my_closest[0], my_closest[1])
            do_c = cheb(ox, oy, opp_closest[0], opp_closest[1])
            move_val += (do_c - ds_c) * 15 - ds_c
            # If opponent is overall closer to their best than we are, prioritize denial.
            if opp_best_dist < my_best_dist:
                move_val += cheb(nx, ny, tx, ty) * -0.02
                move_val += (my_best_dist - cheb(nx, ny, resources[0][0], resources[0][1])) * 0.0
            else:
                # Otherwise prefer tempo: closer to our best target.
                move_val += -0.5 * ds_c
            if move_val > best_val or (move_val == best_val and (dx, dy) < best_move):
                best_val = move_val
                best_move = (dx, dy)
        return [best_move[0], best_move[1]]

    # No resources: move toward the opposite corner deterministically; slightly favor moving away from opponent corner.
    # (Opposite corner assumption: (0,0) vs (w-1,h-1); if not, still works as a generic chase.)
    best = (0, 0)
    bestv = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        v = -cheb(nx, ny, tx, ty) + 0.1 * cheb(nx, ny, ox, oy)
        if v > bestv or (v == bestv and (dx, dy) < best):
            bestv = v
            best = (dx, dy)
    return [best[0], best[1]]
