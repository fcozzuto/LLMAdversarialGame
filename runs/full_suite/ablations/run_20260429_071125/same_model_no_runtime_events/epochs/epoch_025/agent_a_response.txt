def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            obstacles.add((int(p[0]), int(p[1])))

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            resources.append((int(r[0]), int(r[1])))
        elif isinstance(r, dict):
            pos = r.get("pos")
            if isinstance(pos, (list, tuple)) and len(pos) >= 2:
                resources.append((int(pos[0]), int(pos[1])))
            else:
                x = r.get("x"); y = r.get("y")
                if x is not None and y is not None:
                    resources.append((int(x), int(y)))

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx >= dy else dy

    def step_toward(tx, ty, cx, cy):
        dx = tx - cx
        if dx > 0: sxm = 1
        elif dx < 0: sxm = -1
        else: sxm = 0
        dy = ty - cy
        if dy > 0: sym = 1
        elif dy < 0: sym = -1
        else: sym = 0
        return sxm, sym

    # Pick best resource: maximize how much sooner we can reach it than opponent.
    best = None
    bestv = -10**18
    for (rx, ry) in resources:
        if not inside(rx, ry):
            continue
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        # Encourage grabbing closer-than-opponent, and prefer closer absolute distance.
        # Also slightly prefer resources nearer to center to keep options open.
        centerx = (w - 1) / 2.0
        centery = (h - 1) / 2.0
        center_pen = cheb(rx, ry, int(centerx), int(centery)) * 0.2
        v = (do - ds) * 10.0 - ds - center_pen
        if v > bestv or (v == bestv and (ds < (cheb(sx, sy, best[0], best[1]) if best else 10**9))):
            bestv = v
            best = (rx, ry)

    moves = []
    if best is None:
        # No visible resources: move to reduce distance to opponent's side while staying safe.
        tx, ty = (w - 1, h - 1) if sx < ox else (0, 0)
    else:
        tx, ty = best

    for dx in (-1, 0, 1):
        for dy in (-1, 0, 1):
            nx, ny = sx + dx, sy + dy
            if inside(nx, ny):
                moves.append((dx, dy))

    # If best exists, bias moves toward it; otherwise toward (tx,ty).
    target_dx, target_dy = step_toward(tx, ty, sx, sy)

    def move_value(dx, dy):

        # Higher when closer to chosen resource and not giving opponent an advantage.

            rx, ry = best
            ds1 = cheb(nx, ny, rx, ry)
            do = cheb(ox, oy, rx, ry)
