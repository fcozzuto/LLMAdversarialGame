def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obst = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obst.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obst:
                resources.append((x, y))
    if not resources:
        return [0, 0]

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    candidates = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    center_x = (w - 1) / 2.0
    center_y = (h - 1) / 2.0

    best_move = (0, 0)
    best_val = -10**18
    best_tiebreak = 10**18

    for dx, dy in candidates:
        nx, ny = sx + dx, sy + dy
        if not (0 <= nx < w and 0 <= ny < h):
            continue
        if (nx, ny) in obst:
            continue

        # Evaluate move by targeting the single resource with highest "take/contest" value.
        move_val = -10**18
        for rx, ry in resources:
            d_opp = cheb(ox, oy, rx, ry)
            d_self = cheb(nx, ny, rx, ry)
            adv = d_opp - d_self  # positive means we are closer than opponent to that resource
            contested = d_self > d_opp  # opponent currently closer to this resource after our move?
            block_bonus = 0
            if d_self < d_opp:
                # We can take it or deny it: strongly prefer.
                block_bonus = 2.0 * (d_opp - d_self)
            elif contested:
                # If we can't win it, avoid moving into a spot that worsens how far we are.
                block_bonus = -0.3 * (d_self - d_opp)

            center_bias = -0.05 * abs(nx - center_x) - 0.05 * abs(ny - center_y)
            val = adv + block_bonus + center_bias

            if val > move_val:
                move_val = val

        # Secondary tie-break: prefer smaller distance to the nearest resource from this move.
        min_d = min(cheb(nx, ny, rx, ry) for rx, ry in resources)
        if move_val > best_val or (move_val == best_val and min_d < best_tiebreak):
            best_val = move_val
            best_tiebreak = min_d
            best_move = (dx, dy)
