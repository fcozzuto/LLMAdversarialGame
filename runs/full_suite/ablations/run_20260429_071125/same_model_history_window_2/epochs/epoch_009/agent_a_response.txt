def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources", []) or []
    obstacles = set((p[0], p[1]) for p in (observation.get("obstacles", []) or []))

    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def kdist(x1, y1, x2, y2):
        dx = abs(x1 - x2)
        dy = abs(y1 - y2)
        return dx if dx > dy else dy

    if not resources:
        # If no resources, maximize distance from opponent deterministically.
        best = (0, 0)
        bestv = -10**18
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not inside(nx, ny):
                continue
            v = kdist(nx, ny, ox, oy)
            if v > bestv or (v == bestv and (dx, dy) < best):
                bestv = v
                best = (dx, dy)
        return [best[0], best[1]]

    # Pick a resource we are closer to than opponent (race), otherwise the closest overall.
    scored = []
    for rx, ry in resources:
        sd = kdist(sx, sy, rx, ry)
        od = kdist(ox, oy, rx, ry)
        # Prefer winning races (sd < od), then faster capture, then further from opponent as tiebreak.
        winrace = 1 if sd < od else 0
        score = (winrace, -sd, kdist(ox, oy, rx, ry))
        scored.append((score, rx, ry))
    scored.sort(key=lambda t: (-t[0][0], t[0][1], -t[0][2], t[1], t[2]))
    _, tx, ty = scored[0]

    # If adjacent to a resource, go for it (prefer exact adjacency).
    adj = []
    for rx, ry in resources:
        if kdist(sx, sy, rx, ry) == 1:
            adj.append((rx, ry))
    if adj:
        # Deterministic: choose the adjacent resource with smallest kdist, then lexicographically.
        rx, ry = min(adj, key=lambda p: (p[0], p[1]))
        tx, ty = rx, ry

    best = (0, 0)
    bestv = -10**18
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not inside(nx, ny):
            continue
        # Move closer to target while keeping some distance from opponent.
        to_t = kdist(nx, ny, tx, ty)
        from_o = kdist(nx, ny, ox, oy)
        v = (-to_t, from_o)
        if v > bestv or (v == bestv and (dx, dy) < best):
            bestv = v
            best = (dx, dy)
    return [best[0], best[1]]
