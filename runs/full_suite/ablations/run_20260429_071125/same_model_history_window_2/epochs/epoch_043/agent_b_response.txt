def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [0, 0])
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) == 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) == 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))
    if not resources:
        return [0, 0]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def inside(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def choose_target():
        best = None
        best_a = None
        for rx, ry in resources:
            sd = cheb(sx, sy, rx, ry)
            od = cheb(ox, oy, rx, ry)
            margin = od - sd
            # Heuristic: prioritize resources we can beat (margin), then closer.
            a = (margin * 10000) - sd
            # Deterministic tie-break: prefer lower x then lower y (consistent).
            key = (a, -rx, -ry)
            if best is None or key > best_a:
                best = (rx, ry)
                best_a = key
        return best

    tx, ty = choose_target()

    moves = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    best_move = (0, 0)
    best_score = None

    # Secondary strategy shift: if opponent is closer to the chosen target, aim for the best "escape" resource.
    opp_closer = cheb(ox, oy, tx, ty) <= cheb(sx, sy, tx, ty)

    def score_move(nx, ny, target):
        rx, ry = target
        sd = cheb(nx, ny, rx, ry)
        od = cheb(ox, oy, rx, ry)
        margin = od - sd
        # Encourage approaching target while avoiding moves that give opponent immediate win.
        return (margin * 12000) - sd - 50 * (abs(nx - rx) + abs(ny - ry))

    target2 = None
    if opp_closer:
        # Pick an alternate target where we currently have advantage; if none, fall back to chosen target.
        alt = None
        bestk = None
        for rx, ry in resources:
            sd = cheb(sx, sy, rx, ry)
            od = cheb(ox, oy, rx, ry)
            margin = od - sd
            k = (margin, -sd, -rx, -ry)
            if bestk is None or k > bestk:
                bestk = k
                alt = (rx, ry)
        target2 = alt if bestk and bestk[0] > 0 else (tx, ty)

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inside(nx, ny):
            continue
        target = (tx, ty) if not opp_closer else target2
        sc = score_move(nx, ny, target)
        # Deterministic tie-break: prefer fewer steps (Cheb from current to target), then lexicographic move.
        dist_after = cheb(nx, ny, target[0], target[1])
        key = (sc, -dist_after, -dx, -dy)
        if best_score is None or key > best_score:
            best_score = key
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
