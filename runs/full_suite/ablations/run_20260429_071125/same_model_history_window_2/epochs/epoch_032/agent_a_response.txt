def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [0, 0])
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) == 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) == 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def score_cell(tx, ty):
        if (tx, ty) in obstacles:
            return -10**18
        ds = cheb(sx, sy, tx, ty)
        do = cheb(ox, oy, tx, ty)
        # Prefer cells we can claim first; otherwise, prefer denying closer resources.
        # Small center bias to reduce dithering.
        cx, cy = w // 2, h // 2
        center = cheb(tx, ty, cx, cy)
        lead = (do - ds)  # positive if we are closer
        return (5.0 * lead) - (1.2 * ds) - (0.05 * center)

    # If no visible resources, move toward center while spacing from opponent
    if not resources:
        cx, cy = w // 2, h // 2
        best = (0, 0, -10**18)
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not ok(nx, ny):
                continue
            ns = cheb(nx, ny, cx, cy)
            no = cheb(nx, ny, ox, oy)
            val = (-1.0 * ns) + (0.15 * no)
            if val > best[2]:
                best = (dx, dy, val)
        return [best[0], best[1]]

    # Choose best target resource based on static comparison
    target = None
    best_t = -10**18
    for tx, ty in resources:
        val = score_cell(tx, ty)
        if val > best_t:
            best_t = val
            target = (tx, ty)
    tx, ty = target

    # Evaluate each move: get closer to target, and if opponent could contest it, increase their distance
    best = (0, 0, -10**18)
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue
        ds = cheb(nx, ny, tx, ty)
        do = cheb(ox, oy, tx, ty)
        # Track whether we'd become the closer claimant
        lead_after = do - ds
        # Also consider chasing the second-best resource if we are far from the target
        secondary = -1.0
        for rx, ry in resources:
            d2s = cheb(nx, ny, rx, ry)
            d2o = cheb(ox, oy, rx, ry)
            lead2 = d2o - d2s
            if lead2 > secondary:
                secondary = lead2
        # Prefer moves that improve our lead first, then distance, then deny opponent locally
        val = (8.0 * lead_after) - (1.3 * ds) + (0.4 * secondary) - (0.05 * cheb(nx, ny, ox, oy))
        if val > best[2]:
            best = (dx, dy, val)

    return [best[0], best[1]]
