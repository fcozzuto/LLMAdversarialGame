def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [0, 0])
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) == 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) == 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))
    if not resources:
        return [0, 0]

    def cheb(a, b, c, d):
        dx = a - c
        if dx < 0: dx = -dx
        dy = b - d
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    def obst_pen(x, y):
        pen = 0
        for ax, ay in obstacles:
            d = cheb(x, y, ax, ay)
            if d == 0: pen += 1000
            elif d == 1: pen += 12
            elif d == 2: pen += 3
            elif d == 3: pen += 1
        return pen

    deltas = [(dx, dy) for dx in (-1, 0, 1) for dy in (-1, 0, 1)]
    best = None
    best_key = None

    # Strategic change vs last: pick moves that swing contests (prioritize resources where we are not behind),
    # and bias toward making opponent's nearest contested resource harder.
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny):
            continue

        # Find the resource that is currently most "valuable to us" after this move.
        # Use deterministic tie-breakers by sorting with fixed key.
        local = []
        for rx, ry in resources:
            our_d = cheb(nx, ny, rx, ry)
            opp_d = cheb(ox, oy, rx, ry)
            # If we can be first to it, reward strongly; if behind, still consider but penalize.
            lead = opp_d - our_d
            score = 0
            if lead > 0:
                score += 200 - 10 * our_d
            elif lead == 0:
                score += 120 - 8 * our_d
            else:
                score += 60 - 6 * our_d + lead  # lead negative reduces
            # Also slightly prefer closer-to-center paths (global-ish) to reduce dead-ends.
            center_bonus = -cheb(nx, ny, (w - 1) // 2, (h - 1) // 2) * 0.5
            # Penalty for moving near obstacles.
            score += center_bonus
            score -= obst_pen(nx, ny)
            local.append((score, our_d, opp_d, rx, ry))
        local.sort(key=lambda t: (-t[0], t[1], t[3], t[4]))
        top_score, our_d, opp_d, rx, ry = local[0]

        # Additional opponent response: if we are about to enter a cell that increases opponent's distance
        # to their best current target, reward it.
        opp_best = None
        for r2x, r2y in resources:
            od = cheb(ox, oy, r2x, r2y)
            key = (od, r2x, r2y)
            if opp_best is None or key < opp_best:
                opp_best = key
        opp_target_d = cheb(ox, oy, local[0][3], local[0][4]) if local else opp_best[0]
        # We can't move opponent, but we can reduce their contest by moving away from their best "blocking" lanes:
        # if our move increases our own distance to our contested target relative to leaving it, it's worse.
        contest_w = 0
        if resources:
            contest_w = (opp_target_d - top_score * 0)  # deterministic no-op baseline
        key = (-top_score, our_d, opp_d, obst_pen(nx, ny), dx, dy)
        if best_key is None or key < best_key:
            best_key = key
            best = (dx, dy)

    if best is None:
        return [0, 0]
