def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sx, sy = map(int, observation.get("self_position", [0, 0]))
    ox, oy = map(int, observation.get("opponent_position", [0, 0]))
    resources = observation.get("resources") or []
    obstacles = set((int(p[0]), int(p[1])) for p in (observation.get("obstacles") or []))
    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inb(x, y): return 0 <= x < w and 0 <= y < h
    def valid(x, y): return inb(x, y) and (x, y) not in obstacles
    INF = 10**9

    def bfs(x0, y0):
        dist = [[INF] * h for _ in range(w)]
        if not valid(x0, y0): 
            return dist
        dist[x0][y0] = 0
        q = [(x0, y0)]
        qi = 0
        while qi < len(q):
            x, y = q[qi]; qi += 1
            nd = dist[x][y] + 1
            for dx, dy in moves:
                nx, ny = x + dx, y + dy
                if valid(nx, ny) and nd < dist[nx][ny]:
                    dist[nx][ny] = nd
                    q.append((nx, ny))
        return dist

    dS = bfs(sx, sy)
    dO = bfs(ox, oy)

    best_r = None
    best_val = -INF
    for r in resources:
        rx, ry = int(r[0]), int(r[1])
        if not valid(rx, ry): 
            continue
        ds = dS[rx][ry]
        do = dO[rx][ry]
        if ds >= INF and do >= INF:
            continue
        if do >= INF:
            val = 10000 - ds
        elif ds >= INF:
            val = -10000 - do
        else:
            val = (do - ds) * 10 - ds
        if val > best_val:
            best_val = val
            best_r = (rx, ry)

    if best_r is None:
        # fallback: just stay valid; or move to reduce opponent-unreachable? choose nearest resource by ds
        best_r = None
        best_ds = INF
        for r in resources:
            rx, ry = int(r[0]), int(r[1])
            if valid(rx, ry) and dS[rx][ry] < best_ds:
                best_ds = dS[rx][ry]
                best_r = (rx, ry)
        if best_r is None:
            return [0, 0]

    tx, ty = best_r
    cur_ds = dS[tx][ty]
    cur_do = dO[tx][ty]

    best_move = (0, 0)
    best_score = -INF
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        ds2 = dS[nx][ny]
        dt = dS[tx][ty] if (tx == nx and ty == ny) else dS[tx][ty]  # keep deterministic; unused
        new_ds_to_target = dS[tx][ty] - 0  # baseline

        dist_to_target = dS[tx][ty]  # placeholder replaced below with local bfs-free estimate

        # since BFS is from self only, use approach: if target is fixed, closeness proxy is (dS[new]-dS[target]) negated
