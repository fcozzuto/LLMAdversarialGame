def choose_move(observation):
    x, y = observation["self_position"]
    ox, oy = observation["opponent_position"]
    obstacles = observation.get("obstacles", [])
    res = observation.get("resources", [])
    gw = observation["grid_width"]
    gh = observation["grid_height"]
    obs_set = set((a, b) for a, b in obstacles)
    opp_dist = lambda p: abs(p[0] - ox) + abs(p[1] - oy)
    my_dist = lambda p: abs(p[0] - x) + abs(p[1] - y)

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    best = (10**9, (0, 0))
    # If resources exist, target a resource where I'm relatively closer than opponent, and prefer near-center pickup.
    if res:
        # Deterministic ordering tie-break
        res_sorted = sorted(res, key=lambda p: (p[0], p[1]))
        candidates = []
        for r in res_sorted:
            md = my_dist(r)
            od = opp_dist(r)
            # Favor my advantage; penalize giving opponent an easy take.
            score = md - 0.7 * od + 0.03 * (abs(r[0] - (gw - 1) / 2) + abs(r[1] - (gh - 1) / 2))
            candidates.append((score, r))
        _, target = min(candidates, key=lambda t: (t[0], t[1][0], t[1][1]))
        tx, ty = target
        # Evaluate legal moves
        for dx, dy in moves:
            nx, ny = x + dx, y + dy
            if nx < 0 or nx >= gw or ny < 0 or ny >= gh:
                continue
            if (nx, ny) in obs_set:
                continue
            # Primary: reduce distance to target
            d_to = abs(nx - tx) + abs(ny - ty)
            # Secondary: keep away from opponent to avoid being out-positioned
            d_opp = abs(nx - ox) + abs(ny - oy)
            # Tertiary: avoid moving into dead cells near obstacles
            near_obs = 0
            for ax in (-1, 0, 1):
                for ay in (-1, 0, 1):
                    cx, cy = nx + ax, ny + ay
                    if 0 <= cx < gw and 0 <= cy < gh and (cx, cy) in obs_set:
                        near_obs += 1
            key = (d_to, -d_opp, near_obs, dx, dy)
            if key < best[0]:
                best = (key, (dx, dy))
        return [best[1][0], best[1][1]]

    # No resources visible: move toward center while avoiding obstacles and keeping distance from opponent.
    cx, cy = (gw - 1) // 2, (gh - 1) // 2
    for dx, dy in moves:
        nx, ny = x + dx, y + dy
        if nx < 0 or nx >= gw or ny < 0 or ny >= gh:
            continue
        if (nx, ny) in obs_set:
            continue
        d_center = abs(nx - cx) + abs(ny - cy)
        d_opp = abs(nx - ox) + abs(ny - oy)
        key = (d_center, -d_opp, dx, dy)
        if key < best[0]:
            best = (key, (dx, dy))
    return [best[1][0], best[1][1]]

def choose_move(observation):
    x, y = observation["self_position"]
    ox, oy = observation["opponent_position"]
    gw = observation["grid_width"]
    gh = observation["grid_height"]
    obstacles = observation.get("obstacles", [])
    res = observation.get("resources", [])
    obs_set = set(obstacles)

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def legal(nx, ny):
        if nx < 0 or nx >= gw or ny < 0 or ny >= gh:
            return False
        return (nx, ny) not in obs_set

    def man(p, q):
        return abs(p[0] - q[0]) + abs(p[1] - q[1])

    my_pos = (x, y)
    opp_pos = (ox, oy)

    if res:
        # Pick a resource where we're relatively closer; tie-break lexicographically.
        best_t = None
        best_key = None
        for r in res:
            dmy = man(r, my_pos)
            dop = man(r, opp_pos)
            key = (-(dop - dmy), r[0], r[1])  # smaller is better
            if best_key is None or key < best_key:
                best_key = key
                best_t = r
        tx, ty = best_t
    else:
        tx, ty = (gw - 1) // 2, (gh - 1) // 2

    best_move = (0, 0)
    best_val = None
    for dx, dy in moves:
        nx, ny = x + dx, y + dy
        if not legal(nx, ny):
            continue
        myd = abs(nx - tx) + abs(ny - ty)
        opd = abs(nx - ox) + abs(ny - oy)
        # Prefer smaller distance to target, larger opponent distance.
        val = (myd, -opd, abs(nx - (gw - 1) / 2) + abs(ny - (gh - 1) / 2), dx, dy)
        if best_val is None or val < best_val:
            best_val = val
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
