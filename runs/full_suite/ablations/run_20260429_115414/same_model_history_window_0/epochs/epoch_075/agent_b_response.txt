def choose_move(observation):
    w=observation["grid_width"]; h=observation["grid_height"]
    sx,sy=observation["self_position"]; ox,oy=observation["opponent_position"]
    resources=observation["resources"]; obstacles=observation["obstacles"]
    obs=set((x,y) for x,y in obstacles)
    res=resources if resources else []
    oppx,oppy=ox,oy
    def sqdist(a,b,c,d):
        dx=a-c; dy=b-d
        return dx*dx+dy*dy
    moves=[(dx,dy) for dx in (-1,0,1) for dy in (-1,0,1)]
    best=None; best_score=-10**18
    cx,cy=w//2,h//2
    for dx,dy in moves:
        nx,ny=sx+dx,sy+dy
        if nx<0 or ny<0 or nx>=w or ny>=h: 
            continue
        if (nx,ny) in obs: 
            continue
        step_score=0
        if (nx,ny) in set(res):
            step_score += 1000000
        if res:
            chosen_adv=-10**18; chosen_ds=10**18
            for rx,ry in res:
                ds=sqdist(nx,ny,rx,ry)
                do=sqdist(oppx,oppy,rx,ry)
                adv=do-ds
                if adv>chosen_adv or (adv==chosen_adv and ds<chosen_ds):
                    chosen_adv=adv; chosen_ds=ds
            step_score += chosen_adv*10 - chosen_ds
            # small center pull if no clear advantage
            if chosen_adv<=0:
                step_score += -0.2*sqdist(nx,ny,cx,cy)
        else:
            step_score += -sqdist(nx,ny,cx,cy)*0.5
        # obstacle proximity penalty
        for ax,ay in obs:
            if abs(ax-nx)<=1 and abs(ay-ny)<=1:
                step_score -= 3
        if step_score>best_score:
            best_score=step_score; best=(dx,dy)
    if best is None:
        return [0,0]
    return [int(best[0]), int(best[1])]
