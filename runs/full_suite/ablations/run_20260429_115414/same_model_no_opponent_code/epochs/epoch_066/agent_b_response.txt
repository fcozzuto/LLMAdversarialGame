    def valid(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obs

    # Find nearest resource if possible
    target = None
    bestd = 10**9
    for r in resources:
        if r and len(r) >= 2:

            if 0 <= tx < w and 0 <= ty < h and (tx, ty) not in obs:

                if d < bestd:

    scored = []
