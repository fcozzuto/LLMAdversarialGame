def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    def passable(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    resources_sorted = sorted(resources, key=lambda t: (t[0], t[1]))

    # If no resources, drift to center while avoiding obstacles.
    if not resources_sorted:
        target = (w // 2, h // 2)
        bestd, best = None, None
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not passable(nx, ny):
                continue
            d = cheb(nx, ny, target[0], target[1])
            key = (d, nx, ny)
            if bestd is None or key < bestd:
                bestd, best = key, (dx, dy)
        return [best[0], best[1]] if best is not None else [0, 0]

    # Score each move by best resource we could reach soon, and denial against opponent.
    best_key = None
    best_move = (0, 0)
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not passable(nx, ny):
            continue

        # Evaluate up to a few closest resources (deterministic) to keep it small.
        # Use current distances from the moved position and opponent position.
        scored = []
        for (tx, ty) in resources_sorted:
            ds = cheb(nx, ny, tx, ty)
            if ds > 6:
                continue
            do = cheb(ox, oy, tx, ty)
            # Favor being closer than opponent; also slight bias toward nearer absolute distance.
            # Bigger is better.
            val = (do - ds) * 10 - ds
            # Tie-break deterministically.
            scored.append((val, -tx, -ty, ds, do))
        if scored:
            scored.sort(reverse=True)
            top = scored[0]
            val, _, _, ds, do = top
        else:
            # If nothing within threshold, fall back to best overall among all resources.
            val = -999999
            ds = do = 0
            for (tx, ty) in resources_sorted[:]:
                ds1 = cheb(nx, ny, tx, ty)
                do1 = cheb(ox, oy, tx, ty)
                v1 = (do1 - ds1) * 10 - ds1
                if v1 > val:
                    val, ds, do = v1, ds1, do1

        # Prefer moves that increase our "advantage"; and reduce opponent's advantage.
        # Deterministic final tie-break via coordinates and delta ordering.
        key = (-(val), do, ds, nx, ny, dx, dy)
        if best_key is None or key < best_key:
            best_key = key
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
