def choose_move(observation):
    w = int(observation["grid_width"])
    h = int(observation["grid_height"])
    sx, sy = int(observation["self_position"][0]), int(observation["self_position"][1])
    ox, oy = int(observation["opponent_position"][0]), int(observation["opponent_position"][1])

    obstacles = observation.get("obstacles") or []
    obs = {(int(p[0]), int(p[1])) for p in obstacles}

    resources = observation.get("resources") or []
    res = [(int(p[0]), int(p[1])) for p in resources]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def valid(x, y):
        return inb(x, y) and (x, y) not in obs

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # If resources exist, target the best resource based on relative advantage.
    target = None
    best = -10**18
    for rx, ry in res:
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        # Prefer resources we can reach no later than opponent; otherwise prefer blocking/proximity.
        score = (1000 if ds <= do else 0) + (do - ds) * 10 - ds
        # Deterministic tie-breaker
        score += (-rx) * 0.01 + (-ry) * 0.001
        if score > best:
            best = score
            target = (rx, ry)

    # One-step scoring with obstacle avoidance; chase target if we have advantage, else contest.
    if target is None:
        # No resources: maximize distance from opponent (and stay safe).
        best_mv = (0, 0)
        best_sc = -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny):
                continue
            sc = cheb(nx, ny, ox, oy)
            sc += (-abs(dx) - abs(dy)) * 0.001
            if sc > best_sc:
                best_sc = sc
                best_mv = (dx, dy)
        return [int(best_mv[0]), int(best_mv[1])]

    rx, ry = target
    my_dist = cheb(sx, sy, rx, ry)
    op_dist = cheb(ox, oy, rx, ry)
    i_have_edge = my_dist <= op_dist

    best_mv = (0, 0)
    best_sc = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue

        ns = cheb(nx, ny, rx, ry)
        no = cheb(nx, ny, ox, oy)

        # If on resource now, prioritize staying/moving to keep collecting.
        on_target = 1 if (nx == rx and ny == ry) else 0

        # If we have edge: minimize distance to target strongly.
        # If not: minimize distance to target moderately but maximize opponent separation strongly.
        if i_have_edge:
            sc = on_target * 2000 + (op_dist - ns) * 30 - ns * 3 + no * 0.5
        else:
            sc = on_target * 1500 + (op_dist - ns) * 10 - ns * 2 + no * 3.5

        # Small deterministic preference for diagonal progress toward target.
        sx_dir = 0 if rx == sx else (1 if rx > sx else -1)
        sy_dir = 0 if ry == sy else (1 if ry > sy else -1)
        align = 0
        if dx == sx_dir:
            align += 1
        if dy == sy_dir:
            align += 1
        sc += align * 0.05

        if sc > best_sc:
            best_sc = sc
            best_mv = (dx, dy)

    return [int(best_mv[0]), int(best_mv[1])]
