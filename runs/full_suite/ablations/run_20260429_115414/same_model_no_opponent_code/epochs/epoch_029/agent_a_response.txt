def choose_move(observation):
    w = int(observation.get("grid_width", 0) or 0)
    h = int(observation.get("grid_height", 0) or 0)
    if w <= 0 or h <= 0:
        return [0, 0]

    def xy(v, default=(0, 0)):
        try:
            x, y = v
            x = int(x); y = int(y)
            return (x, y)
        except Exception:
            return default

    sx, sy = xy(observation.get("self_position", (0, 0)))
    ox, oy = xy(observation.get("opponent_position", (0, 0)))

    obstacles = set()
    for p in (observation.get("obstacles", []) or []):
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in (observation.get("resources", []) or []):
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))

    dxs = [-1, 0, 1]
    dys = [-1, 0, 1]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(a, b):
        ax, ay = a; bx, by = b
        dx = ax - bx; dy = ay - by
        return (dx if dx >= 0 else -dx) if (abs(dx) > abs(dy)) else (dy if dy >= 0 else -dy)

    self_pos = (sx, sy)
    opp_pos = (ox, oy)

    # If a resource is adjacent, take it.
    adj_targets = []
    for (rx, ry) in resources:
        if max(abs(rx - sx), abs(ry - sy)) == 1:
            adj_targets.append((rx, ry))
    if adj_targets:
        # deterministic: smallest (x,y)
        rx, ry = min(adj_targets, key=lambda t: (t[0], t[1]))
        return [max(-1, min(1, rx - sx)), max(-1, min(1, ry - sy))]

    # Choose the best move by minimizing:
    # - distance to closest resource after move
    # - penalty if opponent would get strictly closer to that resource than we do
    best_move = [0, 0]
    best_score = None

    # Precompute candidate resources list order to keep determinism
    resources_sorted = sorted(resources, key=lambda t: (t[0], t[1]))

    for dx in dxs:
        for dy in dys:
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny) or (nx, ny) in obstacles:
                continue
            np = (nx, ny)

            # Compute distances to resources
            if resources_sorted:
                # closest resource by chebyshev (diagonal-friendly)
                r0 = min(resources_sorted, key=lambda r: cheb(np, r))
                myd = cheb(np, r0)
                opd = cheb(opp_pos, r0)

                # After move, would opponent be "threatening" that resource?
                # If opponent is already closer than us, we get a penalty and try to avoid being outcompeted.
                # Stronger penalty when my move reduces myd but not enough to catch up.
                rel_pen = 0
                if opd < myd:
                    rel_pen = 30 + (myd - opd) * 5
                elif opd == myd:
                    rel_pen = 10

                # Add a mild preference to reduce my distance to opponent (interference)
                # and to keep centered-ish (reduce edge trapping).
                score = myd * 2 + rel_pen + cheb(np, opp_pos) + (abs(nx - (w - 1) / 2) + abs(ny - (h - 1) / 2)) * 0.01

            else:
                # No resources visible: interfere by moving closer to opponent while avoiding obstacles.
                score = cheb(np, opp_pos)

            # Deterministic tie-break: prefer smaller |dx|+|dy| then lexicographic.
            tie = (abs(dx) + abs(dy), dx, dy)
            cand = (score, tie)
            if best_score is None or cand < (best_score, (abs(best_move[0]) + abs(best_move[1]), best_move[0], best_move[1])):
                best_score = score
                best_move = [dx, dy]

    return [int(best_move[0]), int(best_move[1])]
