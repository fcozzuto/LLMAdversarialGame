def choose_move(observation):
    W = observation["grid_width"]; H = observation["grid_height"]
    sx, sy = observation["self_position"]; ox, oy = observation["opponent_position"]
    resources = observation.get("resources") or []
    obstacles = observation.get("obstacles") or []
    obs = set((p[0], p[1]) for p in obstacles)

    moves = [(-1,-1),(-1,0),(-1,1),(0,-1),(0,0),(0,1),(1,-1),(1,0),(1,1)]
    def inb(x, y): return 0 <= x < W and 0 <= y < H
    def legal(x, y): return inb(x, y) and (x, y) not in obs
    def cheb(x1, y1, x2, y2):
        dx = x1 - x2; dx = -dx if dx < 0 else dx
        dy = y1 - y2; dy = -dy if dy < 0 else dy
        return dx if dx > dy else dy
    def clamp_move(dx, dy):
        nx, ny = sx + dx, sy + dy
        return (sx, sy) if not legal(nx, ny) else (nx, ny)

    # No resources: move to center but avoid obstacles-adjacent
    if not resources:
        tx, ty = W // 2, H // 2
        best = (-10**18, 0, 0)
        for dx, dy in moves:
            mx, my = clamp_move(dx, dy)
            adj = 0
            for ax in (-1, 0, 1):
                for ay in (-1, 0, 1):
                    if ax == 0 and ay == 0: 
                        continue
                    if (mx + ax, my + ay) in obs:
                        adj += 1
            score = -cheb(mx, my, tx, ty) - 2 * adj
            if score > best[0]:
                best = (score, dx, dy)
        return [best[1], best[2]]

    # Choose a target resource based on who is closer (and "can take sooner")
    # Then score each move by improving our closeness vs opponent for that target.
    def target_rank(rx, ry):
        if (rx, ry) in obs:  # shouldn't happen, but be safe
            return -10**18
        md = cheb(sx, sy, rx, ry)
        od = cheb(ox, oy, rx, ry)
        # Prefer resources we can reach not later than opponent; break ties by closeness to us and far from opponent.
        take_now = 1 if md == 0 else 0
        return (od - md) * 1000 + (-(md)) * 3 + (od) + take_now * 10000

    targets = sorted(resources, key=lambda p: target_rank(p[0], p[1]), reverse=True)
    # Use top-K to reduce brittleness
    K = 3 if len(targets) >= 3 else len(targets)
    top_targets = targets[:K]

    best_score = -10**18
    best_move = (0, 0)

    for dx, dy in moves:
        mx, my = clamp_move(dx, dy)

        # obstacle adjacency repulsion (local, deterministic)
        adj = 0
        for ax in (-1, 0, 1):
            for ay in (-1, 0, 1):
                if ax == 0 and ay == 0:
                    continue
                if (mx + ax, my + ay) in obs:
                    adj += 1

        # Evaluate against best target by potential advantage after move
        move_val = -10**18
        for rx, ry in top_targets:
            if (rx, ry) in obs:
                continue
            md = cheb(mx, my, rx, ry)
            od = cheb(ox + 0, oy + 0, rx, ry)  # opponent unchanged this turn (simple, deterministic)
            adv = (od - md)  # higher is better
            # Encourage grabbing if we land on it; slight preference for reducing our distance.
            take_bonus = 50000 if (mx, my) == (rx, ry) else 0
            improve = 3 if md < cheb(sx, sy, rx, ry) else 0
            move_val = max(move_val, adv * 1000 + take_bonus + improve - md * 2)
        # Penalize standing adjacent to obstacles and ties by deterministic ordering (dx,dy scan order)
        score = move_val - 6 * adj

        if score > best_score:
            best_score = score
            best_move = (dx, dy)
