def choose_move(observation):
    W = int(observation.get("grid_width") or 8)
    H = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or (0, 0)
    op = observation.get("opponent_position") or (0, 0)
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < W and 0 <= y < H:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < W and 0 <= y < H and (x, y) not in obstacles:
                resources.append((x, y))

    def valid(x, y):
        return 0 <= x < W and 0 <= y < H and (x, y) not in obstacles

    def manh(a, b):
        ax, ay = a
        bx, by = b
        d = ax - bx
        if d < 0:
            d = -d
        e = ay - by
        if e < 0:
            e = -e
        return d + e

    # candidate moves in deterministic order
    deltas = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    # prefer smaller immediate "movement cost" (more stable), then deterministic order
    deltas = sorted(deltas, key=lambda d: (abs(d[0]) + abs(d[1]) * 2, d[0], d[1]))

    my_pos = (sx, sy)
    opp_pos = (ox, oy)

    if not resources:
        # drift toward center while avoiding obstacles
        cx, cy = (W - 1) // 2, (H - 1) // 2
        best = (0, 0)
        best_score = -10**18
        for dx, dy in deltas:
            nx, ny = sx + dx, sy + dy
            if not valid(nx, ny):
                continue
            # higher is better
            score = -(abs(nx - cx) + abs(ny - cy))
            if score > best_score:
                best_score = score
                best = (dx, dy)
        return [best[0], best[1]]

    # Score moves by maximizing advantage in reaching "chosen" targets:
    # - Strongly prefer resources where we are closer than the opponent by a margin.
    # - If all are contested, choose the resource with maximal expected advantage after moving.
    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in deltas:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        np = (nx, ny)

        # Evaluate best target for this immediate move.
        local_best = -10**18
        for r in resources:
            mr = manh(np, r)
            orr = manh(opp_pos, r)
            # advantage: positive if we're closer
            adv = orr - mr

            # discourage moving into "worse" proximity to resources we can't secure
            # and encourage approach: smaller mr better, larger adv better.
            # Add slight tie-break toward capturing resources closer to our current position.
            base = adv * 1000 - mr * 3

            # If opponent is already closer to this resource, heavily penalize,
            # unless we can become much closer immediately.
            if adv <= 0:
                base -= (abs(adv) + 1) * 50

            # Prefer resources that are closer overall to our path (shorter mr),
            # and slightly away from opponent to reduce swap-stealing.
            base -= manh(opp_pos, r) * 0.5

            # Deterministic tie-breaking: favor lower coordinates.
            base -= (r[0] * 0.001 + r[1] * 0.002)

            if base > local_best:
                local_best = base

        if local_best > best_val:
            best_val = local_best
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
