    def inb(x, y):
        return 0 <= x < W and 0 <= y < H and (x, y) not in obstacles

    def dist_cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    legal = []

        nx, ny = sx + dx, sy + dy
        if inb(nx, ny):
            legal.append((dx, dy))
    if not legal:

        # Pick nearest resource to self (deterministic)
        target = None
        bestd = None
        for x, y in resources:
            d = dist_cheb(sx, sy, x, y)
            if bestd is None or d < bestd or (d == bestd and (x, y) < target):
                bestd = d
                target = (x, y)
        tx, ty = target
    else:
        # No resources: drift toward center while keeping away from opponent
        tx, ty = (W - 1) // 2, (H - 1) // 2

    best_move = legal[0]
    best_val = None
    for dx, dy in legal:
        nx, ny = sx + dx, sy + dy
        d_self = dist_cheb(nx, ny, tx, ty)
