def choose_move(observation):
    W = int(observation.get("grid_width", 8))
    H = int(observation.get("grid_height", 8))
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obstacles = observation.get("obstacles") or []
    obs = set()
    for p in obstacles:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < W and 0 <= y < H:
                obs.add((x, y))

    resources = observation.get("resources") or []
    res = []
    for p in resources:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < W and 0 <= y < H and (x, y) not in obs:
                res.append((x, y))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def legal(nx, ny):
        return 0 <= nx < W and 0 <= ny < H and (nx, ny) not in obs

    target = None
    if res:
        best = None
        for rx, ry in res:
            ds = cheb(sx, sy, rx, ry)
            do = cheb(ox, oy, rx, ry)
            # Prefer resources where we're closer; if tied, prefer faster arrival and central-ish
            center_bias = -abs(rx - (W - 1) / 2) - abs(ry - (H - 1) / 2)
            v = (do - ds, -ds, center_bias, rx, ry)
            if best is None or v > best:
                best = v
                target = (rx, ry)
    if target is None:
        # Fallback: head away from opponent toward any in-bounds corner direction
        dx = 0
        dy = 0
        if sx < ox:
            dx = -1
        elif sx > ox:
            dx = 1
        if sy < oy:
            dy = -1
        elif sy > oy:
            dy = 1
        nx, ny = sx + dx, sy + dy
        if legal(nx, ny):
            return [dx, dy]
        return [0, 0]

    tx, ty = target
    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not legal(nx, ny):
            continue
        d_self = cheb(nx, ny, tx, ty)
        d_now = cheb(sx, sy, tx, ty)
        # Move evaluation: reduce distance to target, discourage stepping closer to opponent when contested
        d_opp_now = cheb(ox, oy, tx, ty)
        d_opp_if = d_opp_now  # unknown; use current as proxy
        contested = 1 if d_opp_if <= d_now else 0
        d_opp = cheb(nx, ny, ox, oy)
        v = (-(d_self), (d_now - d_self), -contested * d_opp, -(dx*dx + dy*dy), -abs(nx - tx) - abs(ny - ty))
        if v > best_val:
            best_val = v
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
