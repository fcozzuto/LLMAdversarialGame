def choose_move(observation):
    W = int(observation.get("grid_width", 8))
    H = int(observation.get("grid_height", 8))
    sx, sy = observation.get("self_position", (0, 0))
    ox, oy = observation.get("opponent_position", (0, 0))
    sx, sy, ox, oy = int(sx), int(sy), int(ox), int(oy)

    obs = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < W and 0 <= y < H:
                obs.add((x, y))

    res = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < W and 0 <= y < H and (x, y) not in obs:
                res.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def legal(nx, ny):
        return 0 <= nx < W and 0 <= ny < H and (nx, ny) not in obs

    def best_resource_score(px, py):
        if not res:
            return None
        # Priority: nearest resource; then prefer resources that are far from opponent (reduce contest).
        best = None
        for rx, ry in res:
            d_my = cheb(px, py, rx, ry)
            d_opp = cheb(ox, oy, rx, ry)
            # lower is better: strongly prefer being closer than opponent
            s = (d_my - d_opp) * 10 + d_my
            if best is None or s < best[0]:
                best = (s, rx, ry)
        return best[1], best[2]

    # If no resources, deterministically head to the farthest corner from opponent
    if not res:
        corners = [(0, 0), (0, H - 1), (W - 1, 0), (W - 1, H - 1)]
        tx, ty = max(corners, key=lambda c: cheb(ox, oy, c[0], c[1]))
        bx = ty  # tie-breaker anchor
        best_move = (0, 0)
        best = 10**9
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not legal(nx, ny):
                continue
            s = cheb(nx, ny, tx, ty)
            if s < best or (s == best and (dx, dy) < best_move):
                best = s
                best_move = (dx, dy)
        return [int(best_move[0]), int(best_move[1])]

    # If resources exist: choose move that (1) reduces distance to best contested target, (2) increases distance from opponent,
    # (3) discourages stepping into positions that are equally reachable by opponent.
    target = best_resource_score(sx, sy)
    tx, ty = target if target is not None else res[0]

    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not legal(nx, ny):
            continue
        d_my = cheb(nx, ny, tx, ty)
        d_opp = cheb(ox, oy, tx, ty)
        # Encourage becoming strictly closer than opponent; reward also being far from opponent.
        opp_dist = cheb(nx, ny, ox, oy)
        contest = (d_opp - d_my)
        val = contest * 100 + opp_dist * 2 - d_my
        # deterministic micro tie-break: prefer moves with smaller dx, then smaller dy, then staying still only if needed
        tieb = 0
        if (dx, dy) != (0, 0):
            tieb = (1 if dx != 0 else 0) + (1 if dy != 0 else 0)
        val -= tieb * 0.001
        if val > best_val or (val == best_val and (dx, dy) < best_move):
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
