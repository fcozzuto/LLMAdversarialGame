def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    s = observation.get("self_position") or [0, 0]
    o = observation.get("opponent_position") or [0, 0]
    sx, sy = int(s[0]), int(s[1])
    ox, oy = int(o[0]), int(o[1])

    obstacles = set()
    for b in observation.get("obstacles") or []:
        if b is not None and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                obstacles.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r is not None and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in obstacles:
                resources.append((rx, ry))

    def dist(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx + dy

    def valid(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # Choose target deterministically: prefer resources where we are closer than opponent.
    if resources:
        best = None
        for rx, ry in resources:
            myd = dist(sx, sy, rx, ry)
            opd = dist(ox, oy, rx, ry)
            lead = myd - opd  # negative means we are closer
            # Primary: minimize lead; Secondary: minimize my distance; Tertiary: lexicographic
            cand = (lead, myd, rx, ry)
            if best is None or cand < best:
                best = cand
        _, tx, ty = best
    else:
        # No visible resources: move toward center deterministically
        tx, ty = (w - 1) // 2, (h - 1) // 2

    # Evaluate each possible move: minimize distance to target, and avoid moving to cells where opponent improves too much.
    chosen = None
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        my_to_target = dist(nx, ny, tx, ty)
        opp_to_target = dist(ox, oy, tx, ty)
        my_to_opp = dist(nx, ny, ox, oy)  # deters walking into opponent
        # If opponent is closer to target, we still prefer moves that reduce our lead (myd - opd decreases)
        my_curr = dist(sx, sy, tx, ty)
        lead_now = my_curr - opp_to_target
        lead_after = my_to_target - opp_to_target
        cand = (lead_after, my_to_target, -my_to_opp, dx, dy)
        if chosen is None or cand < chosen:
            chosen = cand

    if chosen is None:
        return [0, 0]
    return [int(chosen[3]), int(chosen[4])]
