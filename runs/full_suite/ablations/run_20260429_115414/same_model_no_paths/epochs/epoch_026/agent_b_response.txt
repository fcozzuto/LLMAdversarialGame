def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    s = observation.get("self_position") or [0, 0]
    o = observation.get("opponent_position") or [0, 0]
    sx, sy = int(s[0]), int(s[1])
    ox, oy = int(o[0]), int(o[1])

    obstacles = set()
    for b in observation.get("obstacles") or []:
        if b is not None and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                obstacles.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r is not None and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in obstacles:
                resources.append((rx, ry))

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def bfs_dist(start, goal):
        if start == goal:
            return 0
        sx1, sy1 = start
        gx, gy = goal
        if (gx, gy) in obstacles or not (0 <= gx < w and 0 <= gy < h):
            return 10**9
        qx = [sx1]
        qy = [sy1]
        head = 0
        dist = [[-1] * h for _ in range(w)]
        dist[sx1][sy1] = 0
        while head < len(qx):
            x = qx[head]
            y = qy[head]
            d = dist[x][y]
            head += 1
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles and dist[nx][ny] == -1:
                    dist[nx][ny] = d + 1
                    if nx == gx and ny == gy:
                        return d + 1
                    qx.append(nx)
                    qy.append(ny)
        return 10**9

    def next_step_towards(cur, target):
        cx, cy = cur
        best = (10**9, 10**9, 0, 0)
        for dx, dy in dirs:
            nx, ny = cx + dx, cy + dy
            if not (0 <= nx < w and 0 <= ny < h) or (nx, ny) in obstacles:
                continue
            md = bfs_dist((nx, ny), target)
            # prefer smaller md; tie-break deterministically by direction then staying
            key = (md, abs(dx) + abs(dy), dx, dy)
            if key < (best[0], best[1], best[2], best[3]):
                best = (key[0], key[1], key[2], key[3])
        return [best[2], best[3]]

    if not resources:
        # retreat/hold: move toward opponent-less corner based on turn parity for determinism
        return [0, 0] if observation.get("turn_index", 0) % 2 == 0 else [1, 0]

    best_res = None
    best_key = None

        myd = bfs_dist((sx, sy), (rx, ry))
        opd = bfs_dist((ox, oy), (rx, ry))
        # maximize advantage (opponent slower than us), then closer to us
        adv = opd - myd
        key = (-adv, myd, rx, ry)
        if best_key is None or key < best_key:
            best_key = key
            best_res = (rx, ry)

    tx, ty = best
