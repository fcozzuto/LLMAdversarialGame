def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    me = tuple(observation["self_position"])
    opp = tuple(observation["opponent_position"])
    resources = observation.get("resources", []) or []
    obstacles = set(tuple(o) for o in (observation.get("obstacles", []) or []))

    def cheb(a, b):
        dx = a[0] - b[0]
        if dx < 0: dx = -dx
        dy = a[1] - b[1]
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    legal = []
    for dx, dy in dirs:
        nx, ny = me[0] + dx, me[1] + dy
        if inb(nx, ny):
            legal.append((dx, dy))
    if not legal:
        return [0, 0]

    if not resources:
        tx, ty = w // 2, h // 2
        bestm, bestv = (0, 0), None
        for dx, dy in legal:
            nx, ny = me[0] + dx, me[1] + dy
            v = cheb((nx, ny), (tx, ty))
            if bestv is None or v < bestv:
                bestv, bestm = v, (dx, dy)
        return [bestm[0], bestm[1]]

    clean_resources = [tuple(r) for r in resources if tuple(r) not in obstacles]
    if not clean_resources:
        return [0, 0]

    opp_nearest = None
    for r in clean_resources:
        d = cheb(opp, r)
        if opp_nearest is None or d < opp_nearest:
            opp_nearest = d

    bestm, bestscore = None, None
    for dx, dy in legal:
        nx, ny = me[0] + dx, me[1] + dy
        me2 = (nx, ny)
        # Try to pick the resource that maximizes how much we catch up / take away from opponent
        best_adv = None
        best_dm = None
        for r in clean_resources:
            dm = cheb(me2, r)
            do = cheb(opp, r)
            adv = do - dm
            # If opponent is strictly closer, heavily penalize that choice; focus on resources we can beat or nearly tie.
            if dm > do:
                adv -= 2 + (dm - do)
            if best_adv is None or adv > best_adv or (adv == best_adv and (best_dm is None or dm < best_dm)):
                best_adv, best_dm = adv, dm

        capture_bonus = 1000 if any(r == me2 for r in clean_resources) else 0
        center = (w // 2, h // 2)
        center_bias = -0.1 * cheb(me2, center)
        score = capture_bonus + best_adv * 10 - 0.5 * opp_nearest + center_bias
