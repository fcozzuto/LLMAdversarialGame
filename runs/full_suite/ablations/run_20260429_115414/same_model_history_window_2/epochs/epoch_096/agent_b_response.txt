def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if ok(rx, ry):
                resources.append((rx, ry))

    cheb = lambda x1, y1, x2, y2: (abs(x1 - x2) if abs(x1 - x2) > abs(y1 - y2) else abs(y1 - y2))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    legal = []
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if ok(nx, ny):
            legal.append((dx, dy, nx, ny))
    if not legal:
        return [0, 0]

    # If no resources, drift toward opponent to potentially block/contend.
    if not resources:
        best = None
        bestv = 10**9
        for dx, dy, nx, ny in legal:
            v = cheb(nx, ny, ox, oy)
            if v < bestv:
                bestv = v
                best = (dx, dy)
        return [int(best[0]), int(best[1])]

    best_move = None
    best_val = 10**18

    # Score a move by contested resource access after this move.
    # Prefer resources where we are at least as close as opponent, then minimize our distance.
    for dx, dy, nx, ny in legal:
        val = 0
        # Small tie-breakers for determinism
        val += 0.001 * (abs(dx) + abs(dy))
        # Evaluate best target
        chosen = None
        chosen_score = 10**18
        for rx, ry in resources:
            myd = cheb(nx, ny, rx, ry)
            opd = cheb(ox, oy, rx, ry)
            # Lead/contestation term
            # If opponent is closer, strongly discourage unless we can immediately take (myd==0).
            if myd == 0:
                score = -1000
            else:
                if opd < myd:
                    score = 500 + (myd - opd) * 25
                else:
                    score = myd * 10 - (opd - myd) * 5
            # Encourage targeting resources that are not behind obstacles-adjacent (soft)
            neigh_block = 0
            for ex in (-1, 0, 1):
                for ey in (-1, 0, 1):
                    if ex == 0 and ey == 0:
                        continue
                    tx, ty = rx + ex, ry + ey
                    if (tx, ty) in blocked:
                        neigh_block += 1
            score += 0.1 * neigh_block
            if score < chosen_score:
                chosen_score = score
                chosen = (rx, ry, myd, opd, score)
        val += chosen_score
        # If we're standing still, slightly worse unless it is already best to take a resource.
        if dx == 0 and dy == 0:
            val += 0.2
        # Deterministic tie-break: prefer smaller dx then smaller dy
        if val < best_val or (val == best_val and (dx, dy) < (best_move[0], best_move[1]) if best_move else True):
            best_val = val
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
