def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if ok(rx, ry):
                resources.append((rx, ry))

    if not resources:
        tx, ty = w // 2, h // 2
    else:
        best = None
        best_val = -10**9
        # choose a target that we can reach earlier than opponent, penalize ambiguity
        for rx, ry in resources:
            sd = cheb(sx, sy, rx, ry)
            od = cheb(ox, oy, rx, ry)
            lead = od - sd
            # small center bias to avoid edge traps
            cd = cheb(rx, ry, w // 2, h // 2)
            val = 5 * lead - sd - 0.1 * cd
            if val > best_val:
                best_val = val
                best = (rx, ry)
        tx, ty = best

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    candidates = []
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if ok(nx, ny):
            candidates.append((dx, dy, nx, ny))

    def opp_step(px, py):
        # deterministic: opponent greedily approaches the same target
        bestd = 10**9
        bestmove = (0, 0, px, py)
        for dx, dy in dirs:
            nx, ny = px + dx, py + dy
            if ok(nx, ny):
                d = cheb(nx, ny, tx, ty)
                if d < bestd:
                    bestd = d
                    bestmove = (dx, dy, nx, ny)
        return bestmove

    ox1 = ox
    oy1 = oy
    # evaluate each self move by its effect against predicted opponent response
    best_sc = -10**18
    best_move = (0, 0)
    for dx, dy, nsx, nsy in candidates:
        # predict opponent moves toward target after we move
        podx, pody, nosx, nosy = opp_step(ox1, oy1)
        myd = cheb(nsx, nsy, tx, ty)
        opd = cheb(nosx, nosy, tx, ty)
        # immediate capture bias and denial bias
        capture = 50 if (nsx == tx and nsy == ty) else 0
        deny = 6 * (opd - myd)  # we want opponent farther than us
        # avoid moving into positions that are strictly worse for us than opponent currently
        safety = -0.5 * (cheb(nsx, nsy, ox1, oy1) - cheb(sx, sy, ox1, oy1))
        sc = capture + deny - myd - 0.1 * (cheb(nsx, nsy, w - 1, h - 1))
        if sc > best_sc:
            best_sc = sc
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
