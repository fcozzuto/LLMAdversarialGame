def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in blocked:
                resources.append((rx, ry))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # Materially different from "race-to-closest": balance (capture progress) vs (denying proximity to opponent).
    # Deterministic tie-break by move order.
    best_mv = (0, 0)
    best_sc = -10**18

    on_res = (sx, sy) in set(resources)
    res_set = set(resources)

    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny):
            continue

        score = 0

        # Immediate pickup incentive
        if (nx, ny) in res_set:
            score += 10_000
        elif on_res:
            score += 150  # already on a resource, prefer staying

        # Capture progress: prefer reducing distance to the closest resource,
        # but if opponent is much closer, we bias toward denial.
        if resources:
            d_self = 10**9
            d_opp = 10**9
            for rx, ry in resources:
                ds = cheb(nx, ny, rx, ry)
                if ds < d_self:
                    d_self = ds
                do = cheb(ox, oy, rx, ry)
                if do < d_opp:
                    d_opp = do
            # baseline: move toward resources
            score += 900 - 40 * d_self
            # denial: keep away from opponent and/or prevent them from being the favored capturer
            if d_opp <= d_self:
                score -= 250  # likely opponent is at least as good at capturing
            else:
                score += 120
        else:
            # No visible resources: head to center to reduce opponent advantage swings
            cx, cy = w // 2, h // 2
            score += 80 - 10 * cheb(nx, ny, cx, cy)

        # Deny opponent: maximize distance after move
        score += 35 * cheb(nx, ny, ox, oy)

        # Small obstacle-pressure avoidance: penalize moves adjacent to obstacles too much
        if (dx, dy) != (0, 0):
            adj_pen = 0
            for ax, ay in moves:
                tx, ty = nx + ax, ny + ay
                if 0 <= tx < w and 0 <= ty < h and (tx, ty) in blocked:
                    adj_pen += 1
            score -= 15 * adj_pen

        if score > best_sc:
            best_sc = score
            best_mv = (dx, dy)

    return [int(best_mv[0]), int(best_mv[1])]
