def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position", [0, 0]) or [0, 0]
    op = observation.get("opponent_position", [w - 1, h - 1]) or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in blocked:
                resources.append((rx, ry))

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    cx, cy = (w - 1) // 2, (h - 1) // 2

    if not resources:
        best = (0, 0)
        bestd = -1
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if ok(nx, ny):
                d = -cheb(nx, ny, cx, cy)
                if d > bestd:
                    bestd = d
                    best = (dx, dy)
        return [best[0], best[1]]

    # Opponent-aware: prefer resources where we can close the gap and/or leave opponent farther behind.
    def resource_score(px, py, rx, ry):
        du = cheb(px, py, rx, ry)
        do = cheb(ox, oy, rx, ry)
        # Big weight on making opponent farther; then reward faster arrival.
        return (do - du) * 10 - du - (rx + ry) * 0.001

    # Choose our step that maximizes best available resource score from the new position.
    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue
        # If stuck by obstacle, staying still may be best; evaluation remains deterministic.
        val = -10**18
        # Evaluate only top few resources deterministically: use heuristic ordering without randomness.
        # Sort by (our dist, opponent dist) locally by scanning.
        cand = []
        for rx, ry in resources:
            du = cheb(nx, ny, rx, ry)
            do = cheb(ox, oy, rx, ry)
            cand.append((du, do, rx, ry))
        # Deterministic selection of first 5 by repeated min (no imports).
        for _ in range(5 if len(cand) > 5 else len(cand)):
            mi = 0
            for i in range(1, len(cand)):
                if cand[i][:2] < cand[mi][:2]:
                    mi = i
            du, do, rx, ry = cand.pop(mi)
            s = resource_score(nx, ny, rx, ry)
            if s > val:
                val = s
        # Small tie-break: prefer moves that reduce cheb to opponent as a secondary pressure signal.
        tie = -cheb(nx, ny, ox, oy) * 0.01
        val += tie
        if val > best_val or (val == best_val and (dx, dy) < best_move):
            best_val = val
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
