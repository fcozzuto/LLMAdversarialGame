def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or (0, 0)
    op = observation.get("opponent_position") or (0, 0)
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for p in observation.get("obstacles") or []:
        if p and len(p) >= 2:
            bx, by = int(p[0]), int(p[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in blocked:
                resources.append((rx, ry))

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    # Deterministic "prefer resources, and if possible, get closer than opponent"
    if resources:
        best_move = (0, 0)
        best_score = -10**18
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if not ok(nx, ny):
                continue
            # If we land on a resource, take it (dominant)
            landed_bonus = 5_000 if (nx, ny) in resources else 0
            # Evaluate against top few closest resources
            scored = []
            for tx, ty in resources:
                ds = cheb(nx, ny, tx, ty)
                do = cheb(ox, oy, tx, ty)
                # positive if we're nearer than opponent (steal potential)
                rel = do - ds
                # closer resources matter more
                scored.append((rel, -ds, tx, ty))
            scored.sort(reverse=True)
            # Combine: steal potential + closeness, with mild safety and anti-stagnation
            top = scored[:3]
            steal = 0
            close = 0
            far_pen = 0
            for rel, negds, tx, ty in top:
                ds = -negds
                steal += rel * 120
                close += (16 - ds) * 12
                far_pen += ds * 1
            # encourage progress: slightly prefer moves that reduce our nearest distance
            d0 = min(cheb(sx, sy, tx, ty) for tx, ty in resources)
            d1 = min(cheb(nx, ny, tx, ty) for tx, ty in resources)
            prog = (d0 - d1) * 20
            # avoid moving away from opponent too much only when contested (keeps pressure)
            opp_dist = cheb(nx, ny, ox, oy)
            pressure = (-opp_dist) * 1
            score = landed_bonus + steal + close + prog + pressure - far_pen
            if score > best_score:
                best_score = score
                best_move = (dx, dy)
        return [int(best_move[0]), int(best_move[1])]

    # No resources visible: move to center-ish, while keeping some pressure on opponent
    tx, ty = (w - 1) // 2, (h - 1) // 2
    best_move = (0, 0)
    best_score = -10**18
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue
        dcen = cheb(nx, ny, tx, ty)
        dopp = cheb(nx, ny, ox, oy)
        score = -dcen * 10 - dopp * 1
        if score > best_score:
            best_score = score
            best_move = (dx, dy)
    return [int(best_move[0]), int(best_move[1])]
