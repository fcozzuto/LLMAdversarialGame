def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [0, 0]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in blocked:
                resources.append((rx, ry))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def cheb(ax, ay, bx, by):
        dx = ax - bx
        if dx < 0:
            dx = -dx
        dy = ay - by
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    def onres(x, y):
        for rx, ry in resources:
            if rx == x and ry == y:
                return True
        return False

    # If no resources are listed, just move toward opponent's corner (opposite bias).
    if not resources:
        tx, ty = 0, h - 1
        if sx == w - 1 and sy == h - 1:
            tx, ty = 0, 0
        best = (0, 0)
        bestd = 10**9
        for dx, dy in moves:
            nx, ny = sx + dx, sy + dy
            if 0 <= nx < w and 0 <= ny < h and (nx, ny) not in blocked:
                d = cheb(nx, ny, tx, ty)
                if d < bestd:
                    bestd = d
                    best = (dx, dy)
        return [best[0], best[1]]

    best_move = (0, 0)
    best_val = -10**18
    # Deterministic tie-break: prefer staying only if equal; otherwise smallest (dx,dy) lexicographically.
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not (0 <= nx < w and 0 <= ny < h) or (nx, ny) in blocked:
            continue

        # Compute how much closer we are than the opponent to the best target resource.
        # Higher is better; prefer immediate collection strongly.
        val = 0
        if onres(nx, ny):
            val += 10**6

        # Also prevent "walking into" opponent's immediate vicinity.
        opp_close = cheb(nx, ny, ox, oy)
        val -= 8 * opp_close

        # Choose best resource target according to lead over opponent.
        local_best = -10**18
        for rx, ry in resources:
            ds = cheb(nx, ny, rx, ry)
            do = cheb(ox, oy, rx, ry)
            lead = do - ds  # positive means we are closer (or opponent farther)
            # Encourage shorter routes and safer spacing from obstacles via slight proximity penalty.
            # (Obstacle avoidance is already via blocked cells; this is just a weak tie-break.)
            dist_pen = ds
            score = 50 * lead - dist_pen
            if score > local_best:
                local_best = score
        val += local_best

        # Extra bias to keep advancing away from center dead-ends deterministically.
        cx, cy = (w - 1) / 2.0, (h - 1) / 2.0
        # Prefer moves that increase distance from center slightly (helps avoid cycles).
        distc = cheb(nx, ny, int(cx), int(cy))
        val += distc

        if val > best_val:
            best_val = val
            best_move = (dx, dy)
        elif val == best_val:
            # Lexicographic tie-break
            if (dx, dy) < best_move:
                best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
