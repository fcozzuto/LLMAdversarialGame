def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    def valid(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if valid(rx, ry):
                resources.append((rx, ry))

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    best_move = (0, 0)
    best_score = -10**30

    # Precompute nearest resource distances from current position
    my_now_best = 10**9
    for rx, ry in resources:
        d = cheb(sx, sy, rx, ry)
        if d < my_now_best:
            my_now_best = d
    toward_center = -cheb(sx, sy, w // 2, h // 2)

    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue

        # Resource pressure: prioritize reducing distance to best target; also prefer targets
        # closer (or equal) for us than for opponent to break ties deterministically.
        if resources:
            best_t = None
            best_t_score = 10**18
            for rx, ry in resources:
                dm = cheb(nx, ny, rx, ry)
                if dm < best_t_score:
                    best_t_score = dm
                    best_t = (rx, ry)
            rx, ry = best_t
            my_d = cheb(nx, ny, rx, ry)
            op_d = cheb(ox, oy, rx, ry)

            # Strong preference to capture/lead; slight preference to keep moving toward/away
            # depending on contest strength.
            capture_bias = 0
            if my_d == 0:
                capture_bias = 10**8
            lead_bonus = 0
            if my_d <= op_d:
                # Closer than opponent is good; tie-break favors ours.
                lead_bonus = 2000 - 10 * my_d
            else:
                # If opponent closer, still move closer but penalize more.
                lead_bonus = -3000 + 5 * op_d

            # Encourage improvement over current best distance to avoid dithering.
            improvement = my_now_best - my_d

            score = ( -my_d * 80 ) + lead_bonus + improvement * 200 + capture_bias
        else:
            # If no resources visible, just approach center while staying away from opponent.
            score = toward_center * 10 - cheb(nx, ny, ox, oy) * 2

        # Obstacle proximity avoidance: discourage being adjacent to obstacles (soft).
        adj_pen = 0
        for bx, by in blocked:
            d = cheb(nx, ny, bx, by)
            if d <= 1 and d != 0:
                adj_pen += 30
        score -= adj_pen

        if score > best_score:
            best_score = score
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
