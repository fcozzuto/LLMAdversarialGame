def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    blocked = set()
    for b in observation.get("obstacles") or []:
        if b and len(b) >= 2:
            bx, by = int(b[0]), int(b[1])
            if 0 <= bx < w and 0 <= by < h:
                blocked.add((bx, by))

    resources = []
    for r in observation.get("resources") or []:
        if r and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if 0 <= rx < w and 0 <= ry < h and (rx, ry) not in blocked:
                resources.append((rx, ry))

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def ok(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in blocked

    # No resource: keep away from opponent and drift to center.
    if not resources:
        cx, cy = (w - 1) / 2.0, (h - 1) / 2.0
        best = (-10**18, (0, 0))
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if not ok(nx, ny):
                continue
            val = cheb(nx, ny, int(ox), int(oy)) * 1.2 - cheb(nx, ny, cx, cy) * 0.25
            if val > best[0]:
                best = (val, (dx, dy))
        return [best[1][0], best[1][1]]

    # Choose move that best "wins" a resource now (self closer, opponent farther).
    best_target = resources[0]
    best_td = 10**9
    for t in resources:
        d = cheb(sx, sy, t[0], t[1])
        if d < best_td:
            best_td = d
            best_target = t

    # If there are multiple, consider only a small deterministic subset near the best target.
    resources_sorted = []
    for t in resources:
        resources_sorted.append((cheb(sx, sy, t[0], t[1]), t))
    resources_sorted.sort(key=lambda z: (z[0], z[1][0], z[1][1]))
    candidates = [t for _, t in resources_sorted[:6]]

    best_val = -10**18
    best_move = (0, 0)
    opp_w = 0.95
    self_w = 1.15
    block_bias = -0.08
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not ok(nx, ny):
            continue
        # Prefer moving toward a target while ensuring opponent is not equally close.
        val = 0.0
        for tx, ty in candidates:
            sd = cheb(nx, ny, tx, ty)
            od = cheb(ox, oy, tx, ty)
            # self priority + "win" margin (bigger margin is better)
            val += (self_w * (-sd)) + (opp_w * (od - sd))
        # Small preference for not clustering behind obstacles/edges is implicit via cheb and ok.
        val += block_bias * (0 if (nx, ny) in blocked else 0)
        # Add slight tie-break toward chosen closest-to-start target.
        val += -0.02 * cheb(nx, ny, best_target[0], best_target[1])
        if val > best_val:
            best_val = val
            best_move = (dx, dy)
