        if p and len(p) >= 2:

                blocked.add((x, y))

                resources.append((x, y))

        dx = ax - bx
        if dx < 0: dx = -dx
        dy = ay - by
        if dy < 0: dy = -dy
        return dx if dx > dy else dy
