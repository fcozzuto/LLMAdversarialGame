def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    resources = observation.get("resources") or []
    obstacles = observation.get("obstacles") or []
    blocked = set((p[0], p[1]) for p in obstacles)

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def legal(x, y):
        return inb(x, y) and (x, y) not in blocked

    def cd(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    viable = []
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if legal(nx, ny):
            viable.append((dx, dy, nx, ny))
    if not viable:
        return [0, 0]

    cx, cy = (w - 1) // 2, (h - 1) // 2
    if not resources:
        best = None
        bestv = -10**9
        for dx, dy, nx, ny in viable:
            myc = cd(nx, ny, cx, cy)
            odis = cd(nx, ny, ox, oy)
            v = -myc * 2 - odis * 0.05
            if v > bestv or (v == bestv and (dx, dy) < best):
                bestv = v
                best = (dx, dy)
        return [best[0], best[1]]

    # Precompute opponent best (one-step) distance to each resource
    opp_moves = [(dx, dy) for dx, dy in moves if legal(ox + dx, oy + dy)]
    opp_nextd = {}
    for rx, ry in resources:
        if not legal(rx, ry):
            continue
        md = 10**9
        for dx, dy in opp_moves:
            nx, ny = ox + dx, oy + dy
            d = cd(nx, ny, rx, ry)
            if d < md:
                md = d
        opp_nextd[(rx, ry)] = md

    best_move = (0, 0)
    bestv = -10**18
    for dx, dy, nx, ny in viable:
        v = 0
        my_step_center = -cd(nx, ny, cx, cy) * 0.02
        v += my_step_center
        best_res = None
        best_res_score = -10**18
        for r in resources:
            rx, ry = r[0], r[1]
            if not legal(rx, ry):
                continue
            myd = cd(nx, ny, rx, ry)
            od_next = opp_nextd.get((rx, ry), 10**9)
            capture = 1 if (nx == rx and ny == ry) else 0
            # advantage: prefer resources opponent is farther from (after their next move)
            adv = (od_next - myd)
            # penalize resources opponent can likely take immediately
            block = 4 if od_next <= 1 else 0
            score = adv * 1.2 - myd * 0.08 - block + capture * 8
            if score > best_res_score:
                best_res_score = score
                best_res = (rx, ry)
        v += best_res_score
        if v > bestv or (v == bestv and (dx, dy) < best_move):
            bestv = v
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
