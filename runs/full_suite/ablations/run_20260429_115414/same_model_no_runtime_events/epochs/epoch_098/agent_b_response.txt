def choose_move(observation):
    w = int(observation.get("grid_width") or 8)
    h = int(observation.get("grid_height") or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))
    if not resources:
        resources = []

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0: dx = -dx
        dy = y1 - y2
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def best_for(posx, posy):
        if not resources:
            return 0, None
        bestd = 10**9
        best = None
        for rx, ry in resources:
            d = cheb(posx, posy, rx, ry)
            if d < bestd:
                bestd = d
                best = (rx, ry)
        return bestd, best

    def valid(nx, ny):
        return 0 <= nx < w and 0 <= ny < h and (nx, ny) not in obstacles

    # If we can reach a resource immediately, do it deterministically.
    myd, target = best_for(sx, sy)
    if target is not None and myd == 0:
        # Prefer to stay if already on resource; otherwise move (but engine should handle pickup).
        return [0, 0]

    current_opd, current_opp_t = best_for(ox, oy)

    best = None
    best_key = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue

        myd2, myt2 = best_for(nx, ny)

        # Blocking: reduce opponent's distance to the best remaining resource after their next move (approx by current move).
        # Since we can't predict their exact move deterministically, use current opponent distance delta.
        opd2, oppt2 = best_for(ox + (1 if ox < sx else (-1 if ox > sx else 0)),
                                 oy + (1 if oy < sy else (-1 if oy > sy else 0)))

        # Immediate capture preference.
        onres = 1 if (myt2 is not None and myd2 == 0) else 0

        # Extra deterministic tie-breaker: align movement toward target and away from opponent.
        if myt2 is not None:
            rx, ry = myt2
            align = -(abs(nx - rx) + abs(ny - ry))
        else:
            align = 0

        # Primary: maximize capture/arrival lead (negative distances are good). Secondary: minimize our distance; maximize opponent disruption.
        # Key uses integers only.
        key = (
            onres,                     # capture now
            -myd2,                     # closer is better
            -(opd2 - (current_opd if current_opp_t is not None else 0)),  # reduce opponent's best distance
            align,                     # deterministic tie-break
            -cheb(nx, ny, ox, oy)      # keep some pressure/spacing deterministically
        )
        if best_key is None or key > best_key:
            best_key = key
            best = (dx, dy)

    if best is None:
        return [0, 0]
    return [int(best[0]), int(best[1])]

    if not resources:

    opp_closer_bias = 2  # prefer resources opponent isn't already winning
    best_r = None
    best_key = None
    for rx, ry in resources:
        ds = cheb(sx, sy, rx, ry)
        do = cheb(ox, oy, rx, ry)
        key = (do - ds, ds, rx, ry)
        if best_key is None or key < best_key:
            best_key = key
            best_r = (rx, ry)

    rx, ry = best_r
    if resources and best_key is not None:
        # if opponent clearly closer to the chosen target, retarget to next best

            alt_r = None
            for ax, ay in resources:
                ds = cheb(sx, sy, ax, ay)
                do = cheb(ox, oy, ax, ay)
                if do - ds >= opp_closer_bias:
                    continue
                key = (ds, ax, ay)
                if alt_key is None or key < alt_key:
                    alt_key = key
                    alt_r = (ax, ay)
            if alt_r is not None:
                rx, ry = alt_r

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
