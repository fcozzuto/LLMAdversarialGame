def choose_move(observation):
    w = int(observation.get("grid_width", 8) or 8)
    h = int(observation.get("grid_height", 8) or 8)
    sp = observation.get("self_position") or [0, 0]
    op = observation.get("opponent_position") or [w - 1, h - 1]
    sx, sy = int(sp[0]), int(sp[1])
    ox, oy = int(op[0]), int(op[1])

    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h:
                obstacles.add((x, y))

    resources = []
    for p in observation.get("resources") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            x, y = int(p[0]), int(p[1])
            if 0 <= x < w and 0 <= y < h and (x, y) not in obstacles:
                resources.append((x, y))
    if not resources:
        resources = [(0, 0), (w - 1, h - 1)]
        resources = [r for r in resources if r not in obstacles] or [(sx, sy)]

    def cheb(x1, y1, x2, y2):
        dx = x1 - x2
        if dx < 0:
            dx = -dx
        dy = y1 - y2
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    def valid(x, y):
        return 0 <= x < w and 0 <= y < h and (x, y) not in obstacles

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    # Pick target resource where we are relatively closer than the opponent.
    best_r = None
    best_sc = None
    for rx, ry in resources:
        my_d = cheb(sx, sy, rx, ry)
        op_d = cheb(ox, oy, rx, ry)
        sc = (op_d - my_d)
        if (best_sc is None or sc > best_sc or (sc == best_sc and (my_d < cheb(sx, sy, best_r[0], best_r[1]) or
           (my_d == cheb(sx, sy, best_r[0], best_r[1]) and (rx, ry) < best_r))))):
            best_sc = sc
            best_r = (rx, ry)
    tx, ty = best_r

    # Choose one-step move that maximizes (op_d - my_d_new) and secondarily minimizes my_d_new.
    best_m = (0, 0)
    best_val = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not valid(nx, ny):
            continue
        my_d2 = cheb(nx, ny, tx, ty)
        op_d = cheb(ox, oy, tx, ty)
        val = (op_d - my_d2) * 1000 - my_d2
        if best_val is None or val > best_val or (val == best_val and (dx, dy) < best_m):
            best_val = val
            best_m = (dx, dy)

    # If somehow all moves invalid, stay.
    return [int(best_m[0]), int(best_m[1])]
