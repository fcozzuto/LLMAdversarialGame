def choose_move(observation):
    w = observation["grid_width"]
    h = observation["grid_height"]
    sx, sy = observation["self_position"]
    ox, oy = observation["opponent_position"]
    obstacles = set(tuple(p) for p in observation.get("obstacles", []))
    resources = [tuple(r) for r in observation.get("resources", [])]

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def free(x, y):
        return inb(x, y) and (x, y) not in obstacles

    def man(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def best_resource_for(pos_self, pos_opp):
        if not resources:
            return None, None
        best_r = None
        best_key = None
        for rx, ry in resources:
            d_s = man(pos_self, (rx, ry))
            d_o = man(pos_opp, (rx, ry))
            # primary: maximize advantage vs opponent; secondary: reduce our distance
            key = (d_o - d_s, -d_s, -(rx + 13 * ry))
            if best_key is None or key > best_key:
                best_key = key
                best_r = (rx, ry)
        return best_r, best_key

    my_target, _ = best_resource_for((sx, sy), (ox, oy))
    if my_target is None:
        # no resources: move toward center while keeping distance from opponent
        tx, ty = w // 2, h // 2
        best_move = (0, 0)
        best_val = None
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if not free(nx, ny):
                continue
            v = (man((nx, ny), (tx, ty)), -man((nx, ny), (ox, oy)))
            if best_val is None or v < best_val:
                best_val = v
                best_move = (dx, dy)
        return [best_move[0], best_move[1]]

    tx, ty = my_target

    # predict opponent's immediate move toward its best target (one-step lookahead)
    opp_target, _ = best_resource_for((ox, oy), (sx, sy))
    if opp_target is None:
        opp_target = (sx, sy)

    def step_toward(px, py, target):
        best = (0, 0)
        best_key = None
        for dx, dy in dirs:
            nx, ny = px + dx, py + dy
            if not free(nx, ny):
                continue
            key = (man((nx, ny), target), man((nx, ny), (sx, sy)))
            if best_key is None or key < best_key:
                best_key = key
                best = (dx, dy)
        return best

    odx, ody = step_toward(ox, oy, opp_target)
    nox, noy = ox + odx, oy + ody

    # evaluate our candidate moves with lookahead against predicted opponent position
    best_move = (0, 0)
    best_score = None
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not free(nx, ny):
            continue
        d_self = man((nx, ny), (tx, ty))
        d_opp = man((nox, noy), (tx, ty))
        # If we can't secure the target, try to deny by getting closer relative to opponent
        advantage = d_opp - d_self
        # Also prioritize blocking-ish routes: avoid moving too close to opponent unless we gain advantage
        proximity = man((nx, ny), (nox, noy))
        # Deterministic tie-breaker by coordinates
        score = (advantage, -d_self, proximity, -(nx + 17 * ny))
        if best_score is None or score > best_score:
            best_score = score
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
