def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [0, 0])
    resources = observation.get("resources", []) or []
    obs = set(tuple(p) for p in (observation.get("obstacles", []) or []))

    dirs = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def free(x, y):
        return inb(x, y) and (x, y) not in obs

    def man(a, b, c, d):
        return abs(a - c) + abs(b - d)

    def best_resource_pref():
        if not resources:
            return None
        best = None
        for rx, ry in resources:
            sd = man(sx, sy, rx, ry)
            od = man(ox, oy, rx, ry)
            adv = od - sd  # positive means we are closer
            # Prefer taking resources we are closer to; otherwise minimize how much we're behind.
            if sd <= od:
                key = (2, adv, -sd, -(rx * 11 + ry))
            else:
                key = (1, -(-adv), -(sd - od), -sd, -(rx * 11 + ry))
            if best is None or key > best[0]:
                best = (key, (rx, ry))
        return best[1]

    target = best_resource_pref()

    # If no target, just drift toward center while avoiding obstacles/opponent collision point.
    if target is None:
        cx, cy = w // 2, h // 2
        best = None
        for dx, dy in dirs:
            nx, ny = sx + dx, sy + dy
            if not free(nx, ny):
                continue
            if (nx, ny) == (ox, oy):
                continue
            key = (-man(nx, ny, cx, cy), -(nx * 11 + ny))
            if best is None or key > best[0]:
                best = (key, (dx, dy))
        return best[1] if best else [0, 0]

    tx, ty = target

    # Evaluate moves by how much they improve our advantage on the chosen target,
    # with a secondary term favoring resources that the opponent is relatively farther from,
    # plus mild discouragement for moving adjacent to opponent (prevents easy contention loss).
    best = None
    for dx, dy in dirs:
        nx, ny = sx + dx, sy + dy
        if not free(nx, ny):
            continue
        if (nx, ny) == (ox, oy):
            continue

        sd2 = man(nx, ny, tx, ty)
        od = man(ox, oy, tx, ty)
        adv2 = od - sd2

        # Adjacent to opponent penalty (contested positions are risky when behind).
        opp_adj = (abs(nx - ox) <= 1 and abs(ny - oy) <= 1) and not (nx == ox and ny == oy)

        # Secondary: consider best (od - sd) among all resources after the move.
        sec = -10**9
        for rx, ry in resources:
            sd = man(nx, ny, rx, ry)
            otd = man(ox, oy, rx, ry)
            a = otd - sd
            if a > sec:
                sec = a

        # Primary: maximize advantage on target; tie-break by smaller distance to target,
        # then maximize secondary advantage; then deterministic tie by coordinates.
        key = (
            adv2,
            -sd2,
            sec,
            -int(opp_adj),
            -(nx * 11 + ny)
        )
        if best is None or key > best[0]:
            best = (key, (dx, dy))

    return list(best[1]) if best else [0, 0]
