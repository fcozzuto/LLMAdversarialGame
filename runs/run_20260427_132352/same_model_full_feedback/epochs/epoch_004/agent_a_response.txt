def choose_move(observation):
    w = observation.get("grid_width", 8)
    h = observation.get("grid_height", 8)
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [w - 1, h - 1])
    resources = observation.get("resources") or []
    obstacles = set(tuple(p) for p in (observation.get("obstacles") or []))
    cand = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(a, b):
        dx = a[0] - b[0]
        if dx < 0:
            dx = -dx
        dy = a[1] - b[1]
        if dy < 0:
            dy = -dy
        return dx if dx > dy else dy

    if not inb(sx, sy):
        return [0, 0]

    # If no resources visible, head toward the nearer corner of the opponent side.
    if not resources:
        tx, ty = (0, h - 1) if cheb((sx, sy), (0, h - 1)) < cheb((sx, sy), (w - 1, 0)) else (w - 1, 0)
        best = (0, 0)
        bestv = -10**18
        for dx, dy in cand:
            nx, ny = sx + dx, sy + dy
            if not inb(nx, ny) or (nx, ny) in obstacles:
                continue
            v = -cheb((nx, ny), (tx, ty)) + 0.02 * cheb((nx, ny), (ox, oy))
            if v > bestv:
                bestv = v
                best = (dx, dy)
        return [best[0], best[1]]

    # Build a couple of relevant resources: closest to us and to opponent.
    closest_r = None
    closest_do = None
    cd = 10**9
    co = 10**9
    for rx, ry in resources:
        if (rx, ry) in obstacles:
            continue
        d_us = cheb((sx, sy), (rx, ry))
        d_op = cheb((ox, oy), (rx, ry))
        if d_us < cd:
            cd = d_us
            closest_r = (rx, ry)
        if d_op < co:
            co = d_op
            closest_do = (rx, ry)

    # Move scoring: prefer positions that win more "next-collector" races.
    best_move = (0, 0)
    best_val = -10**18
    for dx, dy in cand:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obstacles:
            continue
        v = 0.0

        # Strong bonus for resources where we are strictly closer or equal than opponent.
        for rx, ry in resources:
            if (rx, ry) in obstacles:
                continue
            ds = cheb((nx, ny), (rx, ry))
            do = cheb((ox, oy), (rx, ry))
            if ds <= do:
                # Prefer the ones with small ds and large advantage over opponent.
                adv = do - ds
                v += 20.0 + 2.0 * adv - 0.9 * ds
            else:
                # Penalize moving into states that give opponent an even larger edge.
                adv = ds - do
                v -= 3.0 + 0.4 * adv + 0.15 * do

        # Add "contest" pressure: if opponent is much closer to a resource, reduce our distance to it.
        if closest_do is not None:
            v += 3.0 * (cheb((ox, oy), closest_do) - cheb((nx, ny), closest_do)) - 0.05 * cheb((nx, ny), closest_do)

        # Mild safety: avoid staying adjacent only if it doesn't improve contests.
        v += 0.1 * cheb((nx, ny), (ox, oy))

        # Prefer not to get stuck if many resources exist.
        if closest_r is not None:
            v += -0.12 * cheb((nx, ny), closest_r)

        if v > best_val:
            best_val = v
            best_move = (dx, dy)

    return [best_move[0], best_move[1]]
