def choose_move(observation):
    w = observation.get("grid_width", 8) or 8
    h = observation.get("grid_height", 8) or 8
    sx, sy = observation.get("self_position", [0, 0])
    ox, oy = observation.get("opponent_position", [w - 1, h - 1])
    resources = observation.get("resources") or []
    obstacles = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) == 2:
            obstacles.add((int(p[0]), int(p[1])))

    cand = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def cheb(a, b):
        dx = a[0] - b[0]
        if dx < 0: dx = -dx
        dy = a[1] - b[1]
        if dy < 0: dy = -dy
        return dx if dx > dy else dy

    def man(a, b):
        dx = a[0] - b[0]
        if dx < 0: dx = -dx
        dy = a[1] - b[1]
        if dy < 0: dy = -dy
        return dx + dy

    if not inb(sx, sy):
        return [0, 0]

    res = []
    for r in resources:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            rx, ry = int(r[0]), int(r[1])
            if inb(rx, ry):
                res.append((rx, ry))

    cur = (sx, sy)
    opp = (ox, oy)

    # Choose a contested target: resource with smallest (our_dist - their_dist)
    target = None
    best_comp = 10**18
    for r in res:
        if r in obstacles:
            continue
        d_self = cheb(cur, r)
        d_opp = cheb(opp, r)
        comp = d_self - d_opp
        if comp < best_comp:
            best_comp = comp
            target = r
        elif comp == best_comp and d_self < cheb(cur, target):
            target = r
    if target is None:
        # Fallback: move to increase separation while staying central-ish
        target = ((w - 1) // 2, (h - 1) // 2)

    # One-step lookahead: greedy toward target, but also block/avoid opponent approaches.
    best_move = (0, 0)
    best_score = -10**18
    for dx, dy in cand:
        nx, ny = sx + dx, sy + dy
        np = (nx, ny)
        if not inb(nx, ny) or np in obstacles:
            continue

        # Immediate resource value
        gain = 0
        if np in res:
            gain += 20

        # Approach target
        dist_t = cheb(np, target)
        approach = -2.0 * dist_t

        # Contest/deny: if opponent is closer to target than us, prefer moves that reduce that gap
        if res:
            d_self = cheb(np, target)
            d_opp = cheb(opp, target)
            deny = -1.2 * (d_self - d_opp)
        else:
            deny = 0

        # Maintain distance from opponent to reduce interference; but don't run when collecting
        dist_o = cheb(np, opp)
        sep = 0.35 * dist_o

        # Mild preference for staying away from edges unless targeting requires it
        edge = min(nx, w - 1 - nx, ny, h - 1 - ny)
        edge_pref = 0.02 * edge

        score = gain + approach + deny + sep + edge_pref

        if score > best_score:
            best_score = score
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]

def choose_move(observation):

    sp = observation.get("self_position", [0, 0])

    ox, oy = int(op[0]), int(op[1])

    obst = set()
    for p in observation.get("obstacles") or []:
        if isinstance(p, (list, tuple)) and len(p) >= 2:
            obst.add((int(p[0]), int(p[1])))

    def inb(x, y):
        return 0 <= x < w and 0 <= y < h

    def dist(a, b):
        dx = a[0] - b[0]
        if dx < 0: dx = -dx
        dy = a[1] - b[1]
        if dy < 0: dy = -dy
        return dx + dy

    resources = observation.get("resources") or []
    res = []
    for r in resources:
        if isinstance(r, (list, tuple)) and len(r) >= 2:
            x, y = int(r[0]), int(r[1])
            if inb(x, y) and (x, y) not in obst:
                res.append((x, y))

    moves = [(-1, -1), (-1, 0), (-1, 1), (0, -1), (0, 0), (0, 1), (1, -1), (1, 0), (1, 1)]
    if not inb(sx, sy):
        return [0, 0]

    best = None
    best_val = None
    for rx, ry in res:
        d1 = dist((sx, sy), (rx, ry))
        d2 = dist((ox, oy), (rx, ry))
        comp = d1 - d2
        if best is None or comp < best_val:
            best, best_val = (rx, ry), comp

    # If no resources, chase opponent slightly to prevent them scoring (still deterministic)
    target = best if best is not None else (ox, oy)

    best_move = (0, 0)
    best_score = None
    for dx, dy in moves:
        nx, ny = sx + dx, sy + dy
        if not inb(nx, ny) or (nx, ny) in obst:
            continue
        myd = dist((nx, ny), target)
        oppd = dist((ox, oy), target)
        # Prefer decreasing our distance, winning contested targets; break ties to stay deterministic.
        score = (myd - oppd, myd, -dx, -dy)
        if best_score is None or score < best_score:
            best_score = score
            best_move = (dx, dy)

    return [int(best_move[0]), int(best_move[1])]
